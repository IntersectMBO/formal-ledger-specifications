open import Ledger.Conway.Specification.Transaction

module Ledger.Conway.Specification.Ratify.Properties.Computational
  (txs : _) (open TransactionStructure txs) where

open import Ledger.Prelude
open import Ledger.Conway.Specification.Gov txs
open import Ledger.Conway.Specification.Enact govStructure
open import Ledger.Conway.Specification.Enact.Properties.Computational govStructure
open import Ledger.Conway.Specification.Ratify txs

open Computational ... hiding (computeProof; completeness)

private
  module Implementation
     (s : RatifyState) (sig : GovActionID  _)
    (let gid , st = sig)
    where
    open RatifyState s
    open RatifyEnv ; open GovActionState st
    es'  = compute $\begin{pmatrix} \,\href{Ledger.Conway.Specification.Ratify.Properties.Computational.html#742}{\htmlId{857}{\htmlClass{Bound}{\text{gid}}}}\, \\ \,\href{Ledger.Conway.Specification.Ratify.html#7983}{\htmlId{863}{\htmlClass{Function}{\text{treasury}}}}\, \\ \,\href{Ledger.Conway.Specification.Ratify.html#7868}{\htmlId{874}{\htmlClass{Function}{\text{currentEpoch}}}}\, \end{pmatrix}$ es action
    acc? = accepted?  es st
    exp? = expired? currentEpoch st
    del? = delayed? (action .gaType) prevAction es delay

    opaque
      acceptConds? :  a  Dec (acceptConds  s a)
      acceptConds? _ = Dec-   accepted? _ _ _ 
         Dec-  Dec-   delayed? _ _ _ _     ComputationalDec'   .dec

    RATIFY-total : [ s' ]   s  sig ,RATIFY s'
    RATIFY-total
      with acceptConds? sig | exp?
    ... | yes p@(_ , _ , (_ , q)) | _ = -, RATIFY-Accept (p , q)
    ... | no p | no a = -, RATIFY-Continue (p , a)
    ... | no p | yes a = -, RATIFY-Reject (p , a)

    computeProof = success {Err = } RATIFY-total

    RATIFY-completeness :  s'    s  sig ,RATIFY s'  RATIFY-total .proj  s'
    RATIFY-completeness st (RATIFY-Accept (p , a)) with acceptConds? sig
    ... | no h = -elim (h p)
    ... | yes (_ , _ , _ , h) = cong ( st  $\begin{pmatrix} \,\href{Ledger.Conway.Specification.Ratify.Properties.Computational.html#1776}{\htmlId{1784}{\htmlClass{Bound}{\text{st}}}}\, \\ \,\htmlId{1790}{\htmlClass{Symbol}{\text{\_}}}\, \\ \,\htmlId{1794}{\htmlClass{Symbol}{\text{\_}}}\, \end{pmatrix}$) $ computationalrightUnique Computational-ENACT h a
    RATIFY-completeness s' (RATIFY-Reject (p , a))
      rewrite dec-no (acceptConds? _) p | dec-yes exp? a .proj = refl
    RATIFY-completeness s' (RATIFY-Continue (p , a))
      rewrite dec-no (acceptConds? _) p | dec-no exp? a = refl

    completeness = cong (success {Err = })  RATIFY-completeness

instance
  Computational-RATIFY : Computational ___,RATIFY_ 
  Computational-RATIFY = record {Implementation}

Computational-RATIFIES : Computational ___,RATIFIES_ 
Computational-RATIFIES = it

RATIFIES-total :  { s sig}  [ s' ]   s  sig ,RATIFIES s'
RATIFIES-total = ReflexiveTransitiveClosure-total (Implementation.RATIFY-total _ _ _)

RATIFIES-complete :  { s sig s'} 
    s  sig ,RATIFIES s'  RATIFIES-total {} {s} {sig} .proj  s'
RATIFIES-complete = computationalrightUnique Computational-RATIFIES (RATIFIES-total .proj)

opaque
  RATIFIES-total' :  { s sig}  [ s' ]   s  sig ,RATIFIES s'
  RATIFIES-total' = RATIFIES-total

  RATIFIES-complete' :  { s sig s'} 
      s  sig ,RATIFIES s'  RATIFIES-total' {} {s} {sig} .proj  s'
  RATIFIES-complete' = RATIFIES-complete

  RATIFIES-deterministic :  { s sig s' s''}
       s  sig ,RATIFIES s'
       s  sig ,RATIFIES s''
     s'  s''
  RATIFIES-deterministic p p = trans (sym (RATIFIES-complete' p)) (RATIFIES-complete' p)

  RATIFIES-deterministic- :  { ' s s' sig sig' s'' s'''}
       '
     s  s'
     sig  sig'
        s   sig  ,RATIFIES s''
     '  s'  sig' ,RATIFIES s'''
     s''  s'''
  RATIFIES-deterministic- refl refl refl = RATIFIES-deterministic
