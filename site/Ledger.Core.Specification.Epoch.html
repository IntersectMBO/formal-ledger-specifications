module Ledger.Core.Specification.Epoch where

open import Ledger.Prelude hiding (compare; Rel)

open import Agda.Builtin.FromNat
open import Algebra using (Semiring)
open import Relation.Binary
open import Data.Nat.Properties using (+-*-semiring)
open import Data.Rational using ()
import      Data.Rational as 
import      Data.Rational.Properties as 

additionVia : {A : Set}  (A  A)    A  A
additionVia sucFun zero r = r
additionVia sucFun (suc l) r = sucFun (additionVia sucFun l r)

record EpochStructure : Type where
  field Slot : Semiring 0 0
        Epoch : Type;  DecEq-Epoch  : DecEq Epoch;  Show-Epoch  : Show Epoch

  Slot = Semiring.Carrier Slot

  field  DecPo-Slot    : HasDecPartialOrder {A = Slot}
         DecEq-Slot    : DecEq Slot

        epoch           : Slot  Epoch
        firstSlot       : Epoch  Slot
        StabilityWindow : Slot
        suc            : Epoch  Epoch

  _+_ = additionVia suc

  field
        _+'_           :   Epoch  Epoch
        ++'          :  {a b}  a + b  a +' b

  -- preorders and partial orders

  instance
    preoEpoch : HasPreorder
    preoEpoch = hasPreorderFromStrictPartialOrder {_<_ = _<_ on firstSlot}
      record
        { isEquivalence = isEquivalence
        ; irrefl =  where refl  <-irrefl {A = Slot} refl
        ; trans  = <-trans {A = Slot}
        ; <-resp- = ( where refl  id) , ( where refl  id)
        }

  _ = _<_ {A = Slot}    it
  _ = __ {A = Slot}    it
  _ = _<_ {A = Epoch}   it
  _ = __ {A = Epoch}   it

  -- addition

  open Semiring Slot renaming (_+_ to _+_)

  toEpoch :   Epoch
  toEpoch zero    = epoch 0#
  toEpoch (suc n) = suc (toEpoch n)

  instance
    addSlot : HasAdd Slot
    addSlot ._+_ = _+_

    addEpoch : HasAdd Epoch
    addEpoch ._+_ e e' = epoch (firstSlot e + firstSlot e')

    Number-Epoch : Number Epoch
    Number-Epoch .Number.Constraint _ = 
    Number-Epoch .Number.fromNat    x = toEpoch x

record GlobalConstants : Type where
  field  Network : Type;  DecEq-Netw  : DecEq Network;  Show-Network  : Show Network
         SlotsPerEpoch   : ;  NonZero-SlotsPerEpoch  : NonZero SlotsPerEpoch
         ActiveSlotCoeff  : ;  Positive-ActiveSlotCoeff  : .Positive ActiveSlotCoeff
         StabilityWindow : 
         Quorum : 
         NetworkId : Network

  instance
    NonZero-ActiveSlotCoeff : .NonZero ActiveSlotCoeff
    NonZero-ActiveSlotCoeff = .>-nonZero (.positive ActiveSlotCoeff)

  ++' :  {a b}  additionVia suc a b  a + b
  ++' {zero} {b} = refl
  ++' {suc a} {b} = cong suc (++' {a} {b})

  EpochStructure : EpochStructure
  EpochStructure =  where
    .Slot            +-*-semiring
    .Epoch            
    .epoch slot       slot / SlotsPerEpoch
    .firstSlot e      e * SlotsPerEpoch
    .StabilityWindow  StabilityWindow
    .suc             suc
    ._+'_            _+_
    .++' {a} {b}   ++' {a} {b}

   where open EpochStructure

open GlobalConstants using (EpochStructure) public
