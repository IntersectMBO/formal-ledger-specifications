instance
  Computational-UTXOS : Computational ___,UTXOS_ String
  Computational-UTXOS = record {go} where
    module go  s tx
      (let H-Yes ,  H-Yes? = Scripts-Yes-premises {} {s} {tx})
      (let H-No  ,  H-No?  = Scripts-No-premises {} {s} {tx}) where
      open Tx tx renaming (body to txb); open TxBody txb
      open UTxOEnv  renaming (pparams to pp)
      open UTxOState s
      sLst = collectPhaseTwoScriptInputs pp tx utxo

      computeProof =
        case H-Yes? , H-No? of  where
          (yes p , no _ )  success (_ , (Scripts-Yes p))
          (no _  , yes p)  success (_ , (Scripts-No p))
          (_     , _    )  failure "isValid check failed"

      completeness :  s'    s  tx ,UTXOS s'  map proj computeProof  success s'
      completeness _ (Scripts-Yes p) with H-No? | H-Yes?
      ... | yes (_ , refl) | _     = case proj p of  ()
      ... | no _           | yes _ = refl
      ... | no _           | no p = case p p of  ()
      completeness _ (Scripts-No p) with H-Yes? | H-No?
      ... | yes (_ , refl) | _     = case proj p of  ()
      ... | no _           | yes _ = refl
      ... | no _           | no p = case p p of  ()

instance
  Computational-UTXO : Computational ___,UTXO_ String
  Computational-UTXO = record {Go}
    where
      module Go  s tx (let H ,  H? = UTXO-premises {tx}{}{s}) where

        open Computational Computational-UTXOS
          renaming (computeProof to computeProof'; completeness to completeness')

        computeProofH : Dec H  ComputationResult String ([ s' ]   s  tx ,UTXO s')
        computeProofH (yes (x , y , z , e , k , l , m , c , v , j , n , o , p , q , r , t , u)) =
            map (UTXO-inductive _ _ _ x y z e k l m c v j n o p q r t u) <$> computeProof'  s tx
        computeProofH (no p) = failure $ genErrors p

        computeProof : ComputationResult String ([ s' ]   s  tx ,UTXO s')
        computeProof = computeProofH H?

        completeness :  s'    s  tx ,UTXO s'  map proj computeProof  success s'
        completeness s' (UTXO-inductive _ _ _ x y z e k l m c v j n o p q r t u h) with H?
        ... | no p = -elim $ p (x , y , z , e , k , l , m , c , v , j , n , o , p , q , r , t , u )
        ... | yes _ with computeProof'  s tx | completeness' _ _ _ _ h
        ... | success _ | refl = refl
