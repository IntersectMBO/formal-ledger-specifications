-- ** Proof that LEDGER is computational.

instance
  _ = Monad-ComputationResult

  Computational-LEDGER : Computational ___,LEDGER_ String
  Computational-LEDGER = record {go}
    where
    open Computational ... renaming (computeProof to comp; completeness to complete)
    computeUtxow = comp {STS = ___,UTXOW_}
    computeCerts = comp {STS = ___,CERTS_}
    computeGov   = comp {STS = ___,GOVS_}

    module go
      ( : LEnv)   (let open LEnv )
      (s : LState) (let open LState s)
      (tx : Tx)    (let open Tx tx renaming (body to txb); open TxBody txb)
      where
      utxo = UTxOEnv  record { LEnv  }
      cert = CertEnv  $\begin{pmatrix} \,\href{Ledger.Core.Specification.Epoch.html#818}{\htmlId{2652}{\htmlClass{Function}{\text{epoch}}}}\, \,\href{Ledger.Conway.Specification.Ledger.html#1496}{\htmlId{2658}{\htmlClass{Field}{\text{slot}}}}\, \\ \,\href{Ledger.Conway.Specification.Ledger.html#1554}{\htmlId{2665}{\htmlClass{Field}{\text{pparams}}}}\, \\ \,\href{Ledger.Conway.Specification.Transaction.html#5184}{\htmlId{2675}{\htmlClass{Function}{\text{txvote}}}}\, \\ \,\href{Ledger.Conway.Specification.Transaction.html#5156}{\htmlId{2684}{\htmlClass{Function}{\text{txwdrls}}}}\, \\ \,\htmlId{2694}{\htmlClass{Symbol}{\text{\_}}}\, \end{pmatrix}$
      gov : CertState  GovEnv
      gov certState = $\begin{pmatrix} \,\href{Ledger.Conway.Specification.Transaction.html#5431}{\htmlId{2755}{\htmlClass{Function}{\text{txid}}}}\, \\ \,\href{Ledger.Core.Specification.Epoch.html#818}{\htmlId{2762}{\htmlClass{Function}{\text{epoch}}}}\, \,\href{Ledger.Conway.Specification.Ledger.html#1496}{\htmlId{2768}{\htmlClass{Field}{\text{slot}}}}\, \\ \,\href{Ledger.Conway.Specification.Ledger.html#1554}{\htmlId{2775}{\htmlClass{Field}{\text{pparams}}}}\, \\ \,\href{Ledger.Conway.Specification.Ledger.html#1519}{\htmlId{2785}{\htmlClass{Field}{\text{ppolicy}}}}\, \\ \,\href{Ledger.Conway.Specification.Ledger.html#1580}{\htmlId{2795}{\htmlClass{Field}{\text{enactState}}}}\, \\ \,\href{Ledger.Conway.Conformance.Equivalence.Convert.html#460}{\htmlId{2808}{\htmlClass{Function}{\text{conv}}}}\, \,\href{Ledger.Conway.Conformance.Ledger.Properties.html#2741}{\htmlId{2813}{\htmlClass{Bound}{\text{certState}}}}\, \\ \,\htmlId{2825}{\htmlClass{Symbol}{\text{\_}}}\, \end{pmatrix}$

      computeProof : ComputationResult String ([ s' ]   s  tx ,LEDGER s')
      computeProof = case isValid  true of  where
        (yes p)  do
          (utxoSt' , utxoStep)  computeUtxow utxo utxoSt tx
          (certSt' , certStep)  computeCerts cert certState txcerts
          (govSt'  , govStep)   computeGov (gov certSt') (rmOrphanDRepVotes (conv certSt') govSt) (txgov txb)
          success (_ , LEDGER-V p utxoStep certStep govStep)
        (no p)  do
          (utxoSt' , utxoStep)  computeUtxow utxo utxoSt tx
          success (_ , LEDGER-I (-not p) utxoStep)

      completeness :  s'    s  tx ,LEDGER s'  (proj <$> computeProof)  success s'
      completeness  ls' (LEDGER-V v utxoStep certStep govStep)
        with isValid  true
      ... | no v = contradiction v v
      ... | yes refl
        with computeUtxow utxo utxoSt tx | complete _ _ _ _ utxoStep
      ... | success (utxoSt' , _) | refl
        with computeCerts cert certState txcerts | complete _ _ _ _ certStep
      ... | success (certSt' , _) | refl
        with computeGov (gov certSt') (rmOrphanDRepVotes (conv certSt') govSt) (txgov txb) | complete {STS = ___,GOVS_} (gov certSt') _ _ _ govStep
      ... | success (govSt' , _) | refl = refl
      completeness ls' (LEDGER-I i utxoStep)
        with isValid  true
      ... | yes refl = case i of  ()
      ... | no v
        with computeUtxow utxo utxoSt tx | complete _ _ _ _ utxoStep
      ... | success (utxoSt' , _) | refl = refl
