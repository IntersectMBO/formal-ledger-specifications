-- ** Proof that LEDGER is computational.

instance
  _ = Monad-ComputationResult

  Computational-LEDGER : Computational ___,LEDGER_ String
  Computational-LEDGER = record {go}
    where
    open Computational ... renaming (computeProof to comp; completeness to complete)
    computeUtxow = comp {STS = ___,UTXOW_}
    computeCerts = comp {STS = ___,CERTS_}
    computeGov   = comp {STS = ___,GOVS_}

    module go
      ( : LEnv)   (let open LEnv )
      (s : LState) (let open LState s)
      (tx : Tx)    (let open Tx tx renaming (body to txb); open TxBody txb)
      where
      utxo = UTxOEnv  record { LEnv  }
      cert = CertEnv  $\begin{pmatrix} \,\href{Ledger.Core.Specification.Epoch.html#954}{\htmlId{2767}{\htmlClass{Function}{\text{epoch}}}}\, \,\href{Ledger.Conway.Specification.Ledger.html#1064}{\htmlId{2773}{\htmlClass{Field}{\text{slot}}}}\, \\ \,\href{Ledger.Conway.Specification.Ledger.html#1122}{\htmlId{2780}{\htmlClass{Field}{\text{pparams}}}}\, \\ \,\href{Ledger.Conway.Specification.Transaction.html#5232}{\htmlId{2790}{\htmlClass{Function}{\text{txGovVotes}}}}\, \\ \,\href{Ledger.Conway.Specification.Transaction.html#5057}{\htmlId{2803}{\htmlClass{Function}{\text{txWithdrawals}}}}\, \\ \,\htmlId{2819}{\htmlClass{Symbol}{\text{\_}}}\, \end{pmatrix}$
      gov : CertState  GovEnv
      gov certState = $\begin{pmatrix} \,\href{Ledger.Conway.Specification.Transaction.html#4949}{\htmlId{2880}{\htmlClass{Function}{\text{txId}}}}\, \\ \,\href{Ledger.Core.Specification.Epoch.html#954}{\htmlId{2887}{\htmlClass{Function}{\text{epoch}}}}\, \,\href{Ledger.Conway.Specification.Ledger.html#1064}{\htmlId{2893}{\htmlClass{Field}{\text{slot}}}}\, \\ \,\href{Ledger.Conway.Specification.Ledger.html#1122}{\htmlId{2900}{\htmlClass{Field}{\text{pparams}}}}\, \\ \,\href{Ledger.Conway.Specification.Ledger.html#1087}{\htmlId{2910}{\htmlClass{Field}{\text{ppolicy}}}}\, \\ \,\href{Ledger.Conway.Specification.Ledger.html#1148}{\htmlId{2920}{\htmlClass{Field}{\text{enactState}}}}\, \\ \,\href{Ledger.Conway.Conformance.Equivalence.Convert.html#559}{\htmlId{2933}{\htmlClass{Function}{\text{conv}}}}\, \,\href{Ledger.Conway.Conformance.Ledger.Properties.html#2866}{\htmlId{2938}{\htmlClass{Bound}{\text{certState}}}}\, \\ \,\htmlId{2950}{\htmlClass{Symbol}{\text{\_}}}\, \end{pmatrix}$

      computeProof : ComputationResult String ([ s' ]   s  tx ,LEDGER s')
      computeProof = case isValid  true of  where
        (yes p)  do
          (utxoSt' , utxoStep)  computeUtxow utxo utxoSt tx
          (certSt' , certStep)  computeCerts cert certState txCerts
          (govSt'  , govStep)   computeGov (gov certSt') (rmOrphanDRepVotes (conv certSt') govSt) (txgov txb)
          success (_ , LEDGER-V p utxoStep certStep govStep)
        (no p)  do
          (utxoSt' , utxoStep)  computeUtxow utxo utxoSt tx
          success (_ , LEDGER-I (-not p) utxoStep)

      completeness :  s'    s  tx ,LEDGER s'  (proj <$> computeProof)  success s'
      completeness  ls' (LEDGER-V v utxoStep certStep govStep)
        with isValid  true
      ... | no v = contradiction v v
      ... | yes refl
        with computeUtxow utxo utxoSt tx | complete _ _ _ _ utxoStep
      ... | success (utxoSt' , _) | refl
        with computeCerts cert certState txCerts | complete _ _ _ _ certStep
      ... | success (certSt' , _) | refl
        with computeGov (gov certSt') (rmOrphanDRepVotes (conv certSt') govSt) (txgov txb) | complete {STS = ___,GOVS_} (gov certSt') _ _ _ govStep
      ... | success (govSt' , _) | refl = refl
      completeness ls' (LEDGER-I i utxoStep)
        with isValid  true
      ... | yes refl = case i of  ()
      ... | no v
        with computeUtxow utxo utxoSt tx | complete _ _ _ _ utxoStep
      ... | success (utxoSt' , _) | refl = refl
