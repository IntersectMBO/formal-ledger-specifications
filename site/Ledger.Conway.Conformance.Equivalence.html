lem-cert-deposits-valid :  { s tx s'} (open L.LEnv  using (pparams))

lem-cert-deposits-valid {} {s} {tx} {s'} refl (L.LEDGER-V refl utxow certs gov) rewrite sym (lemUpdateDeposits refl utxow) =
  lem-upd-ddeps pparams deps tx ,
  lem-upd-gdeps pparams deps tx

instance
  LEDGERToConf :  { s tx s'}
                 L. s  tx ,LEDGER s' 
                 [ certDeposits-s' ]
                    certDeposits-s'  certDeposits s'
                    C. (certDeposits s conv s)  tx ,LEDGER (certDeposits-s' conv s')
  LEDGERToConf {} {s} {tx} {s'} .conv _ r@(L.LEDGER-V refl utxow certs gov) =
    updateLedgerDeps pparams tx (certDeposits s)
    , lem-cert-deposits-valid refl r
    , subst (     C. getCertDeps* cdeposits conv s  tx ,LEDGER $\begin{pmatrix} \,\href{Ledger.Conway.Conformance.Equivalence.html#5961}{\htmlId{6021}{\htmlClass{Bound}{\text{}}}}\, \\ \,\htmlId{6025}{\htmlClass{Symbol}{\text{\_}}}\, \\ \,\href{Ledger.Conway.Conformance.Equivalence.html#5963}{\htmlId{6029}{\htmlClass{Bound}{\text{}}}}\, \end{pmatrix}$)
             utxoEq certsEq ledger'
    where
      open L.LEnv 
      open L.LState s
      open L.LState s' renaming (utxoSt to utxoSt'; certState to certState'; govSt to govSt')
      open TxBody (body tx) using (txCerts)
      deposits = L.UTxOState.deposits utxoSt
      utxow' : _ C. utxoSt  tx ,UTXOW setDeposits deposits utxoSt'
      utxow' = conv utxow
      utxoStC'    = setDeposits (L.updateDeposits pparams (body tx) deposits) utxoSt'
      cdeposits  = makeCertDeps* refl utxow
      cdeposits' = updateCertDeps* txCerts cdeposits
      certStateC' = getCertDeps* cdeposits' conv certState'
      certs' : _ C. (getCertDeps* cdeposits conv certState)  txCerts ,CERTS certStateC'
      certs' = cdeposits conv certs
      ledger' :  C. (getCertDeps* cdeposits conv s)  tx ,LEDGER C. utxoStC' , govSt' , certStateC' 
      ledger' = C.LEDGER-V refl utxow' certs' gov
      utxoEq  : utxoStC'  utxoSt'
      utxoEq  = cong (   $\begin{pmatrix} \,\htmlId{7001}{\htmlClass{Symbol}{\text{\_}}}\, \\ \,\htmlId{7005}{\htmlClass{Symbol}{\text{\_}}}\, \\ \,\href{Ledger.Conway.Conformance.Equivalence.html#6995}{\htmlId{7009}{\htmlClass{Bound}{\text{}}}}\, \\ \,\htmlId{7013}{\htmlClass{Symbol}{\text{\_}}}\, \end{pmatrix}$)
                     (lemUpdateDeposits refl utxow)
      ddeps = getCertDeps* cdeposits .proj
      gdeps = getCertDeps* cdeposits .proj
      certsEq : certStateC'  (updateDDeps pparams txCerts ddeps , updateGDeps pparams txCerts gdeps) conv certState'
      certsEq = cong (    $\begin{pmatrix} \begin{pmatrix} \,\htmlId{7312}{\htmlClass{Symbol}{\text{\_}}}\, \\ \,\htmlId{7316}{\htmlClass{Symbol}{\text{\_}}}\, \\ \,\htmlId{7320}{\htmlClass{Symbol}{\text{\_}}}\, \\ \,\href{Ledger.Conway.Conformance.Equivalence.html#7302}{\htmlId{7324}{\htmlClass{Bound}{\text{}}}}\, \end{pmatrix} \\ \,\htmlId{7330}{\htmlClass{Symbol}{\text{\_}}}\, \\ \begin{pmatrix} \,\htmlId{7336}{\htmlClass{Symbol}{\text{\_}}}\, \\ \,\htmlId{7340}{\htmlClass{Symbol}{\text{\_}}}\, \\ \,\href{Ledger.Conway.Conformance.Equivalence.html#7304}{\htmlId{7344}{\htmlClass{Bound}{\text{}}}}\, \end{pmatrix} \end{pmatrix}$)
                     (lem-ddeps cdeposits)
                     (lem-gdeps cdeposits)

  -- LEDGERToConf {} {s} {tx} {s'} .conv (cdeps , cdeps-eq) r@(L.LEDGER-I refl utxow) =

    subst (    C. _ conv s  tx ,LEDGER $\begin{pmatrix} \begin{pmatrix} \,\htmlId{7708}{\htmlClass{Symbol}{\text{\_}}}\, \\ \,\htmlId{7712}{\htmlClass{Symbol}{\text{\_}}}\, \\ \,\href{Ledger.Conway.Conformance.Equivalence.html#7669}{\htmlId{7716}{\htmlClass{Bound}{\text{}}}}\, \\ \,\htmlId{7720}{\htmlClass{Symbol}{\text{\_}}}\, \end{pmatrix} \\ \,\htmlId{7726}{\htmlClass{Symbol}{\text{\_}}}\, \\ \,\htmlId{7730}{\htmlClass{Symbol}{\text{\_}}}\, \end{pmatrix}$)

  LEDGERFromConf {} {s} {tx} {s'} .conv wf (C.LEDGER-V refl utxow certs gov) =
    subst (    L. conv s  tx ,LEDGER $\begin{pmatrix} \,\href{Ledger.Conway.Conformance.Equivalence.html#12281}{\htmlId{12315}{\htmlClass{Bound}{\text{}}}}\, \\ \,\href{Ledger.Conway.Conformance.Equivalence.html#12513}{\htmlId{12319}{\htmlClass{Function}{\text{govSt'}}}}\, \\ \,\href{Ledger.Conway.Conformance.Equivalence.Convert.html#559}{\htmlId{12328}{\htmlClass{Function}{\text{conv}}}}\, \,\href{Ledger.Conway.Conformance.Equivalence.html#12534}{\htmlId{12333}{\htmlClass{Function}{\text{certSt'}}}}\, \end{pmatrix}$) eqUtxo ledger'

      open TxBody (body tx)
      open C.UTxOState utxoSt using (deposits)

      utxow' : _ L. utxoSt  tx ,UTXOW (setDeposits (utxowDeposits utxow) utxoSt')

      eqUtxo : setDeposits (utxowDeposits utxow) utxoSt'  utxoSt'

                                              , govSt'
                                              , conv certSt'

lemWellformed :  { s tx s'}  WellformedLState s   C. s  tx ,LEDGER s'  WellformedLState s'
lemWellformed {} {s = ls} {tx} {s' = ls'} wf (C.LEDGER-V refl utxo certs gov) = goal
  where
    open C.LState ls  renaming (certState to certSt)
    open C.LState ls' renaming (utxoSt to utxoSt'; certState to certSt')
    open L.LEnv  using (pparams)
    open TxBody (body tx)

    deposits  = L.UTxOState.deposits utxoSt
    deposits' = L.UTxOState.deposits utxoSt'

    depositsL' = L.updateDeposits pparams (body tx) deposits

    ddeps = certDepositsC certSt .proj
    gdeps = certDepositsC certSt .proj

    ddeps' = certDepositsC certSt' .proj
    gdeps' = certDepositsC certSt' .proj

    lem : deposits'  depositsL'
    lem rewrite lemDepositsC utxo = refl

    lem : (ddeps' , gdeps')  (updateDDeps pparams txCerts ddeps , updateGDeps pparams txCerts gdeps)
    lem = lemCERTSDepositsC certs

    lem :  (updateDDeps pparams txCerts (certDDeps deposits) , updateGDeps pparams txCerts (certGDeps deposits))
          (certDDeps deposits' , certGDeps deposits')
    lem rewrite lem = lem-upd-ddeps pparams deposits tx , lem-upd-gdeps pparams deposits tx

    lem :  (updateDDeps pparams txCerts ddeps , updateGDeps pparams txCerts gdeps)
          (updateDDeps pparams txCerts (certDDeps deposits) , updateGDeps pparams txCerts (certGDeps deposits))
    lem =  cong-updateDDeps txCerts , cong-updateGDeps txCerts  wf

    goal : (ddeps' , gdeps')  (certDDeps deposits' , certGDeps deposits')
    goal with refl  lem = -trans {ddeps' , gdeps'}
                                     {updateDDeps pparams txCerts (certDDeps deposits) , updateGDeps pparams txCerts (certGDeps deposits)}
                                     {certDDeps deposits' , certGDeps deposits'}
                              lem
                              lem
lemWellformed wf (C.LEDGER-I refl utxo) rewrite lemDepositsC utxo = wf

open import Ledger.Conway.Conformance.Equivalence.Bisimilarity

-- An LState s from the Spec and an LState s from Conformance are
-- bisimular iff
--   the deposits in the CertState of s are equal (as maps) to the
--    appropriate restrictions of the deposits in the UTxOState of s
--   the only difference between s and s are the extra deposits in
--    the CertState of s.
__ : L.LState  C.LState  Type
s  s = certDeposits s  certDepositsC (C.LState.certState s)  s  conv s

-- This gives us the main result: a bisimulation between the LEDGER
-- rules in Ledger and in Conformance.
bisimilarityProof : Bisimilar L.___,LEDGER_ C.___,LEDGER_
bisimilarityProof .Bisimilar.__ = __
bisimilarityProof .Bisimilar.to {} {tx} {s} {s'} {s} (eq , refl) r =
  let deps  , eqd  , r'  = conv r
      deps' , eqd' , r'' = castLEDGER (certDeposits (conv s))
                                      (certDepositsC (C.LState.certState s))
                                      deps
                                      eq
                                      r'
      eqd'' : certDeposits s'  deps'
      eqd'' = -trans {certDeposits s'} {deps} {deps'} (-sym {deps} {certDeposits s'} eqd) eqd'
  in
  (deps' conv s') , (eqd'' , refl) , r''
bisimilarityProof .Bisimilar.from {} {tx} {s} {s} {s'} (eqd , refl) r =
  conv s' , (wf' , refl) , wf conv r
  where
    wf  = -sym {certDeposits (conv s)} {certDepositsC (C.LState.certState s)} eqd
    wf' = -sym {certDepositsC (C.LState.certState s')} {certDeposits (conv s')} (lemWellformed wf r)
    r' = wf conv r
