startTxOut : Value  PlutusScript  TxOut
startTxOut v script = inj (record { net = 0 ;
                           pay = ScriptObj (proj script) ;
                           stake = just (ScriptObj (proj script)) })
                           , v
                           , just (inj (inj (inj (Always [])))) , nothing
