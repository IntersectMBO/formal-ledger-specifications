-- Positive and negative part of integers
open import Data.Integer using (sign; _; __)
open import Data.Integer.Properties using ([1+m][1+n]mn)
open import Data.Sign using (Sign)

posPart :   
posPart x with sign x
... | Sign.+ =  x 
... | Sign.- = 0

negPart :   
negPart x with sign x
... | Sign.- =  x 
... | Sign.+ = 0

posPart : {m n : }  (m  n)  posPart (m  n)
posPart {zero} {zero} = __.refl
posPart {zero} {.suc n} = __.refl
posPart {.suc m} {zero} = __.refl
posPart {.suc m} {.suc n} = trans (posPart{m}{n}) (sym (cong posPart (([1+m][1+n]mn m n))))

instance
  Dec-NonZero :  {n}  NonZero n

  Dec-NonZero {suc n} .dec = yes (record { nonZero = tt })
