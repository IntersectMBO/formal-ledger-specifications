    The EPOCH transition system behaves parametrically w.r.t. the set of expired DReps that are part of the EpochState.

    The following family of equivalence relations indexed by Epoch formalize the concept of two maps of DReps being the same on non-expired DReps:

    DReps-[_]__ : Epoch  B.Rel DReps 0
DReps-[_]__ e dreps dreps
  = filter (activeInEpoch e) dreps  filter (activeInEpoch e) dreps

    This family of relations is monotonic w.r.t. increment on Epoch and respects suc:

    DReps--mono :  (e : Epoch) {dreps dreps : DReps}  DReps-[ e ] dreps  dreps  DReps-[ suc e ] dreps  dreps
DReps--mono e {dreps} {dreps} drepsdreps =
    begin
      filter (activeInEpoch (suc e)) dreps 
         filter- PQ 
      filter (activeInEpoch (suc e)) (filter (activeInEpoch e) dreps) 
         filter-cong drepsdreps 
      filter (activeInEpoch (suc e)) (filter (activeInEpoch e) dreps) 
         filter- PQ 
      filter (activeInEpoch (suc e)) dreps 
    
    where
      open import Relation.Binary.Reasoning.Setoid -Setoid
      import Relation.Unary as U
      module esp = HasPreorder (EpochStructure.preoEpoch epochStructure)
      PQ : U.Universal (( ((_ , e') : Credential  Epoch)   suc e  e') U. ( (_ , e')  e  e'))
      PQ _ p = esp.-trans (proj (esp.< e<suc)) p

DReps--suc :  (e : Epoch) {dreps dreps : DReps}  DReps-[ e ] dreps  dreps  DReps-[ suc e ] mapValues suc dreps  mapValues suc dreps
DReps--suc e {dreps} {dreps} drepsdreps =
    begin
      filter (activeInEpoch (suc e)) (mapValues suc dreps) 
         filter-map PQ 
      filter (activeInEpoch (suc e)) (mapValues suc (filter (activeInEpoch e) dreps)) 
         filter-cong (map- drepsdreps) 
      filter (activeInEpoch (suc e)) (mapValues suc (filter (activeInEpoch e) dreps)) 
         filter-map PQ 
      filter (activeInEpoch (suc e)) (mapValues suc dreps) 
    
    where
      open import Relation.Binary.Reasoning.Setoid -Setoid
      import Relation.Unary as U
      PQ : U.Universal (( ((_ , e') : Credential  Epoch)   suc e  suc e') U. ( (_ , e')  e  e'))
      PQ _ p = -pred p

    DReps--sym : (e : Epoch) {dreps dreps : DReps}  DReps-[ e ] dreps  dreps  DReps-[ e ] dreps  dreps
DReps--sym _ = -isEquivalence.sym
  where
    open import Relation.Binary.Structures __
    module -isEquivalence = IsEquivalence -isEquivalence

module AcceptedByDRep- { ' : RatifyEnv} (' : RatifyEnv-   ') (eSt : EnactState) (gaSt : GovActionState) where
  module  = RatifyEnv 
  module ' = RatifyEnv '
  module ' = RatifyEnv-__ '
  module abdr  = AcceptedByDRep   eSt gaSt
  module abdr' = AcceptedByDRep ' eSt gaSt

  castVotes : abdr.castVotes  abdr'.castVotes
  castVotes = refl

  activeDReps- : abdr.activeDReps  abdr'.activeDReps
  activeDReps- with '.currentEpoch
  ... | refl = dom-cong $
    begin
      filter (activeInEpoch .currentEpoch) .dreps 
         filter-idem {m = .dreps} 
      filter (activeInEpoch _) (filter (activeInEpoch _) .dreps) 
         filter-cong '.dreps 
      filter (activeInEpoch _) (filter (activeInEpoch _) '.dreps) 
         filter-idem {m = '.dreps} 
      filter (activeInEpoch '.currentEpoch) '.dreps 
    
    where open import Relation.Binary.Reasoning.Setoid -Setoid

  predeterminedDRepVotes : abdr.predeterminedDRepVotes  abdr'.predeterminedDRepVotes
  predeterminedDRepVotes = refl

  defaultDRepCredentialVotes : abdr.defaultDRepCredentialVotes  abdr'.defaultDRepCredentialVotes
  defaultDRepCredentialVotes = map- (map- activeDReps-)

  actualVotes : abdr.actualVotes  abdr'.actualVotes
  actualVotes = -cong {m = abdr.castVotes} {m' = abdr.defaultDRepCredentialVotes  abdr.predeterminedDRepVotes}
                       {m'' = abdr'.castVotes} {m''' = abdr'.defaultDRepCredentialVotes  abdr'.predeterminedDRepVotes}
                       (-isEquivalence.reflexive (P.cong proj castVotes))
                       (-cong {m = abdr.defaultDRepCredentialVotes} {m' = abdr.predeterminedDRepVotes }
                               {m'' = abdr'.defaultDRepCredentialVotes} {m''' = abdr'.predeterminedDRepVotes }
                               defaultDRepCredentialVotes (-isEquivalence.reflexive (P.cong proj predeterminedDRepVotes)))
    where
      open import Relation.Binary.Structures __
      module -isEquivalence = IsEquivalence -isEquivalence

  t : abdr.t  abdr'.t
  t = refl

  acceptedStake : abdr.acceptedStake  abdr'.acceptedStake
  acceptedStake =
    indexedSum'-cong {f = id}
      {x = StakeDistrs.stakeDistrVDeleg .stakeDistrs  abdr.actualVotes  _}
      {y = StakeDistrs.stakeDistrVDeleg '.stakeDistrs  abdr'.actualVotes  _}
      (res-cong {m = StakeDistrs.stakeDistrVDeleg .stakeDistrs}
                 {m' = StakeDistrs.stakeDistrVDeleg '.stakeDistrs}
        (StakeDistrs-__.stakeDistrVDeleg '.stakeDistrs)
        (-cong {m = abdr.actualVotes} {m' = abdr'.actualVotes} actualVotes))

  totalStake : abdr.totalStake  abdr'.totalStake
  totalStake = indexedSum'-cong {f = id}
                 {x = StakeDistrs.stakeDistrVDeleg .stakeDistrs  dom (abdr.actualVotes ^ ( Vote.yes    Vote.no ))}
                 {y = StakeDistrs.stakeDistrVDeleg '.stakeDistrs  dom (abdr'.actualVotes ^ ( Vote.yes    Vote.no ))}
                 (res-cong {m = StakeDistrs.stakeDistrVDeleg .stakeDistrs}
                            {m' = StakeDistrs.stakeDistrVDeleg '.stakeDistrs}
                            (StakeDistrs-__.stakeDistrVDeleg '.stakeDistrs)
                            (dom-cong (cores-cong {m = abdr.actualVotes} {m' = abdr'.actualVotes} actualVotes -isEquivalence.refl)))
    where
    open import Relation.Binary.Structures __
    module -isEquivalence = IsEquivalence -isEquivalence

  accepted- : abdr.accepted  abdr'.accepted
  accepted- x =
    subst ( totalStake  (abdr'.acceptedStake / totalStake)  abdr'.t ) totalStake
      (subst ( acceptedStake  (acceptedStake / abdr.totalStake)  abdr'.t ) acceptedStake
        (subst ( t  (abdr.acceptedStake / abdr.totalStake)  t ) t x))

module RATIFY { ' : RatifyEnv} (' : RatifyEnv-   ') {a a' : _} {rSt rSt' : RatifyState} where
  module   = RatifyEnv 
  module ' = RatifyEnv '
  module rSt  = RatifyState rSt
  module rSt' = RatifyState rSt'

  module ' = RatifyEnv-__ '

  cong :  (rStrSt' : rSt  rSt') {rSt'' rSt'''}
           rSt   a ,RATIFY rSt''
        '  rSt'  a ,RATIFY rSt'''
        rSt''  rSt'''
  cong rStrSt' (RATIFY-Accept {es = es}((ac , de , ex) , en)) (RATIFY-Accept {es = es'} ((ac' , ex') , en'))
    with  '.treasury |  '.currentEpoch | rStrSt'
  ... | refl | refl | refl with ENACT-deterministic en en'
  ... | refl = refl
  cong refl (RATIFY-Accept (ac , _)) (RATIFY-Reject (ac , _))
    = -elim (ac (AcceptedConds-.acceptConds- ' rSt a ac))
  cong refl (RATIFY-Accept (ac , _)) (RATIFY-Continue (ac , _))
    = -elim (ac (AcceptedConds-.acceptConds- ' rSt a ac))
  cong refl (RATIFY-Reject (ac , _)) (RATIFY-Accept (ac , _))
    = -elim (ac (AcceptedConds-.acceptConds- (RatifyEnv--sym ') rSt' a ac))
  cong refl (RATIFY-Reject (ac , _)) (RATIFY-Reject (ac' , _))
    = refl
  cong refl (RATIFY-Reject (_ , ex)) (RATIFY-Continue (_ , ex))
    rewrite '.currentEpoch = -elim (ex ex)
  cong refl (RATIFY-Continue (ac , _)) (RATIFY-Accept (ac , _))
    = -elim (ac (AcceptedConds-.acceptConds- (RatifyEnv--sym ') rSt' a ac))
  cong refl (RATIFY-Continue (_ , ex)) (RATIFY-Reject (_ , ex))
    rewrite '.currentEpoch = -elim (ex ex)
  cong refl (RATIFY-Continue (ac , ex)) (RATIFY-Continue (ac' , ex')) = refl

module RATIFIES { ' : RatifyEnv} (' : RatifyEnv-   ') where
  cong
    :  {rSt rSt' : RatifyState} (rStrSt' : rSt  rSt') {govSt govSt' : GovState} (govStgovSt' : govSt  govSt') {rSt'' rSt''' : RatifyState}
           rSt   govSt  ,RATIFIES rSt''
        '  rSt'  govSt' ,RATIFIES rSt'''
        rSt''  rSt'''

    with RATIFY.cong ' {a = a} {a' = a'} refl p q
  ... | refl = cong refl refl ps  qs
