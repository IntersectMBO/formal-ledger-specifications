    module Foreign.Convertible where

open import Ledger.Prelude
open import Foreign.HaskellTypes

record Convertible (A B : Type) : Type where
  field to   : A  B
        from : B  A
open Convertible ... public

HsConvertible : (A : Set)   HasHsType A   Set
HsConvertible A = Convertible A (HsType A)

Convertible-Refl :  {A}  Convertible A A
Convertible-Refl =  where .to  id; .from  id

Convertible : (Type  Type)  (Type  Type)  Type
Convertible T U =  {A B}   Convertible A B   Convertible (T A) (U B)

Convertible : (Type  Type  Type)  (Type  Type  Type)  Type
Convertible T U =  {A B}   Convertible A B   Convertible (T A) (U B)

FunctorConvertible :  {F : Type}   Functor F   Convertible F F
FunctorConvertible =  where
  .to    map to
  .from  map from

BifunctorConvertible :  {F}   Bifunctor F   Convertible F F
BifunctorConvertible =  where
  .to    bimap to to
  .from  bimap from from

__ :  {A B C}  Convertible A B  Convertible B C  Convertible A C
(c  c') .to   = c' .to    c  .to
(c  c') .from = c  .from  c' .from

-- ** instances

open import Foreign.Haskell
open import Foreign.Haskell.Coerce using (coerce)

instance
  Convertible-Maybe : Convertible Maybe Maybe
  Convertible-Maybe = FunctorConvertible

  Convertible- : Convertible __ __
  Convertible- = BifunctorConvertible

  Convertible-Pair : Convertible __ Pair
  Convertible-Pair =  where
    .to    coerce  bimap to to
    .from  bimap from from  coerce

  Convertible- : Convertible __ __
  Convertible- = BifunctorConvertible

  Convertible-Either : Convertible __ Either
  Convertible-Either =  where
    .to    coerce  bimap to to
    .from  bimap from from  coerce

  Convertible-FinSet : Convertible _ List
  Convertible-FinSet =  where
    .to    map to    setToList
    .from  fromList  map from

  Convertible-Map :  {K K' V V'}   DecEq K 
      Convertible K K'    Convertible V V' 
     Convertible (K  V) (List $ Pair K' V')
  Convertible-Map =  where
    .to    to         proj
    .from  fromList  map from

  Convertible-List : Convertible List List
  Convertible-List =  where
    .to    map to
    .from  map from

  Convertible-Fun :  {A A' B B'}   Convertible A A'    Convertible B B'   Convertible (A  B) (A'  B')
  Convertible-Fun =  where
    .to     f  to    f  from
    .from   f  from  f  to
