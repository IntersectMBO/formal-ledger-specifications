data ValidDeps (pp : PParams) (deps : L.Deposits) : List L.DCert  Set where
  []         : ValidDeps pp deps []
  delegate   :  {c del kh v certs}
              ValidDeps pp (C.updateCertDeposit pp (L.delegate c del kh v) deps) certs
              ValidDeps pp deps (L.delegate c del kh v  certs)
  dereg      :  {c md d certs}
              (L.CredentialDeposit c , d)  deps
              md  nothing  md  just d
              ValidDeps pp (C.updateCertDeposit pp (L.dereg c md) deps) certs
              ValidDeps pp deps (L.dereg c md  certs)
  regdrep    :  {c v a certs}
              ValidDeps pp deps certs
              ValidDeps pp deps (L.regdrep c v a  certs)
  deregdrep  :  {c d certs}
              ValidDeps pp deps certs
              ValidDeps pp deps (L.deregdrep c d  certs)
  regpool    :  {kh p certs}
              ValidDeps pp deps certs
              ValidDeps pp deps (L.regpool kh p  certs)
  retirepool :  {kh e certs}
              ValidDeps pp deps certs
              ValidDeps pp deps (L.retirepool kh e   certs)
  ccreghot   :  {c v certs}
              ValidDeps pp deps certs
              ValidDeps pp deps (L.ccreghot c v  certs)
  reg        :  {c d certs}
              ValidDeps pp (C.updateCertDeposit pp (L.reg c d) deps) certs
              ValidDeps pp deps (L.reg c d  certs)

data ValidDeps (pp : PParams) (deps : L.Deposits) : List L.DCert  Set where
  []         : ValidDeps pp deps []
  regdrep    :  {c v a certs}
              ValidDeps pp (C.updateCertDeposit pp (L.regdrep c v a) deps) certs
              ValidDeps pp deps (L.regdrep c v a  certs)
  deregdrep  :  {c d certs}
              (L.DRepDeposit c , d)  deps
              ValidDeps pp (C.updateCertDeposit pp (L.deregdrep c d) deps) certs
              ValidDeps pp deps (L.deregdrep c d  certs)
  delegate   :  {c del kh v certs}
              ValidDeps pp deps certs
              ValidDeps pp deps (L.delegate c del kh v  certs)
  dereg      :  {c d certs}
              ValidDeps pp deps certs
              ValidDeps pp deps (L.dereg c d  certs)
  regpool    :  {kh p certs}
              ValidDeps pp deps certs
              ValidDeps pp deps (L.regpool kh p  certs)
  retirepool :  {kh e certs}
              ValidDeps pp deps certs
              ValidDeps pp deps (L.retirepool kh e   certs)
  ccreghot   :  {c v certs}
              ValidDeps pp deps certs
              ValidDeps pp deps (L.ccreghot c v  certs)
  reg        :  {c d certs}
              ValidDeps pp deps certs
              ValidDeps pp deps (L.reg c d  certs)

  CERTToConf .conv deposits@(retirepool* _ _)  (L.CERT-pool pool)    = C.CERT-pool (conv pool)

  CERTS'ToConf :  { s dcerts s'} (let open L.CertEnv )
                CertDeps* pp dcerts
                  ReflexiveTransitiveClosure {sts = L.___,CERT_}  s dcerts s'   deposits _ 
                   ReflexiveTransitiveClosure {sts = C.___,CERT_}

  CERTSToConf :  { s dcerts s'} (let open L.CertEnv )

                        (getCertDeps* (updateCertDeps* dcerts deposits) conv s')
  CERTSToConf .conv deposits (RTC (base , step)) =
    RTC (getCertDeps* deposits conv base , deposits conv step)
