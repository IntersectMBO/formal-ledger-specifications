      computeProof : ComputationResult String ([ s' ]   s  tx ,LEDGER s')
      computeProof = case isValid  true of  where
        (yes p)  do
          (utxoSt' , utxoStep)  computeUtxow utxo utxoSt tx
          (certSt' , certStep)  computeCerts cert certState txCerts
          (govSt'  , govStep)   computeGov (gov certSt') (rmOrphanDRepVotes certSt' govSt) (txgov txb)
          success (_ , LEDGER-V p utxoStep certStep govStep)
        (no p)  do
          (utxoSt' , utxoStep)  computeUtxow utxo utxoSt tx
          success (_ , LEDGER-I (-not p) utxoStep)

      completeness :  s'    s  tx ,LEDGER s'  (proj <$> computeProof)  success s'
      completeness ledgerSt (LEDGER-V v utxoStep certStep govStep)
        with isValid  true
      ... | no v = contradiction v v
      ... | yes refl
        with computeUtxow utxo utxoSt tx | complete _ _ _ _ utxoStep
      ... | success (utxoSt' , _) | refl
        with computeCerts cert certState txCerts | complete _ _ _ _ certStep
      ... | success (certSt' , _) | refl
        with computeGov (gov certSt') (rmOrphanDRepVotes certSt' govSt ) (txgov txb) | complete {STS = ___,GOVS_} (gov certSt') _ _ _ govStep
      ... | success (govSt' , _) | refl = refl
      completeness ledgerSt (LEDGER-I i utxoStep)
        with isValid  true
      ... | yes refl = case i of  ()
      ... | no v
        with computeUtxow utxo utxoSt tx | complete _ _ _ _ utxoStep
      ... | success (utxoSt' , _) | refl = refl

module LEDGER-PROPS (tx : Tx) ( : LEnv) (s : LState) where
  open Tx tx renaming (body to txb); open TxBody txb
  open LEnv  renaming (pparams to pp)
  open PParams pp using (govActionDeposit; govActionLifetime)
  open LState s
  open CertState certState
  open DState dState

  -- initial utxo deposits
  utxoDeps : Deposits
  utxoDeps = UTxOState.deposits (LState.utxoSt s)

  -- GovState definitions and lemmas --
  mkAction : GovProposal    GovActionID  GovActionState
  mkAction p n = let open GovProposal p in
    mkGovStatePair
      (govActionLifetime + epoch slot)
      (txId , n) returnAddr action prevAction

  -- update GovState with a proposal
  propUpdate : GovState  GovProposal    GovState
  propUpdate s p n = insertGovAction s (mkAction p n)

  -- update GovState with a vote
  voteUpdate : GovState  GovVote  GovState
  voteUpdate s v = addVote s gid voter vote
    where open GovVote v

  -- update GovState with a list of votes and proposals
  updateGovStates : List (GovVote  GovProposal)    GovState  GovState
  updateGovStates [] _ s = s
  updateGovStates (inj v  vps) k s = updateGovStates vps (suc k) (voteUpdate s v)
  updateGovStates (inj p  vps) k s = updateGovStates vps (suc k) (propUpdate s p k)

  -- updateGovStates faithfully represents a step of the LEDGER sts
  STSGovSt :  {s' : LState}    s  tx ,LEDGER s'
                isValid  true  GovStateOf s'  updateGovStates (txgov txb) 0 (rmOrphanDRepVotes (CertStateOf s') (GovStateOf s))
  STSGovSt (LEDGER-V x) refl = STSupdateGovSt (txgov txb) 0 (proj (proj (proj x)))
    where
    STSupdateGovSt : (vps : List (GovVote  GovProposal)) (k : ) {certSt : CertState} {govSt govSt' : GovState}
       (___*'_ {____ = IdSTS}{___,GOV_} ($\begin{pmatrix} \,\href{Ledger.Conway.Specification.Transaction.html#4998}{\htmlId{6762}{\htmlClass{Function}{\text{txId}}}}\, \\ \,\href{Ledger.Core.Specification.Epoch.html#818}{\htmlId{6769}{\htmlClass{Function}{\text{epoch}}}}\, \,\href{Ledger.Conway.Specification.Ledger.html#1496}{\htmlId{6775}{\htmlClass{Function}{\text{slot}}}}\, \\ \,\href{Ledger.Conway.Specification.Ledger.Properties.html#5128}{\htmlId{6782}{\htmlClass{Function}{\text{pp}}}}\, \\ \,\href{Ledger.Conway.Specification.Ledger.html#1519}{\htmlId{6787}{\htmlClass{Function}{\text{ppolicy}}}}\, \\ \,\href{Ledger.Conway.Specification.Ledger.html#1580}{\htmlId{6797}{\htmlClass{Function}{\text{enactState}}}}\, \\ \,\href{Ledger.Conway.Specification.Ledger.Properties.html#6658}{\htmlId{6810}{\htmlClass{Bound}{\text{certSt}}}}\, \\ \,\href{Class.IsSet.html#916}{\htmlId{6819}{\htmlClass{Function}{\text{dom}}}}\, \,\href{Ledger.Conway.Specification.Certs.html#4558}{\htmlId{6823}{\htmlClass{Function}{\text{rewards}}}}\, \end{pmatrix}$ , k) govSt vps govSt')
