-- ** Proof that the set equality `govDepsMatch` (below) is a LEDGER invariant.

-- Mapping a list of `GovActionID  GovActionState`s to a list of
-- `DepositPurpose`s is so common, we give it a name `dpMap`;
-- it's equivalent to `map ( (id , _)  GovActionDeposit id)`.
dpMap : GovState  List DepositPurpose
dpMap = map (GovActionDeposit  proj)

isGADeposit : DepositPurpose  Type
isGADeposit dp = isGADeposit dp  true
  where
  isGADeposit : DepositPurpose  Bool
  isGADeposit (GovActionDeposit _) = true
  isGADeposit _                    = false

govDepsMatch : LState  Type
govDepsMatch ls =
  filter isGADeposit (dom (DepositsOf ls))  fromList (dpMap (GovStateOf ls))

module LEDGER-PROPS (tx : Tx) ( : LEnv) (s : LState) where
  open Tx tx renaming (body to txb); open TxBody txb
  open LEnv  renaming (pparams to pp)
  open PParams pp using (govActionDeposit; govActionLifetime)
  open LState s
  open CertState certState
  open DState dState

  -- initial utxo deposits
  utxoDeps : Deposits
  utxoDeps = UTxOState.deposits (LState.utxoSt s)

  -- GovState definitions and lemmas --
  mkAction : GovProposal    GovActionID  GovActionState
  mkAction p n = let open GovProposal p in
    mkGovStatePair
      (govActionLifetime + epoch slot)
      (txId , n) returnAddr action prevAction

  -- update GovState with a proposal
  propUpdate : GovState  GovProposal    GovState
  propUpdate s p n = insertGovAction s (mkAction p n)

  -- update GovState with a vote
  voteUpdate : GovState  GovVote  GovState
  voteUpdate s v = addVote s gid voter vote
    where open GovVote v

  -- update GovState with a list of votes and proposals
  updateGovStates : List (GovVote  GovProposal)    GovState  GovState
  updateGovStates [] _ s = s
  updateGovStates (inj v  vps) k s = updateGovStates vps (suc k) (voteUpdate s v)
  updateGovStates (inj p  vps) k s = updateGovStates vps (suc k) (propUpdate s p k)

  -- updateGovStates faithfully represents a step of the LEDGER sts
  STSGovSt :  {s' : LState}    s  tx ,LEDGER s'
                isValid  true  GovStateOf s'  updateGovStates (txgov txb) 0 (rmOrphanDRepVotes (CertStateOf s') (GovStateOf s))
  STSGovSt (LEDGER-V x) refl = STSupdateGovSt (txgov txb) 0 (proj (proj (proj x)))
    where
    STSupdateGovSt : (vps : List (GovVote  GovProposal)) (k : ) {certSt : CertState} {govSt govSt' : GovState}
       (___*'_ {____ = IdSTS}{___,GOV_} ($\begin{pmatrix} \,\href{Ledger.Conway.Specification.Transaction.html#4766}{\htmlId{4056}{\htmlClass{Function}{\text{txId}}}}\, \\ \,\href{Ledger.Core.Specification.Epoch.html#954}{\htmlId{4063}{\htmlClass{Function}{\text{epoch}}}}\, \,\href{Ledger.Conway.Specification.Ledger.html#1055}{\htmlId{4069}{\htmlClass{Function}{\text{slot}}}}\, \\ \,\href{Ledger.Conway.Specification.Ledger.Properties.Base.html#2422}{\htmlId{4076}{\htmlClass{Function}{\text{pp}}}}\, \\ \,\href{Ledger.Conway.Specification.Ledger.html#1078}{\htmlId{4081}{\htmlClass{Function}{\text{ppolicy}}}}\, \\ \,\href{Ledger.Conway.Specification.Ledger.html#1139}{\htmlId{4091}{\htmlClass{Function}{\text{enactState}}}}\, \\ \,\href{Ledger.Conway.Specification.Ledger.Properties.Base.html#3952}{\htmlId{4104}{\htmlClass{Bound}{\text{certSt}}}}\, \\ \,\href{Class.IsSet.html#916}{\htmlId{4113}{\htmlClass{Function}{\text{dom}}}}\, \,\href{Ledger.Conway.Specification.Certs.html#4479}{\htmlId{4117}{\htmlClass{Function}{\text{rewards}}}}\, \end{pmatrix}$ , k) govSt vps govSt')

  filterCR : (c : DCert) (deps : Deposits)
              filter isGADeposit (dom ( deps  certRefund c   ))  filter isGADeposit (dom (deps ))
  filterCR (dereg c _) deps = .sym $ begin
    filter isGADeposit (dom (deps ))  filter-cong $ dom-cong (res-ex- Dec--singleton) 
    filter isGADeposit (dom ((deps  cr)  (deps  cr )))  filter-cong dom 
    filter isGADeposit (dom ((deps  cr) )  dom ((deps  cr )  ))  filter-hom- 
    filter isGADeposit (dom ((deps  cr) ))  filter isGADeposit (dom ((deps  cr )  ))  -cong filter0 .refl 
      filter isGADeposit (dom ((deps  cr )  ))  -identity $ filter isGADeposit (dom ((deps  cr )  )) 
    filter isGADeposit (dom ((deps  cr ) )) 
    where
    cr =  CredentialDeposit c 
    filter0 = filter- ( _  CredDepIsNotGADep  (from -singleton)  res-dom)
  filterCR (deregdrep c _) deps = .sym $ begin
    filter isGADeposit (dom (deps ))  filter-cong $ dom-cong (res-ex- Dec--singleton) 
    filter isGADeposit (dom ((deps  cr)  (deps  cr )))  filter-cong dom 
    filter isGADeposit (dom ((deps  cr) )  dom ((deps  cr )  ))  filter-hom- 
    filter isGADeposit (dom ((deps  cr) ))  filter isGADeposit (dom ((deps  cr )  ))  -cong filter0 .refl 
      filter isGADeposit (dom ((deps  cr )  ))  -identity $ filter isGADeposit (dom ((deps  cr )  )) 
    filter isGADeposit (dom ((deps  cr ) )) 
    where
    cr =  DRepDeposit c 
    filter0 = filter- ( _  DRepDepIsNotGADep  (from -singleton)  res-dom)
  filterCR (delegate _ _ _ _)  deps = filter-cong (dom-cong (res- {M = deps}))
  filterCR (regpool _ _)       deps = filter-cong (dom-cong (res- {M = deps}))
  filterCR (regdrep _ _ _)     deps = filter-cong (dom-cong (res- {M = deps}))
  filterCR (retirepool _ _)    deps = filter-cong (dom-cong (res- {M = deps}))
  filterCR (ccreghot _ _)      deps = filter-cong (dom-cong (res- {M = deps}))
  filterCR (reg _ _)           deps = filter-cong (dom-cong (res- {M = deps}))

  filterCD : (c : DCert) (deps : Deposits)  filter isGADeposit (dom (certDeposit c pp ))  
  filterCD (delegate _ _ _ _)  deps = filter-  _  CredDepIsNotGADep  from -singleton  dom-singlesingle
  filterCD (reg _ _)           deps = filter-  _  CredDepIsNotGADep  from -singleton  dom-singlesingle
  filterCD (regpool _ _)       deps = filter-  _  PoolDepIsNotGADep  from -singleton  dom-singlesingle
  filterCD (regdrep _ _ _)     deps = filter-  _  DRepDepIsNotGADep  from -singleton  dom-singlesingle
  filterCD (dereg _ _)         deps = .trans (filter-cong dom) $ filter-  _ a _  - a
  filterCD (retirepool _ _)    deps = .trans (filter-cong dom) $ filter-  _ a _  - a
  filterCD (deregdrep _ _)     deps = .trans (filter-cong dom) $ filter-  _ a _  - a
  filterCD (ccreghot _ _)      deps = .trans (filter-cong dom) $ filter-  _ a _  - a

  noGACerts : (cs : List DCert) (deps : Deposits)
     filter isGADeposit (dom (updateCertDeposits pp cs deps))  filter isGADeposit (dom deps)
  noGACerts [] _ = filter-cong .refl
  noGACerts (dcert@(delegate _ _ _ _)  cs) deps = begin
    filter isGADeposit (dom (updateCertDeposits pp cs (deps  cd)))  noGACerts cs _ 
    filter isGADeposit (dom (deps  cd))  filter-cong domdom 
    filter isGADeposit (dom deps  dom (cd  ))  filter-hom- 
    filter isGADeposit (dom deps)  filter isGADeposit (dom (cd ))  -cong .refl $ filterCD dcert deps 
    filter isGADeposit (dom deps)    -identity $ filter isGADeposit (dom deps) 
    filter isGADeposit (dom deps) 
    where
      cd = certDeposit dcert pp
      filter0 = filterCD dcert deps
  noGACerts (dcert@(reg _ _)  cs) deps = begin
    filter isGADeposit (dom (updateCertDeposits pp cs (deps  cd)))  noGACerts cs _ 
    filter isGADeposit (dom (deps  cd))  filter-cong domdom 
    filter isGADeposit (dom deps  dom (cd  ))  filter-hom- 
    filter isGADeposit (dom deps)  filter isGADeposit (dom (cd ))  -cong .refl $ filterCD dcert deps 
    filter isGADeposit (dom deps)    -identity $ filter isGADeposit (dom deps) 
    filter isGADeposit (dom deps) 
    where

  noGACerts (dcert@(regpool _ _)  cs) deps = begin

    filter isGADeposit (dom (deps  cd))  filter-cong domdom

    filter isGADeposit (dom deps) 
    where

  noGACerts (dcert@(regdrep _ _ _)  cs) deps = begin

    filter isGADeposit (dom (deps  certRefund (dereg c v)))  filterCR dcert deps

  noGACerts (ccreghot _ _  cs) deps = noGACerts cs deps

  dpMap-update- :  gSt p k
     fromList (dpMap gSt)   GovActionDeposit (txId , k) 
         fromList (dpMap (propUpdate gSt p k))
  dpMap-update- [] p k = -identity (fromList (dpMap [ mkAction p k ]))
  dpMap-update- (g@(gaID , gaSt)  gSt) p k
    with (govActionPriority (GovActionTypeOf gaSt))
         ? (govActionPriority (GovActionTypeOf (proj (mkAction p k))))
  ... | yes _  = begin
      fromList (dpMap (g  gSt))   GovActionDeposit (txId , k) 
         -cong fromList--singleton .refl 
      ( GovActionDeposit gaID   fromList (dpMap gSt))   GovActionDeposit (txId , k) 
         -assoc  GovActionDeposit gaID  (fromList (dpMap gSt))  GovActionDeposit (txId , k)

         -cong .refl (dpMap-update- gSt p k) 
       GovActionDeposit gaID   fromList (dpMap (propUpdate gSt p k))
         fromList--singleton 
      fromList (dpMap (g  insertGovAction gSt (mkAction p k)))
        
  ... | no _   = begin
      fromList (dpMap (g  gSt))   GovActionDeposit (txId , k)

       GovActionDeposit (txId , k)   fromList (dpMap (g  gSt))
         fromList--singleton

         connex-lemma (propUpdate gSt p k) p' ps 
    fromList (dpMap (updateGovStates (map inj (p'  ps)) (suc k) (propUpdate gSt p k)))
