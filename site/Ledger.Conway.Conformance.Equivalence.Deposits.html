updateDDep : PParams  L.DCert  L.Deposits  L.Deposits
updateDDep pp cert@(L.delegate _ _ _ _) deps = C.updateCertDeposit pp cert deps
updateDDep pp cert@(L.dereg _ _)        deps = C.updateCertDeposit pp cert deps
updateDDep pp cert@(L.reg _ _)          deps = C.updateCertDeposit pp cert deps
updateDDep pp cert                      deps = deps

updateDDeps : PParams  List L.DCert  L.Deposits  L.Deposits
updateDDeps pp []             deps = deps
updateDDeps pp (cert  certs) deps = updateDDeps pp certs (updateDDep pp cert deps)

updateGDep : PParams  L.DCert  L.Deposits  L.Deposits
updateGDep pp cert@(L.regdrep _ _ _) deps = C.updateCertDeposit pp cert deps
updateGDep pp cert@(L.deregdrep _ _) deps = C.updateCertDeposit pp cert deps
updateGDep pp cert                   deps = deps

updateGDeps : PParams  List L.DCert  L.Deposits  L.Deposits
updateGDeps pp []             deps = deps
updateGDeps pp (cert  certs) deps = updateGDeps pp certs (updateGDep pp cert deps)

updateLedgerDeps : PParams  Tx  L.Deposits  L.Deposits  L.Deposits  L.Deposits
updateLedgerDeps pp tx deps@(ddeps , gdeps) = updateDDeps pp certs ddeps , updateGDeps pp certs gdeps
  where
    certs = tx .Tx.body .TxBody.txcerts

data DPurpose : L.DepositPurpose  Set where
  CredentialDeposit :  {c}  DPurpose (L.CredentialDeposit c)

data GPurpose : L.DepositPurpose  Set where
  DRepDeposit :  {c}  GPurpose (L.DRepDeposit c)

instance
  Dec-DPurpose? :  {p}  DPurpose p 
  Dec-DPurpose? {L.CredentialDeposit _} =  yes CredentialDeposit
  Dec-DPurpose? {L.PoolDeposit       _} =  no  ()
  Dec-DPurpose? {L.DRepDeposit       _} =  no  ()
  Dec-DPurpose? {L.GovActionDeposit  _} =  no  ()

  Dec-GPurpose? :  {p}  GPurpose p 
  Dec-GPurpose? {L.CredentialDeposit _} =  no  ()
  Dec-GPurpose? {L.PoolDeposit       _} =  no  ()
  Dec-GPurpose? {L.DRepDeposit       _} =  yes DRepDeposit
  Dec-GPurpose? {L.GovActionDeposit  _} =  no  ()

-- Compute DDeps for the CertState from full Deposits
certDDeps : L.Deposits  L.Deposits
certDDeps deps = filter ( (k , _)  DPurpose k) deps

-- Compute GDeps for the CertState from full Deposits
certGDeps : L.Deposits  L.Deposits
certGDeps deps = filter ( (k , _)  GPurpose k) deps

cong-updateCertDeposit :  pp cert {deps deps}
                        deps  deps
                        C.updateCertDeposit pp cert deps  C.updateCertDeposit pp cert deps
cong-updateCertDeposit pp (L.delegate c kh del v)            eq = -cong-r eq
cong-updateCertDeposit pp (L.dereg x x) {deps} {deps}     eq = restrict-cong deps deps eq
cong-updateCertDeposit pp (L.regpool x x)                   eq = -cong-r eq
cong-updateCertDeposit pp (L.retirepool x x)                eq = eq
cong-updateCertDeposit pp (L.regdrep x x x)                eq = -cong-r eq
cong-updateCertDeposit pp (L.deregdrep x x) {deps} {deps} eq = restrict-cong deps deps eq
cong-updateCertDeposit pp (L.ccreghot x x)                  eq = eq
cong-updateCertDeposit pp (L.reg x x)                       eq = -cong-r eq

cong-certDDeps :  deps deps  deps  deps  certDDeps deps  certDDeps deps
cong-certDDeps = cong-filter

cong-certGDeps :  deps deps  deps  deps  certGDeps deps  certGDeps deps
cong-certGDeps = cong-filter

castValidDeps :  {pp deps deps certs}  deps  deps  ValidDeps pp deps certs  ValidDeps pp deps certs
castValidDeps                         eq [] = []
castValidDeps {pp} {certs = cert  _} eq (delegate   deps) = delegate           (castValidDeps (cong-updateCertDeposit pp cert eq) deps)
castValidDeps {pp} {deps} {deps} {certs = cert  _} eq (dereg h h' deps) = 
  dereg (proj eq h) h' 
        (castValidDeps (cong-updateCertDeposit pp cert {deps} {deps} eq) deps)
--castValidDeps {pp} {deps} {deps}
--                    {certs = cert  _} eq (dereg h    deps) = dereg (map (proj eq) h) (castValidDeps (cong-updateCertDeposit
--                                                                                                    pp cert {deps} {deps} eq) deps)
castValidDeps {pp} {certs = cert  _} eq (reg        deps) = reg (castValidDeps (cong-updateCertDeposit pp cert eq) deps)
castValidDeps                         eq (regdrep    deps) = regdrep            (castValidDeps eq deps)
castValidDeps                         eq (deregdrep  deps) = deregdrep          (castValidDeps eq deps)
castValidDeps                         eq (regpool    deps) = regpool            (castValidDeps eq deps)
castValidDeps                         eq (retirepool deps) = retirepool         (castValidDeps eq deps)
castValidDeps                         eq (ccreghot   deps) = ccreghot           (castValidDeps eq deps)

castValidDeps :  {pp deps deps certs}  deps  deps  ValidDeps pp deps certs  ValidDeps pp deps certs
castValidDeps   eq [] = []
castValidDeps   eq (delegate    deps) = delegate   (castValidDeps eq deps)
castValidDeps   eq (dereg       deps) = dereg      (castValidDeps eq deps)
castValidDeps {pp} {certs = cert  _}
                 eq (regdrep     deps) = regdrep    (castValidDeps (cong-updateCertDeposit pp cert eq) deps)
castValidDeps {pp} {deps} {deps} {certs = cert  _}
  eq (deregdrep h deps) = deregdrep (proj eq h)    (castValidDeps (cong-updateCertDeposit pp cert {deps} {deps} eq) deps)
castValidDeps   eq (regpool     deps) = regpool    (castValidDeps eq deps)
castValidDeps   eq (retirepool  deps) = retirepool (castValidDeps eq deps)
castValidDeps   eq (ccreghot    deps) = ccreghot   (castValidDeps eq deps)
castValidDeps   eq (reg         deps) = reg        (castValidDeps eq deps)

validDDeps :  {pp certs deps}  L.ValidCertDeposits pp deps certs  ValidDeps pp (certDDeps deps) certs
validDDeps                L.[]             = []
validDDeps               (L.delegate    v) = delegate   (castValidDeps (lem-add-included CredentialDeposit) (validDDeps v))
validDDeps               (L.regpool     v) = regpool    (castValidDeps (lem-add-excluded  ()) (validDDeps v))
validDDeps               (L.regdrep     v) = regdrep    (castValidDeps (lem-add-excluded  ()) (validDDeps v))
validDDeps {deps = deps} (L.dereg h h'  v) = dereg      (filter- deps CredentialDeposit h) h'
                                                        (castValidDeps (filter-restrict deps) (validDDeps v))
validDDeps {deps = deps} (L.deregdrep _ v) = deregdrep  (castValidDeps (lem-del-excluded deps  ()) (validDDeps v))
validDDeps               (L.ccreghot    v) = ccreghot   (validDDeps v)
validDDeps               (L.retirepool  v) = retirepool (validDDeps v)
validDDeps {deps = deps} (L.reg         v) = reg (castValidDeps (lem-add-included CredentialDeposit) (validDDeps v))

validGDeps :  {pp certs deps}  L.ValidCertDeposits pp deps certs  ValidDeps pp (certGDeps deps) certs
validGDeps                L.[]             = []
validGDeps               (L.delegate    v) = delegate   (castValidDeps (lem-add-excluded  ()) (validGDeps v))
validGDeps               (L.regpool     v) = regpool    (castValidDeps (lem-add-excluded  ()) (validGDeps v))
validGDeps               (L.regdrep     v) = regdrep    (castValidDeps (lem-add-included DRepDeposit) (validGDeps v))
validGDeps {deps = deps} (L.dereg _ _   v) = dereg      (castValidDeps (lem-del-excluded deps  ()) (validGDeps v))
validGDeps {deps = deps} (L.deregdrep h v) = deregdrep  (filter- deps DRepDeposit h)
                                                        (castValidDeps (filter-restrict deps) (validGDeps v))
validGDeps               (L.ccreghot    v) = ccreghot   (validGDeps v)
validGDeps               (L.retirepool  v) = retirepool (validGDeps v)
validGDeps               (L.reg         v) = reg (castValidDeps (lem-add-excluded  ()) (validGDeps v))

  where deps = s .L.LState.utxoSt .L.UTxOState.deposits

-isEquivalence : IsEquivalence __
-isEquivalence .IsEquivalence.refl = (id , id) , (id , id)
-isEquivalence .IsEquivalence.sym (deq , geq) = -isEquivalence .IsEquivalence.sym deq , -isEquivalence .IsEquivalence.sym geq
-isEquivalence .IsEquivalence.trans (deq , geq) (deq , geq) =
  -isEquivalence .IsEquivalence.trans deq deq ,
  -isEquivalence .IsEquivalence.trans geq geq

cong-updateDDep :  {pp} cert {deps deps}
                 deps  deps
                 updateDDep pp cert deps  updateDDep pp cert deps
cong-updateDDep {pp} cert@(L.delegate c del kh v)
                                        eq = cong-updateCertDeposit pp cert eq
cong-updateDDep {pp} cert@(L.dereg c v) {deps} {deps}
                                        eq = cong-updateCertDeposit pp cert {deps} {deps} eq
cong-updateDDep {pp} cert@(L.reg c v)   eq = cong-updateCertDeposit pp cert eq
cong-updateDDep      (L.regpool _ _)    eq = eq
cong-updateDDep      (L.regdrep _ _ _)  eq = eq
cong-updateDDep      (L.deregdrep _ _)  eq = eq
cong-updateDDep      (L.retirepool _ _) eq = eq
cong-updateDDep      (L.ccreghot _ _)   eq = eq

cong-updateDDeps :  {pp} certs {deps deps}
                  deps  deps
                  updateDDeps pp certs deps  updateDDeps pp certs deps
cong-updateDDeps []             eq = eq
cong-updateDDeps (cert  certs) eq = cong-updateDDeps certs (cong-updateDDep cert eq)

                 deps  deps
                 updateGDep pp cert deps  updateGDep pp cert deps
cong-updateGDep      (L.delegate _ _ _ _)   eq = eq
cong-updateGDep      (L.dereg _ _)          eq = eq
cong-updateGDep      (L.reg _ _)            eq = eq
cong-updateGDep      (L.regpool _ _)        eq = eq
cong-updateGDep {pp} cert@(L.regdrep _ _ _) eq = cong-updateCertDeposit pp cert eq
cong-updateGDep {pp} cert@(L.deregdrep _ _) {deps} {deps}
                                            eq = cong-updateCertDeposit pp cert {deps} {deps} eq
cong-updateGDep      (L.retirepool _ _)     eq = eq
cong-updateGDep      (L.ccreghot _ _  )     eq = eq

cong-updateGDeps :  {pp} certs {deps deps}
                  deps  deps
                  updateGDeps pp certs deps  updateGDeps pp certs deps
cong-updateGDeps []             eq = eq
cong-updateGDeps (cert  certs) eq = cong-updateGDeps certs (cong-updateGDep cert eq)

private open module S {A} = Setoid (-Setoid {A = A}) using () renaming (sym to -sym; trans to __)

lem-upd-cert-ddeps :  {pp} deps certs
                   updateDDeps pp certs (certDDeps deps)  certDDeps (L.updateCertDeposits pp certs deps)
lem-upd-cert-ddeps deps [] = id , id
lem-upd-cert-ddeps {pp} deps (L.delegate c del kh v  certs) =
  -sym (cong-updateDDeps certs (lem-add-included CredentialDeposit)) 
  lem-upd-cert-ddeps (deps  dep) certs
  where dep =  L.CredentialDeposit c , v 
lem-upd-cert-ddeps {pp} deps (L.dereg c v  certs) =
  -sym (cong-updateDDeps certs (filter-restrict deps)) 
  lem-upd-cert-ddeps (deps  cs ) certs
  where cs =  L.CredentialDeposit c

  where dep =  L.PoolDeposit kh , pp .PParams.poolDeposit

lem-upd-cert-ddeps {pp} deps (L.deregdrep c v  certs) =

  where cs =  L.DRepDeposit c

lemUpdCert :  pp ((ddeps , gdeps) : L.Deposits  L.Deposits) deps cert
            (ddeps , gdeps)  (certDDeps deps , certGDeps deps)
            (updateDDep pp cert ddeps , updateGDep pp cert gdeps) 
             (certDDeps (C.updateCertDeposit pp cert deps) , certGDeps (C.updateCertDeposit pp cert deps))
lemUpdCert pp (ddeps , gdeps) deps (L.delegate _ _ _ _) (deq , geq) = -cong-r deq
                                                                       -sym (lem-add-included CredentialDeposit)
                                                                    , geq  -sym (lem-add-excluded  ())
lemUpdCert pp (ddeps , gdeps) deps (L.dereg _ _)        (deq , geq) = restrict-cong ddeps (certDDeps deps) deq
                                                                       -sym (filter-restrict deps)
                                                                    , geq  -sym (lem-del-excluded deps  ())
lemUpdCert pp (ddeps , gdeps) deps (L.reg _ _)          (deq , geq) = (-cong-r deq
                                                                       -sym (lem-add-included CredentialDeposit))
                                                                    , geq  -sym (lem-add-excluded  ())
lemUpdCert pp (ddeps , gdeps) deps (L.regpool x x)     (deq , geq) = deq  -sym (lem-add-excluded  ())
                                                                    , geq  -sym (lem-add-excluded  ())
lemUpdCert pp (ddeps , gdeps) deps (L.regdrep _ _ _)    (deq , geq) = deq  -sym (lem-add-excluded  ())
                                                                    , -cong-r geq
                                                                       -sym (lem-add-included DRepDeposit)
lemUpdCert pp (ddeps , gdeps) deps (L.deregdrep _ _)    (deq , geq) = deq  -sym (lem-del-excluded deps  ())
                                                                    , (restrict-cong gdeps (certGDeps deps) geq
                                                                       -sym (filter-restrict deps))
lemUpdCert pp (ddeps , gdeps) deps (L.retirepool _ _)   (deq , geq) = deq , geq
lemUpdCert pp (ddeps , gdeps) deps (L.ccreghot _ _)     (deq , geq) = deq , geq
