    rdptr : Tx ℓ → ScriptPurpose → Maybe RedeemerPtr
rdptr tx = λ where
  $\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#3039}{\htmlId{838}{\htmlClass{InductiveConstructor}{\text{Cert}}}}\,          \\ \,\href{Ledger.Dijkstra.Specification.Script.Validation.html#854}{\htmlId{854}{\htmlClass{Bound}{\text{h}}}}\, \end{pmatrix}$ → map (Cert           ,_) $ indexOfDCert          h (DCertsOf tx)
  $\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#3044}{\htmlId{930}{\htmlClass{InductiveConstructor}{\text{Reward}}}}\,        \\ \,\href{Ledger.Dijkstra.Specification.Script.Validation.html#946}{\htmlId{946}{\htmlClass{Bound}{\text{h}}}}\, \end{pmatrix}$ → map (Reward         ,_) $ indexOfRewardAddress  h (WithdrawalsOf tx)
  $\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#3034}{\htmlId{1027}{\htmlClass{InductiveConstructor}{\text{Mint}}}}\,          \\ \,\href{Ledger.Dijkstra.Specification.Script.Validation.html#1043}{\htmlId{1043}{\htmlClass{Bound}{\text{h}}}}\, \end{pmatrix}$ → map (Mint           ,_) $ indexOfPolicyId       h (policies (MintedValueOf tx))
  $\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#3028}{\htmlId{1135}{\htmlClass{InductiveConstructor}{\text{Spend}}}}\,         \\ \,\href{Ledger.Dijkstra.Specification.Script.Validation.html#1151}{\htmlId{1151}{\htmlClass{Bound}{\text{h}}}}\, \end{pmatrix}$ → map (Spend          ,_) $ indexOfTxIn           h (SpendInputsOf tx)
  $\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#3051}{\htmlId{1232}{\htmlClass{InductiveConstructor}{\text{Vote}}}}\,          \\ \,\href{Ledger.Dijkstra.Specification.Script.Validation.html#1248}{\htmlId{1248}{\htmlClass{Bound}{\text{h}}}}\, \end{pmatrix}$ → map (Vote           ,_) $ indexOfVote           h (map GovVote.voter (ListOfGovVotesOf tx))
  $\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#3056}{\htmlId{1352}{\htmlClass{InductiveConstructor}{\text{Propose}}}}\,       \\ \,\href{Ledger.Dijkstra.Specification.Script.Validation.html#1368}{\htmlId{1368}{\htmlClass{Bound}{\text{h}}}}\, \end{pmatrix}$ → map (Propose        ,_) $ indexOfProposal       h (ListOfGovProposalsOf tx)
  $\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#3064}{\htmlId{1456}{\htmlClass{InductiveConstructor}{\text{Guard}}}}\,         \\ \,\href{Ledger.Dijkstra.Specification.Script.Validation.html#1472}{\htmlId{1472}{\htmlClass{Bound}{\text{h}}}}\, \end{pmatrix}$ → map (Guard          ,_) $ indexOfGuard          h (setToList (GuardsOf tx))

indexedRdmrs : Tx ℓ → ScriptPurpose → Maybe (Redeemer × ExUnits)
indexedRdmrs tx sp = maybe (λ x → lookupᵐ? (RedeemersOf tx) x) nothing (rdptr tx sp)

    The function getDatumSpend (formerly getDatum) retrieves the datum required by a phase-2 script associated to a spending input. Note that, even though this function formally returns an element of type Maybe Datum, the preconditions of the UTXOW rule ensure that the datum is always present (otherwise, it is a phase-1 validation failure).

    getDatumSpend
  : Tx ℓ
  → UTxO
  → TxIn
  → Maybe Datum
getDatumSpend tx utxo₀ txin
  with lookupᵐ? utxo₀ txin
... | just (_ , _ , just d , _) =
  case d of λ where
    (inj₁ d) → just d
    (inj₂ h) → lookupHash h (DataOf tx)
... | _ = nothing

    txInfo : (ℓ : TxLevel) → UTxO → Tx ℓ → TxInfo

txInfo TxLevelTop utxo tx =
  record  { realizedInputs      = utxo ∣ (SpendInputsOf tx)
          ; txOuts              = TxOutsOf tx
          ; txFee               = FeesOf? tx
          ; mint                = MintedValueOf tx
          ; txCerts             = DCertsOf tx
          ; txWithdrawals       = WithdrawalsOf tx
          ; txVldt              = ValidIntervalOf tx
          ; vkKey               = RequiredSignerHashesOf tx
          ; txGuards            = GuardsOf tx
          ; txData              = DataOf tx
          ; txId                = TxIdOf tx
          ; txInfoSubTxs        = nothing
          ; txDirectDeposits    = DirectDepositsOf tx
          ; txBalanceIntervals  = BalanceIntervalsOf tx
          }

txInfo TxLevelSub utxo tx =
  record  { realizedInputs      = utxo ∣ (TxBody.txIns txBody)
          ; txOuts              = TxBody.txOuts txBody
          ; txFee               = nothing
          ; mint                = TxBody.mint txBody
          ; txCerts             = TxBody.txCerts txBody
          ; txWithdrawals       = TxBody.txWithdrawals txBody
          ; txVldt              = TxBody.txVldt txBody
          ; vkKey               = TxBody.requiredSignerHashes txBody
          ; txGuards            = TxBody.txGuards txBody
          ; txData              = DataOf tx
          ; txId                = TxBody.txId txBody
          ; txInfoSubTxs        = nothing
          ; txDirectDeposits    = DirectDepositsOf tx
          ; txBalanceIntervals  = BalanceIntervalsOf tx
          } where open Tx tx

txInfoForPurpose : {ℓ : TxLevel} → UTxO → Tx ℓ → ScriptPurpose → TxInfo

-- subtransactions: never get subTx infos (even if the ScriptPurpose is Guard).
txInfoForPurpose {TxLevelSub} utxo tx _ = txInfo TxLevelSub utxo tx

-- top-level transactions:
txInfoForPurpose {TxLevelTop} utxo tx sp with sp
   -- · guard scripts see subTx infos
... | $\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#3064}{\htmlId{4336}{\htmlClass{InductiveConstructor}{\text{Guard}}}}\, \\ \,\htmlId{4344}{\htmlClass{Symbol}{\text{\_}}}\, \end{pmatrix}$ =  record base { txInfoSubTxs = just subInfos }
                 where
                 base : TxInfo
                 base = txInfo TxLevelTop utxo tx
                 subInfos : List SubTxInfo
                 subInfos = map (txInfo TxLevelSub utxo) (SubTransactionsOf tx)
   -- · other top-level scripts see no subTx infos
... | _ = txInfo TxLevelTop utxo tx

    credsNeeded : UTxO → Tx ℓ → ℙ (ScriptPurpose × Credential)
credsNeeded utxo tx =
    mapˢ        (λ (i , o) → ($\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#3028}{\htmlId{5409}{\htmlClass{InductiveConstructor}{\text{Spend}}}}\,  \\ \,\href{Ledger.Dijkstra.Specification.Script.Validation.html#5397}{\htmlId{5418}{\htmlClass{Bound}{\text{i}}}}\, \end{pmatrix}$ , payCred (proj₁ o)))     ((utxo ∣ (SpendInputsOf tx ∪ collateralInputs tx)) ˢ)
  ∪ mapˢ        (λ a       → ($\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#3044}{\htmlId{5536}{\htmlClass{InductiveConstructor}{\text{Reward}}}}\, \\ \,\href{Ledger.Dijkstra.Specification.Script.Validation.html#5523}{\htmlId{5545}{\htmlClass{Bound}{\text{a}}}}\, \end{pmatrix}$ , CredentialOf a))        (dom ∣ WithdrawalsOf tx ∣)
  ∪ mapPartial  (λ c       → (($\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#3039}{\htmlId{5637}{\htmlClass{InductiveConstructor}{\text{Cert}}}}\,  \\ \,\href{Ledger.Dijkstra.Specification.Script.Validation.html#5623}{\htmlId{5645}{\htmlClass{Bound}{\text{c}}}}\, \end{pmatrix}$ ,_) <$> cwitness c))      (fromList (DCertsOf tx))
  ∪ mapˢ        (λ x       → ($\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#3034}{\htmlId{5734}{\htmlClass{InductiveConstructor}{\text{Mint}}}}\,   \\ \,\href{Ledger.Dijkstra.Specification.Script.Validation.html#5721}{\htmlId{5743}{\htmlClass{Bound}{\text{x}}}}\, \end{pmatrix}$ , ScriptObj x))           (policies (MintedValueOf tx))
  ∪ mapˢ        (λ v       → ($\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#3051}{\htmlId{5837}{\htmlClass{InductiveConstructor}{\text{Vote}}}}\,   \\ \,\href{Ledger.Dijkstra.Specification.Script.Validation.html#5824}{\htmlId{5846}{\htmlClass{Bound}{\text{v}}}}\, \end{pmatrix}$ , govVoterCredential v))  (fromList (map GovVoterOf (ListOfGovVotesOf tx)))
  ∪ mapPartial  (λ p       →  if PolicyOf p
                                 then (λ {sh} → just ($\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#3056}{\htmlId{6028}{\htmlClass{InductiveConstructor}{\text{Propose}}}}\, \\ \,\href{Ledger.Dijkstra.Specification.Script.Validation.html#5947}{\htmlId{6038}{\htmlClass{Bound}{\text{p}}}}\, \end{pmatrix}$ , ScriptObj sh))
                                 else nothing)                           (fromList (ListOfGovProposalsOf tx))
  ∪ mapˢ        (λ c       → ($\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#3064}{\htmlId{6203}{\htmlClass{InductiveConstructor}{\text{Guard}}}}\, \\ \,\href{Ledger.Dijkstra.Specification.Script.Validation.html#6190}{\htmlId{6211}{\htmlClass{Bound}{\text{c}}}}\, \end{pmatrix}$ , c))                      (GuardsOf tx)

  where
    collateralInputs : Tx ℓ → ℙ TxIn
    collateralInputs {TxLevelTop} tx = CollateralInputsOf tx
    collateralInputs {TxLevelSub} tx = ∅

allCredsNeeded : UTxO → TopLevelTx → ℙ (ScriptPurpose × Credential) 
allCredsNeeded utxo txTop =
  foldl (λ acc txSub → acc ∪ credsNeeded utxo txSub)
        (credsNeeded utxo txTop) (SubTransactionsOf txTop)

    CIP-159 and credsNeeded. CIP-159 does not introduce new ScriptPurpose values for direct deposits or balance intervals, so credsNeeded requires no changes. Direct deposits do not require a witness from the receiving credential. Balance intervals are Phase-1 validity conditions that do not trigger script execution. The existing Reward case already handles partial withdrawals; it generates a Reward script purpose for each entry in txWithdrawals, regardless of whether the withdrawal is full or partial.

    The function collectP2ScriptsWithContext.{AgdaFunction} builds a list of phase-2 scripts paired with their contexts for phase-2 validation. The scripts that are needed for validation are retrieved from the transaction using the function credsNeeded.

    In Dijkstra, the execution of a guard script can require several (including none) data. The function assembleData accounts for this situation by returning a list of lists of data (a list of data is part of the context of a script).

      collectP2ScriptsWithContext
    : PParams
    → Tx ℓ
    → UTxO
    → ℙ Script
    → List (P2Script × List Data × ExUnits × CostModel)
  collectP2ScriptsWithContext pp tx utxo allScripts
    = concat (setToList (mapˢ toScript (credsNeeded utxo tx)))
    where
      context : ScriptPurpose → Data
      context sp = valContext (txInfoForPurpose utxo tx sp) sp

      getScript : Credential → ScriptPurpose → Maybe (P2Script × Redeemer × ExUnits × CostModel)
      getScript c sp = do
             script               ← isScriptObj c >>= λ sh → lookupHash sh allScripts >>= toP2Script
             (reedemer , exUnits) ← indexedRdmrs tx sp
             costModel            ← lookupᵐ? (PParams.costmdls pp) (language script)
             return (script , reedemer , exUnits , costModel)

      assembleData : Redeemer → ScriptPurpose → List (List Data)
      assembleData redeemer sp@($\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#3028}{\htmlId{8603}{\htmlClass{InductiveConstructor}{\text{Spend}}}}\, \\ \,\href{Ledger.Dijkstra.Specification.Script.Validation.html#8611}{\htmlId{8611}{\htmlClass{Bound}{\text{txin}}}}\, \end{pmatrix}$)
        = [ fromMaybe (getDatumSpend tx utxo txin) ++ (redeemer ∷ [ context sp ]) ]
      assembleData redeemer sp           = [ redeemer ∷ [ context sp ] ]

      toScript
        : ScriptPurpose × Credential
        → List (P2Script × List Data × ExUnits × CostModel)
      toScript (sp , c) =
        do (script , reedemer , exUnits , costModel) ← fromMaybe (getScript c sp)
           data′ ← assembleData reedemer sp
           [ (script , data′ , exUnits , costModel) ]

evalP2Scripts : List (P2Script × List Data × ExUnits × CostModel) → Bool
evalP2Scripts = all (λ (s , d , eu , cm) → ¿ validP2Script cm d eu s ¿ᵇ)
