    data ScriptPurpose : Type where
  Cert     : DCert                ScriptPurpose
  Rwrd     : RewardAddress        ScriptPurpose
  Mint     : ScriptHash           ScriptPurpose
  Spend    : TxIn                 ScriptPurpose
  Vote     : GovVoter             ScriptPurpose
  Propose  : GovProposal          ScriptPurpose
  Guard    : (TxId  ScriptHash)  ScriptPurpose

    private variable
   : TxLevel

rdptr : (Tx )  ScriptPurpose  Maybe (RedeemerPtr )
rdptr tx =  where
  (Cert h)               map (Cert    ,_) $ indexOfDCert    h txCerts
  (Rwrd h)               map (Reward  ,_) $ indexOfRewardAddress  h txWithdrawals
  (Mint h)               map (Mint    ,_) $ indexOfPolicyId h (policies mint)
  (Spend h)              map (Spend   ,_) $ indexOfTxIn     h txIns
  (Vote h)               map (Vote    ,_) $ indexOfVote     h (map GovVote.voter txGovVotes)
  (Propose h)            map (Propose ,_) $ indexOfProposal h txGovProposals
  (Guard h)              map (Guard   ,_) $ indexOfGuard    h (getTxScripts tx)
 where open TxBody (TxBodyOf tx)
-- getSubTxScripts : TopLevelTx   (TxId  ScriptHash)

indexedRdmrs : (Tx )  ScriptPurpose  Maybe (Redeemer  ExUnits)
indexedRdmrs tx sp = maybe ( x  lookup? txRedeemers x) nothing (rdptr tx sp)
  where open Tx tx; open TxWitnesses txWitnesses

getDatum : Tx   UTxO  ScriptPurpose  Maybe Datum
getDatum tx utxo (Spend txin) =
  do (_ , _ , just d , _)  lookup? utxo txin where
                            (_ , _ , nothing , _)  nothing
     case d of  where
       (inj d)  just d
       (inj h)  lookup? (setToMap (map < hash , id > (DataOf tx))) h
getDatum tx utxo _ = nothing

    record TxInfo : Type where
  field realizedInputs : UTxO
        txOuts         : Ix  TxOut
        txFee          : Maybe Fees
        mint           : Value
        txCerts        : List DCert
        txWithdrawals  : Withdrawals
        txVldt         : Maybe Slot  Maybe Slot
        vkKey          :  KeyHash
        txData         :  Datum
        txId           : TxId


txInfo : ( : TxLevel)  UTxO  Tx   TxInfo

txInfo TxLevelTop utxo tx =
  record  { realizedInputs = utxo  txIns
          ; txOuts = txOuts
          ; txFee = just txFee
          ; mint = mint
          ; txCerts = txCerts
          ; txWithdrawals = txWithdrawals
          ; txVldt = txVldt
          ; vkKey = txRequiredGuards -- ?
          ; txData = DataOf tx
          ; txId = txId
          } where open Tx tx; open TxBody txBody

txInfo TxLevelSub utxo tx =
  record  { realizedInputs = utxo  txIns
          ; txOuts = txOuts
          ; txFee = nothing
          ; mint = mint
          ; txCerts = txCerts
          ; txWithdrawals = txWithdrawals
          ; txVldt = txVldt
          ; vkKey = txRequiredGuards  -- ?
          ; txData = DataOf tx
          ; txId = txId
          } where open Tx tx; open TxBody txBody

    credsNeededMinusCollateral : { : TxLevel}  TxBody    (ScriptPurpose  Credential)
credsNeededMinusCollateral txb = a  b  c  d  e
  where
  a b c d e :  (ScriptPurpose  Credential)
  a = map ( a  (Rwrd a , CredentialOf a)) (dom  WithdrawalsOf txb )
  b = mapPartial ( c  (Cert c ,_) <$> cwitness c) (fromList (DCertsOf txb))
  c = map ( x  (Mint x , ScriptObj x)) (policies (ValueOf txb))
  d = mapPartial ( v  if isGovVoterCredential v then ( {c}  just (Vote v , c)) else nothing)
                 (fromList (map GovVoterOf (GovVotesOf txb)))
  e = mapPartial ( p  if PolicyOf p then ( {sh}  just (Propose  p , ScriptObj sh)) else nothing)
                 (fromList (GovProposalsOf txb))

credsNeeded : ( : TxLevel)  UTxO  (TxBody )   (ScriptPurpose  Credential)
credsNeeded TxLevelTop utxo txb = credsNeededMinusCollateral txb
   map ( (i , o)  (Spend  i , payCred (proj o))) ((utxo  (txIns  collateralInputs)) )
  where open TxBody txb

credsNeeded TxLevelSub utxo txb = credsNeededMinusCollateral txb
   map ( (i , o)  (Spend  i , payCred (proj o))) ((utxo  txIns) )
  where open TxBody txb
