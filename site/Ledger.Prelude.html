-- Ledger prelude
--
-- Re-exports modules relating to STS, set theory and other
-- miscellaneous things used to write the ledger rules. If something
-- is used in more than two Ledger.* modules, it should probably go
-- here.
--------------------------------------------------------------------------------

dec-de-morgan : {P Q : Type}   P     (P  Q)   P   Q
dec-de-morgan   no p  pq = inj p
dec-de-morgan   yes p  pq = inj  q  pq (p , q)

-getCoin :  {A}   _ : DecEq A   (s s' : A  Coin)  s   s'   getCoin s  getCoin s'
-getCoin {A}  decEqA  s s' ss' = indexedSum'-cong {C = Coin} {x = s} {y = s'} ss'

setToMap :  {A B : Type}   DecEq A    (A  B)  A  B
setToMap = fromList  setToList

-- division of natural numbers with completion by 0.
opaque
  _/_ :     
  x / 0 = .0
  x / y@(suc _) = .+ x ./ y

-- Division of rational numbers with completion by 0.
__ :     
__ x y = case y  .0 of  where
  (yes _)  .0
  (no y0)  let instance nonZero-y = .-nonZero y0 in x . y

_ :  {A B : Type}  f : HasCast A B   A  B
_  f = f  = f.cast
  where module f = HasCast f

_ :  {A B : Type}  f : HasCast A B   A  B
_  f = f  = f.cast
  where module f = HasCast f

infix 5 _
infix 6 _

module Filter where
  filter :  {a} {p} {A : Type a}  (P : Pred A p)   P    List A  List A
  filter P = Data.List.filter  P 

lookupHash :  {T H : Type}  _ : DecEq H   _ : Hashable T H   H   T  Maybe T
lookupHash h s =
  if h  map proj (m ) then just (lookup m h) else nothing
  where m = setToMap (map < hash , id > s)

Is- : {A : Type}   A  Type
Is- X = Is-[] (setToList X)
