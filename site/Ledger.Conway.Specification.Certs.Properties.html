    let open DelegEnv de; open DState st in
     where
    (delegate c mv mc d)  case  (c  dom rewards  d  pparams .PParams.keyDeposit)
                                 (c  dom rewards  d  0)
                                 mv  map (just  vDelegCredential) delegatees

                                 mc  map just (dom pools)   nothing

      (yes p)  success (-, DELEG-delegate p)
      (no p)  failure (genErrors p)
    (dereg c d)  case  (c , 0)  rewards  of  where
      (yes p)  success (-, DELEG-dereg p)
      (no p)  failure (genErrors p)
    (reg c d)  case  c  dom rewards
                      (d  pparams .PParams.keyDeposit  d  0)
                      of  where

    s' (DELEG-delegate p) rewrite dec-yes ( (c  dom (DState.rewards st)  d  DelegEnv.pparams de .PParams.keyDeposit)

                              ( .trans (-cong .refl (res-singleton'{m = rwds} x))
                                         (.sym $ disjoint-- disj) )
  CERT-pov (CERT-pool x) = refl
  CERT-pov (CERT-vdel x) = refl
