-- ** Proof that LEDGER preserves values.

  HasCoin-LState : HasCoin LState
  HasCoin-LState .getCoin s = getCoin (LState.utxoSt s) + getCoin (LState.certState s)

    Theorem (LEDGER rule preserves value).

    Informally.

    Let s and s' be ledger states and let tx : Tx be a fresh transaction, that is, a transaction that is not already part of the UTxOState of s. If s tx ,LEDGER s', then the coin values of s and s' are equal, that is, getCoin s \(\) getCoin s'.

    Formally.

      LEDGER-pov : { : LEnv} {s s' : LState}
     txId  map proj (dom (UTxOOf s))
       s  tx ,LEDGER s'  getCoin s  getCoin s'

    Proof.

      LEDGER-pov
    {s  = s}
    {s' = s'}
    h (LEDGER-V {utxoSt' = utxoSt''} ( valid , UTXOWUTXO st@(UTXO-induction r) , h' , _ )) =
    let
      open LState s
      open CertState certState
      open LState s' renaming (utxoSt to utxoSt'; govSt to govSt'; certState to certState')
      open CertState certState'
      open -Reasoning
      open CERTSpov indexedSum'- sumConstZero res-decomp  getCoin-cong -getCoin r
      zeroMap    = constMap (map RwdAddr.stake (dom txWithdrawals)) 0
    in
    begin
      getCoin utxoSt + getCoin certState
         cong (getCoin utxoSt +_) (CERTS-pov h') 
      getCoin utxoSt + (getCoin certState' + getCoin txWithdrawals)
         cong ( x  getCoin utxoSt + (getCoin certState' + x )) (*-identity (getCoin txWithdrawals))   -- cong ( u  getCoin utxoSt + (getCoin certState' + getCoin txWithdrawals *  u)) valid 
      getCoin utxoSt + (getCoin certState' + getCoin txWithdrawals * 1)
         cong ( u  getCoin utxoSt + (getCoin certState' + getCoin txWithdrawals *  u)) valid 
      getCoin utxoSt + (getCoin certState' + getCoin txWithdrawals *  isValid)
         cong (getCoin utxoSt +_) (+-comm (getCoin certState') _) 
      getCoin utxoSt + (getCoin txWithdrawals *  isValid + getCoin certState')
         +-assoc (getCoin utxoSt) (getCoin txWithdrawals *  isValid) (getCoin certState') 
      getCoin utxoSt + getCoin txWithdrawals *  isValid + getCoin certState'
         cong (_+ getCoin certState') (UTXOpov h st) 
      getCoin utxoSt' + getCoin certState'
