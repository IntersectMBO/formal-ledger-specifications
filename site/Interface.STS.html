module Interface.STS where

open import Prelude

open import Prelude.InferenceRules public

private
  variable C S Sig : Type
            : C
           s s' s'' : S
           sig : Sig
           sigs : List Sig
           n : 

-- small-step to big-step transformer

module _ {____ : C  S    S  Type} {____ : C  S  Sig  S  Type} where
  data ___*_ : C  S  List Sig  S  Type where

    BS-base :
        s  _  s'
      
        s  [] * s'

    BS-ind :
          s   sig    s'
         s'  sigs * s''
        
          s   sig  sigs * s''

module _ {____ : C  S    S  Type} {____ : C    S  Sig  S  Type} where
  data ___*'_ : C    S  List Sig  S  Type where

    BS-base :
        s  _  s'
      
      ( , n)  s  [] *' s'

    BS-ind :
        ( , n)      s   sig    s'
       ( , suc n)  s'  sigs *' s''
        
        ( , n)      s   sig  sigs *' s''

  ___*_ : C  S  List Sig  S  Type
  ___*_  = ___*'_ ( , 0)

data IdSTS {C S} : C  S    S  Type where
  Id-nop : IdSTS  s _ s

module _ {____ : C  S    S  Type} {____ : C  S  Sig  S  Type} where
  data ___*'_ : C  S  List Sig  S  Type where
      RTC :
             s  _  s'
           ___*_ {____ = IdSTS}{____}  s' sigs s''
          
            s  sigs *' s''


-- with a trivial base case
ReflexiveTransitiveClosure : {sts : C  S  Sig  S  Type}  C  S  List Sig  S  Type
ReflexiveTransitiveClosure {sts = sts} = ___*_ {____ = IdSTS}{sts}

STS-total : (C  S  Sig  S  Type)  Type
STS-total ____ =  { s sig}  [ s' ]   s  sig  s'

ReflexiveTransitiveClosure-total : {____ : C  S  Sig  S  Type}
   STS-total ____  STS-total (ReflexiveTransitiveClosure {sts = ____})
ReflexiveTransitiveClosure-total SS-total {} {s} {[]} = s , BS-base Id-nop
ReflexiveTransitiveClosure-total SS-total {} {s} {x  sig} =
  case SS-total of  where
    (s' , Ps')  map (BS-ind Ps') $ ReflexiveTransitiveClosure-total SS-total

ReflexiveTransitiveClosure : {sts : C    S  Sig  S  Type}  C  S  List Sig  S  Type
ReflexiveTransitiveClosure {sts = sts} = ___*_ {____ = IdSTS}{sts}

ReflexiveTransitiveClosure-total : {____ : C    S  Sig  S  Type}
   STS-total ____  STS-total (ReflexiveTransitiveClosure {sts = ____})
ReflexiveTransitiveClosure-total SS-total = helper SS-total
  where
    helper : {____ : C    S  Sig  S  Type}
       STS-total ____  STS-total (___*'_ {____ = IdSTS}{____})
    helper SS-total {s = s} {[]} = s , BS-base Id-nop
    helper SS-total {s = s} {x  sig} =
      case SS-total of  where
        (s' , Ps')  map (BS-ind Ps') $ helper SS-total

LedgerInvariant : (C  S  Sig  S  Type)  (S  Type)  Type
LedgerInvariant STS P =  {c s sig s'}  STS c s sig s'  P s  P s'

RTC-preserves-inv :  {STS : C  S  Sig  S  Type} {P}
                   LedgerInvariant STS P  LedgerInvariant (ReflexiveTransitiveClosure {sts = STS}) P
RTC-preserves-inv inv (BS-base Id-nop) = id
RTC-preserves-inv inv (BS-ind p p)   = RTC-preserves-inv inv p  inv p
