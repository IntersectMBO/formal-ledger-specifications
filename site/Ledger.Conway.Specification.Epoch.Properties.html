module _ {lstate : LState} {ss : Snapshots} where
  SNAP-total : [ ss' ] lstate  ss  tt ,SNAP ss'
  SNAP-total = -, SNAP

  SNAP-complete :  ss'  lstate  ss  tt ,SNAP ss'  proj SNAP-total  ss'
  SNAP-complete ss' SNAP = refl

  SNAP-deterministic :  {ss' ss''}
                      lstate  ss  tt ,SNAP ss'
                      lstate  ss  tt ,SNAP ss''  ss'  ss''
  SNAP-deterministic SNAP SNAP = refl

module _ {e : Epoch} (prs : PoolReapState) where
  POOLREAP-total : [ prs' ] _  prs  e ,POOLREAP prs'
  POOLREAP-total = -, POOLREAP

  POOLREAP-complete
    :  prs'  _  prs  e ,POOLREAP prs'  proj POOLREAP-total  prs'
  POOLREAP-complete prs' POOLREAP = refl

  POOLREAP-deterministic
    :  {prs' prs''}
     _  prs  e ,POOLREAP prs'
     _  prs  e ,POOLREAP prs''
     prs'  prs''
  POOLREAP-deterministic POOLREAP POOLREAP = refl

module _ {eps : EpochState} {e : Epoch} where

  open EpochState eps hiding (es)

  prs = $\begin{pmatrix} \,\href{Ledger.Conway.Specification.Epoch.Properties.html#2110}{\htmlId{1957}{\htmlClass{Function}{\text{u0}}}}\, \,\htmlId{1960}{\htmlClass{Symbol}{\text{.}}}\,\,\href{Ledger.Conway.Specification.Epoch.html#17313}{\htmlId{1961}{\htmlClass{Field}{\text{utxoSt'}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.html#2174}{\htmlId{1971}{\htmlClass{Function}{\text{acnt}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.html#2085}{\htmlId{1978}{\htmlClass{Function}{\text{cs}}}}\, \,\htmlId{1981}{\htmlClass{Symbol}{\text{.}}}\,\,\href{Ledger.Conway.Specification.Certs.html#5175}{\htmlId{1982}{\htmlClass{Field}{\text{dState}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.html#2085}{\htmlId{1991}{\htmlClass{Function}{\text{cs}}}}\, \,\htmlId{1994}{\htmlClass{Symbol}{\text{.}}}\,\,\href{Ledger.Conway.Specification.Certs.html#5195}{\htmlId{1995}{\htmlClass{Field}{\text{pState}}}}\, \end{pmatrix}$
    where
      open LState
      open CertState
      open EPOCH-Updates0
      cs = ls .certState
      u0 = EPOCH-updates0 fut ls

  EPOCH-total : [ eps' ] _  eps  e ,EPOCH eps'
  EPOCH-total =
    -, EPOCH
         ( SNAP-total .proj
         , RATIFIES-total' .proj
         , POOLREAP-total prs .proj)

  private
    EPOCH-state : Snapshots  RatifyState  PoolReapState  EpochState
    EPOCH-state ss fut' ($\begin{pmatrix} \,\href{Ledger.Conway.Specification.Epoch.Properties.html#2429}{\htmlId{2429}{\htmlClass{Bound}{\text{utxoSt''}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.html#2440}{\htmlId{2440}{\htmlClass{Bound}{\text{acnt'}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.html#2448}{\htmlId{2448}{\htmlClass{Bound}{\text{dState'}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.html#2458}{\htmlId{2458}{\htmlClass{Bound}{\text{pState'}}}}\, \end{pmatrix}$) =
      let
        EPOCHUpdates es govSt' dState'' gState' _ acnt'' =
          EPOCH-updates fut ls dState' acnt'
        certState' = $\begin{pmatrix} \,\href{Ledger.Conway.Specification.Epoch.Properties.html#2513}{\htmlId{2609}{\htmlClass{Bound}{\text{dState''}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.html#2458}{\htmlId{2620}{\htmlClass{Bound}{\text{pState'}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.html#2522}{\htmlId{2630}{\htmlClass{Bound}{\text{gState'}}}}\, \end{pmatrix}$
       in
          record
            { acnt = acnt''
            ; ss = ss
            ; ls = $\begin{pmatrix} \,\href{Ledger.Conway.Specification.Epoch.Properties.html#2429}{\htmlId{2740}{\htmlClass{Bound}{\text{utxoSt''}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.html#2506}{\htmlId{2751}{\htmlClass{Bound}{\text{govSt'}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.html#2594}{\htmlId{2760}{\htmlClass{Bound}{\text{certState'}}}}\, \end{pmatrix}$
            ; es = es
            ; fut = fut'
            }

  EPOCH-deterministic :  eps' eps''
                       _  eps  e ,EPOCH eps'
                       _  eps  e ,EPOCH eps''
                       eps'  eps''
  EPOCH-deterministic
      eps'
      eps''
      (EPOCH
        {utxoSt'' = utxoSt''}
        {acnt' = acnt'}
        {dState' = dState'}
        {pState' = pState'}
        (p , p , p)
      )
      (EPOCH
        {utxoSt'' = utxoSt''}
        {acnt' = acnt'}
        {dState' = dState'}
        {pState' = pState'}
        (p' , p' , p')
      ) =
    cong _$_ (cong EPOCH-state ss'ss'' fut'fut'') prs'prs''
    where
      ss'ss'' : EpochState.ss eps'  EpochState.ss eps''
      ss'ss'' = SNAP-deterministic p p'

      fut'fut'' : EpochState.fut eps'  EpochState.fut eps''
      fut'fut'' = RATIFIES-deterministic-
                    (cong ( x  record
                                   { stakeDistrs = mkStakeDistrs (Snapshots.mark x) _ _ _ _ _
                                   ; currentEpoch = _
                                   ; dreps = _
                                   ; ccHotKeys = _
                                   ; treasury = _
                                   }) ss'ss'')
                                   refl refl p p'

      prs'prs'' : $\begin{pmatrix} \,\href{Ledger.Conway.Specification.Epoch.Properties.html#3089}{\htmlId{4118}{\htmlClass{Bound}{\text{utxoSt''}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.html#3117}{\htmlId{4130}{\htmlClass{Bound}{\text{acnt'}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.html#3144}{\htmlId{4139}{\htmlClass{Bound}{\text{dState'}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.html#3173}{\htmlId{4150}{\htmlClass{Bound}{\text{pState'}}}}\, \end{pmatrix}$ 
                   $\begin{pmatrix} \,\href{Ledger.Conway.Specification.Epoch.Properties.html#3247}{\htmlId{4185}{\htmlClass{Bound}{\text{utxoSt''}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.html#3275}{\htmlId{4197}{\htmlClass{Bound}{\text{acnt'}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.html#3302}{\htmlId{4206}{\htmlClass{Bound}{\text{dState'}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.html#3331}{\htmlId{4217}{\htmlClass{Bound}{\text{pState'}}}}\, \end{pmatrix}$
      prs'prs'' = POOLREAP-deterministic prs p p'

  EPOCH-complete :  eps'  _  eps  e ,EPOCH eps'  proj EPOCH-total  eps'
  EPOCH-complete eps' p = EPOCH-deterministic (proj EPOCH-total) eps' (proj EPOCH-total) p

  abstract
    EPOCH-total' : [ eps' ] _  eps  e ,EPOCH eps'
    EPOCH-total' = EPOCH-total

    EPOCH-complete' :  eps'  _  eps  e ,EPOCH eps'  proj EPOCH-total'  eps'
    EPOCH-complete' = EPOCH-complete

instance
  Computational-EPOCH : Computational ___,EPOCH_ 
  Computational-EPOCH .computeProof  s sig = success EPOCH-total'
  Computational-EPOCH .completeness  s sig s' h = cong success (EPOCH-complete' s' h)

  NEWEPOCH-total :  nes''  [ nes' ] _  nes''  e ,NEWEPOCH nes'
  NEWEPOCH-total nes with e  NewEpochState.lastEpoch nes + 1 | NewEpochState.ru nes | inspect NewEpochState.ru nes
  ... | yes p | just ru | PE.[ refl ] =  $\begin{pmatrix} \,\href{Ledger.Conway.Specification.Epoch.Properties.html#4907}{\htmlId{5154}{\htmlClass{Bound}{\text{e}}}}\, \\ \,\htmlId{5158}{\htmlClass{Symbol}{\text{\_}}}\, \\ \,\htmlId{5162}{\htmlClass{Symbol}{\text{\_}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.html#4472}{\htmlId{5166}{\htmlClass{Function}{\text{EPOCH-total'}}}}\, \,\htmlId{5179}{\htmlClass{Symbol}{\text{.}}}\,\,\href{Data.Product.Base.html#636}{\htmlId{5180}{\htmlClass{Field}{\text{proj}}}}\, \\ \,\href{Agda.Builtin.Maybe.html#194}{\htmlId{5188}{\htmlClass{InductiveConstructor}{\text{nothing}}}}\, \\ \,\htmlId{5198}{\htmlClass{Symbol}{\text{\_}}}\, \end{pmatrix}$
                                      , NEWEPOCH-New (p , EPOCH-total' .proj)
  ... | yes p | nothing | PE.[ refl ] = $\begin{pmatrix} \,\href{Ledger.Conway.Specification.Epoch.Properties.html#4907}{\htmlId{5323}{\htmlClass{Bound}{\text{e}}}}\, \\ \,\htmlId{5327}{\htmlClass{Symbol}{\text{\_}}}\, \\ \,\htmlId{5331}{\htmlClass{Symbol}{\text{\_}}}\, \\ \,\href{Data.Product.Base.html#636}{\htmlId{5335}{\htmlClass{Field}{\text{proj}}}}\, \,\href{Ledger.Conway.Specification.Epoch.Properties.html#4472}{\htmlId{5341}{\htmlClass{Function}{\text{EPOCH-total'}}}}\, \\ \,\href{Agda.Builtin.Maybe.html#194}{\htmlId{5356}{\htmlClass{InductiveConstructor}{\text{nothing}}}}\, \\ \,\htmlId{5366}{\htmlClass{Symbol}{\text{\_}}}\, \end{pmatrix}$
                                      , NEWEPOCH-No-Reward-Update (p , EPOCH-total' .proj)
  ... | no p | _ | _ = -, NEWEPOCH-Not-New p

  NEWEPOCH-complete :  nes nes'  _  nes  e ,NEWEPOCH nes'  proj (NEWEPOCH-total nes)  nes'
  -- NEWEPOCH-complete nes nes' h with e  NewEpochState.lastEpoch nes + 1 | NewEpochState.ru nes | h
  NEWEPOCH-complete nes nes' h with e  NewEpochState.lastEpoch nes + 1 | NewEpochState.ru nes | inspect NewEpochState.ru nes | h
  ... | yes p | just ru | PE.[ refl ] | NEWEPOCH-New (x , x) rewrite EPOCH-complete' _ x = refl
  ... | yes p | ru | PE.[ refl ] | NEWEPOCH-Not-New x = -elim $ x p
  ... | yes p | nothing | PE.[ refl ] | NEWEPOCH-No-Reward-Update (x , x) rewrite EPOCH-complete' _ x = refl
  ... | no p | ru | PE.[ refl ] | NEWEPOCH-New (x , x)  = -elim $ p x
  ... | no p | ru | PE.[ refl ] | NEWEPOCH-Not-New x = refl
  ... | no p | nothing | PE.[ refl ] | NEWEPOCH-No-Reward-Update (x , x) = -elim $ p x
