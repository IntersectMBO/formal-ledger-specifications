module EPOCH-Body (eps : EpochState) where
  open EpochState eps hiding (es) renaming (ls to epsLState; fut to epsRState) public
  open RatifyState renaming (es to ensRState) public
  open LState epsLState public
  open PState public
  open GovActionState public

  ens      = record (epsRState .ensRState) { withdrawals =  }
  tmpGovSt = filter ( x   proj x  map proj (epsRState .removed) ) govSt
  orphans  = fromList $ getOrphans (epsRState .ensRState) tmpGovSt
  removed' :  (GovActionID  GovActionState)
  removed' = (epsRState .removed)  orphans
  removedGovActions = flip concatMap removed'  (gaid , gaSt) 
    map (returnAddr gaSt ,_) ((DepositsOf utxoSt   GovActionDeposit gaid ) )

module EPOCH-PROPS {eps : EpochState} where
  open EPOCH-Body eps

    Lemma (govDepsMatch is invariant of EPOCH rule).

    Informally.

    Let eps, eps' : EpochState be two epoch states and let e : Epoch be an epoch. Recall, LStateOf eps gives the ledger state of eps. If eps e ,EPOCH eps', then (under a certain special condition) govDepsMatch (LStateOf eps) implies govDepsMatch (LStateOf eps').

    The special condition under which the property holds is the same as the one in Chain.Properties.GovDepsMatch: let removed' be the union of the governance actions in the removed field of the ratify state of eps and the orphaned governance actions in the GovState of eps.

    For the formal statement of the lemma,

    • let \(\) be the set \(\{\)GovActionDeposit id : id \(\) proj removed'\(\}\), and

    • assume \(\) is a subset of the set of deposits of (the governance state of) eps.

    Formally.

      EPOCH-govDepsMatch :  {eps' : EpochState} {e : Epoch}
     map (GovActionDeposit  proj) removed'  dom (DepositsOf eps)
     _  eps  e ,EPOCH eps'
     govDepsMatch (LStateOf eps)  govDepsMatch (LStateOf eps')

    Proof.

      EPOCH-govDepsMatch {eps'} {e} ratify-removed (EPOCH (x , POOLREAP , _)) =
      poolReapMatch  ratifiesSnapMatch
    where

    -- the combinator used in the EPOCH rule
     :  DepositPurpose
     = map (proj  proj) removedGovActions

    -- a simpler combinator that suffices here;
    ' :  DepositPurpose
    ' = map (GovActionDeposit  proj) removed'
    -- Below we prove  and ' are essentially equivalent.

    P : GovActionID  GovActionState  Type
    P =  u  proj u  map proj removed'

    P? : Decidable P
    P? =  u   P u 

    utxoDeps : Deposits
    utxoDeps = UTxOState.deposits (LState.utxoSt epsLState)

    -- utxo deposits restricted to new form of set used in EPOCH rule
    utxoDeps' : Deposits
    utxoDeps' = utxoDeps  ' 

    ' : '  
    ' = ' , '
      where
      ' : '  
      ' {a} x with from -map x
      ... | (gaid , gast) , refl , gaidgastrem with from -map (ratify-removed x)
      ... | (dp , c) , refl , dpcutxoDeps = let gadc = (GovActionDeposit gaid , c) in
        to -map ((returnAddr {txs} gast , gadc)
                 , refl
                 , to -concatMap ((gaid , gast)
                                   , gaidgastrem
                                   , to -map (gadc , refl , res-singleton {m = utxoDeps} dpcutxoDeps)))
      ' :   '
      ' {a} x with from -map x
      ... | (rwa , dp , c) , refl , rwa-dp-c with (from -concatMap rwa-dp-c)
      ... | (gaid , gast) , gaid-gast--removed , rwa-dp-c--map with (from -map rwa-dp-c--map)
      ... | (_ , _) , refl , q =
        to -map ((gaid , gast)
                 , proj (-, (proj (res-singleton'' {m = utxoDeps} q)))
                 , gaid-gast--removed)

    map-filter-decomp :  a  (a  '  a  map' (GovActionDeposit  proj) govSt)
                                (a  map' (GovActionDeposit  proj)(filter P? govSt))
    map-filter-decomp a = mk i ( h  ii h , iii h)
      where
      i : ((a  ')  (a  map' (GovActionDeposit  proj) govSt))
           a  map' (GovActionDeposit  proj) (filter P? govSt)
      i (a' , a) with Inverse.from (map- (GovActionDeposit  proj)) a
      ... | b , b , refl = Inverse.to (map- (GovActionDeposit  proj))
                                       (b , -filter P? b (a'  -map-) , refl)

      ii : a  map' (GovActionDeposit  proj) (filter P? govSt)  a  '
      ii a a' with from (-map-filter {l = govSt} {P? = P?}) a
      ... | _ , _ , refl , Pb with -map' a'
      ... | q , refl , qrem = Pb (to -map (q , refl , qrem))

      iii : a  map' (GovActionDeposit  proj) (filter P? govSt)
             a  map' (GovActionDeposit  proj) govSt
      iii a with from (-map-filter {l = govSt} {P? = P?}) a
      ... | b , b , refl , Pb = Inverse.to (map- (GovActionDeposit  proj)) (b , (b , refl))


    main-invariance-lemma :
        filter isGADeposit (dom utxoDeps) ' fromList (map' (GovActionDeposit  proj) govSt)
        ---------------------------------------------------------------------------------------------------
       filter isGADeposit (dom utxoDeps') ' fromList (map' (GovActionDeposit  proj) (filter P? govSt))

    main-invariance-lemma HYP a = let open R.EquationalReasoning in
      a  filter isGADeposit (dom utxoDeps')                           R.SK-sym -filter 
      (isGADeposit a  a  dom utxoDeps')                               R.K-refl -cong -res-dom 
      (isGADeposit a  a  '  [ q ] (a , q)  utxoDeps)              --swap 
      (a  '  isGADeposit a  [ q ] (a , q)  utxoDeps)              R.K-refl -cong (R.K-refl -cong dom)
      (a  '  isGADeposit a  a  dom utxoDeps)                       R.K-refl -cong -filter 
      (a  '  a  filter isGADeposit (dom utxoDeps))                 R.K-refl -cong (HYP a) 
      (a  '  a  fromList (map' (GovActionDeposit  proj) govSt))   R.K-refl -cong (R.SK-sym -fromList)
      (a  '  a  map' (GovActionDeposit  proj) govSt)             map-filter-decomp a 
      a  map' (GovActionDeposit  proj) (filter P? govSt)            -fromList 
      a  fromList (map' (GovActionDeposit  proj) (filter P? govSt)) 

    ls = record (LStateOf eps')
           { utxoSt = Pre-POOLREAPUpdate.utxoSt' (LStateOf eps) (EnactStateOf eps) (GovernanceUpdate.updates ((LStateOf eps)) ((RatifyStateOf eps))) }

    open LState
    open CertState

    retiredDeposits :  DepositPurpose
    retiredDeposits = map PoolDeposit ((PStateOf eps) .retiring  e)

    dPoolDepositA
      : (d : DepositPurpose)
       d  dom (DepositsOf ls  retiredDeposits)
       [ kh ] d  PoolDeposit kh
    dPoolDepositA d dres =
      Product.map proj $
        -map' $       -- ([ a ] d  PoolDeposit a  a  _)
         res-dom dres  -- d  retiredDeposits
      where import Data.Product.Base as Product using (map)

    ratifiesSnapMatch : govDepsMatch (LStateOf eps)  govDepsMatch ls
    ratifiesSnapMatch =
       .trans (filter-cong $ dom-cong (res-comp-cong $ .sym '))
        from '  main-invariance-lemma  to '

    noGADepositIsRetired
      : (d : DepositPurpose)
       d  dom (DepositsOf ls  retiredDeposits)
        isGADeposit d
    noGADepositIsRetired d dres dIsGA
     rewrite (proj $ dPoolDepositA d dres)
     with dIsGA
    ... | ()

    dropRetiredDeposits :
      filter isGADeposit (dom (DepositsOf ls  retiredDeposits )) 
        filter isGADeposit (dom (DepositsOf ls))
    dropRetiredDeposits = begin
      filter isGADeposit (dom (DepositsOf ls  retiredDeposits ))

         -identity _ 

        filter isGADeposit (dom (DepositsOf ls  retiredDeposits ))

         -cong
             (filter- noGADepositIsRetired)
             (IsEquivalence.refl -isEquivalence)
         

      filter isGADeposit (dom (DepositsOf ls  retiredDeposits))
      
      filter isGADeposit (dom (DepositsOf ls  retiredDeposits ))

         filter-hom- 

      filter isGADeposit
        (dom (DepositsOf ls  retiredDeposits)
         
         dom (DepositsOf ls  retiredDeposits )
        )

         filter-cong dom 

      filter isGADeposit
        (dom
          ((DepositsOf ls  retiredDeposits) 
            
           (DepositsOf ls  retiredDeposits ) 
          )
        )

         IsEquivalence.refl -isEquivalence 

      filter isGADeposit
        (Rel.dom
          (((DepositsOf ls )  retiredDeposits)
            
           ((DepositsOf ls )  retiredDeposits )
          )
        )

         filter-cong $ dom-cong (res-ex- dec) 

      filter isGADeposit (dom (DepositsOf ls))
      

      where
        open SetoidReasoning (-Setoid {A = DepositPurpose})
        open import Relation.Binary using (IsEquivalence)
        import Axiom.Set.Rel th as Rel

    poolReapMatch : govDepsMatch ls  govDepsMatch (LStateOf eps')
    poolReapMatch = .trans dropRetiredDeposits
