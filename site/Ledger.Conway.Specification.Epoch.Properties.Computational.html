module _ {eps : EpochState} {e : Epoch} where

  EPOCH-total : [ eps' ] _  eps  e ,EPOCH eps'
  EPOCH-total =
    -, EPOCH
         ( SNAP-total      .proj
         , POOLREAP-total  .proj
         , RATIFIES-total' .proj)

  EPOCH-deterministic :  eps' eps''
                       _  eps  e ,EPOCH eps'
                       _  eps  e ,EPOCH eps''
                       eps'  eps''
  EPOCH-deterministic
      eps'
      eps''
      (EPOCH
        {dState' = dState'}
        {acnt' = acnt'}
        {ss' = ss'}
        {utxoSt'' = utxoSt''}
        {pState'' = pState'}
        (p , p , p)
      )
      (EPOCH
        {dState' = dState'}
        {acnt' = acnt'}
        {ss' = ss'}
        {utxoSt'' = utxoSt''}
        {pState'' = pState'}
        (p' , p' , p')
      ) = eps'eps''
       where
         ls : LState
         ls = LStateOf eps

         es : EnactState
         es = EnactStateOf (RatifyStateOf eps)

         govUpd : Governance-Update
         govUpd = GovernanceUpdate.updates ls (RatifyStateOf eps)

         govSt' = Governance-Update.govSt' govUpd

         ss'ss' : ss'  ss'
         ss'ss' = SNAP-deterministic p p'

         module pPRUpd =  Pre-POOLREAP-Update (Pre-POOLREAPUpdate.updates ls es govUpd)

         pPRUpd = Post-POOLREAPUpdate.updates es ls dState' acnt' govUpd
         module pPRUpd = Post-POOLREAP-Update pPRUpd

         pPRUpd = Post-POOLREAPUpdate.updates es ls dState' acnt' govUpd
         module pPRUpd = Post-POOLREAP-Update pPRUpd

         prs'prs'' : $\begin{pmatrix} \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#1758}{\htmlId{2773}{\htmlClass{Bound}{\text{utxoSt''}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#1709}{\htmlId{2785}{\htmlClass{Bound}{\text{acnt'}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#1682}{\htmlId{2794}{\htmlClass{Bound}{\text{dState'}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#1789}{\htmlId{2805}{\htmlClass{Bound}{\text{pState'}}}}\, \end{pmatrix}$ 
                      $\begin{pmatrix} \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#1938}{\htmlId{2843}{\htmlClass{Bound}{\text{utxoSt''}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#1889}{\htmlId{2855}{\htmlClass{Bound}{\text{acnt'}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#1862}{\htmlId{2864}{\htmlClass{Bound}{\text{dState'}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#1969}{\htmlId{2875}{\htmlClass{Bound}{\text{pState'}}}}\, \end{pmatrix}$
         prs'prs'' = POOLREAP-deterministic- refl refl p p'

         pPRUpdpPRUpd : pPRUpd  pPRUpd
         pPRUpdpPRUpd rewrite (cong PoolReapState.dState prs'prs'') | (cong PoolReapState.acnt prs'prs'') = refl

         stakeDistrsstakeDistrs : mkStakeDistrs (Snapshots.mark ss') e pPRUpd.utxoSt' govSt' (GStateOf ls) (DStateOf ls)
                                      mkStakeDistrs (Snapshots.mark ss') e pPRUpd.utxoSt' govSt' (GStateOf ls) (DStateOf ls)
         stakeDistrsstakeDistrs = cong ( ss'  mkStakeDistrs (Snapshots.mark ss') e pPRUpd.utxoSt' govSt' (GStateOf ls) (DStateOf ls)) ss'ss'

         ' = cong ( sd acnt  $\begin{pmatrix} \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#3542}{\htmlId{3554}{\htmlClass{Bound}{\text{sd}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#1227}{\htmlId{3559}{\htmlClass{Bound}{\text{e}}}}\, \\ \,\href{Ledger.Conway.Specification.Certs.html#1900}{\htmlId{3563}{\htmlClass{Field}{\text{DRepsOf}}}}\, \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#2048}{\htmlId{3571}{\htmlClass{Function}{\text{ls}}}}\, \\ \,\href{Ledger.Conway.Specification.Certs.html#1815}{\htmlId{3576}{\htmlClass{Field}{\text{CCHotKeysOf}}}}\, \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#2048}{\htmlId{3588}{\htmlClass{Function}{\text{ls}}}}\, \\ \,\href{Ledger.Prelude.Base.html#985}{\htmlId{3593}{\htmlClass{Field}{\text{TreasuryOf}}}}\, \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#3545}{\htmlId{3604}{\htmlClass{Bound}{\text{acnt}}}}\, \\ \,\href{Ledger.Conway.Specification.Certs.html#1977}{\htmlId{3611}{\htmlClass{Field}{\text{PoolsOf}}}}\, \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#2048}{\htmlId{3619}{\htmlClass{Function}{\text{ls}}}}\, \\ \,\href{Ledger.Core.Specification.Gov.Actions.html#7072}{\htmlId{3624}{\htmlClass{Field}{\text{VoteDelegsOf}}}}\, \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#2048}{\htmlId{3637}{\htmlClass{Function}{\text{ls}}}}\, \end{pmatrix}$)
                      stakeDistrsstakeDistrs (cong Post-POOLREAP-Update.acnt'' pPRUpdpPRUpd)

         fut'fut'' : RatifyStateOf eps'  RatifyStateOf eps''
         fut'fut'' = RATIFIES-deterministic- ' refl refl p p'

         eps'eps'' : eps'  eps''
         eps'eps''
           rewrite ss'ss'
             | cong PoolReapState.utxoSt prs'prs''
             | cong PoolReapState.pState prs'prs''
             | cong Post-POOLREAP-Update.acnt'' pPRUpdpPRUpd
             | cong Post-POOLREAP-Update.dState'' pPRUpdpPRUpd
             | fut'fut'' = refl

  EPOCH-complete :  eps'  _  eps  e ,EPOCH eps'  proj EPOCH-total  eps'
  EPOCH-complete eps' p = EPOCH-deterministic (proj EPOCH-total) eps' (proj EPOCH-total) p

  abstract
    EPOCH-total' : [ eps' ] _  eps  e ,EPOCH eps'
    EPOCH-total' = EPOCH-total

    EPOCH-complete' :  eps'  _  eps  e ,EPOCH eps'  proj EPOCH-total'  eps'
    EPOCH-complete' = EPOCH-complete

instance
  Computational-EPOCH : Computational ___,EPOCH_ 
  Computational-EPOCH .computeProof  s sig = success EPOCH-total'
  Computational-EPOCH .completeness  s sig s' h = cong success (EPOCH-complete' s' h)

  ... | no p | _ | _ = -, NEWEPOCH-Not-New p

  NEWEPOCH-complete :  nes nes'  _  nes  e ,NEWEPOCH nes'  proj (NEWEPOCH-total nes)  nes'

  NEWEPOCH-complete nes nes' h with e  NewEpochState.lastEpoch nes + 1 | NewEpochState.ru nes | inspect NewEpochState.ru nes | h
  ... | yes p | just ru | PE.[ refl ] | NEWEPOCH-New (x , x) rewrite EPOCH-complete' _ x = refl

  ... | no p | ru | PE.[ refl ] | NEWEPOCH-Not-New x = refl
  ... | no p | nothing | PE.[ refl ] | NEWEPOCH-No-Reward-Update (x , x) = -elim $ p x
