module _ {lstate : LState} {ss : Snapshots} where
  SNAP-total : ∃[ ss' ] lstate ⊢ ss ⇀⦇ tt ,SNAP⦈ ss'
  SNAP-total = -, SNAP

  SNAP-complete : ∀ ss' → lstate ⊢ ss ⇀⦇ tt ,SNAP⦈ ss' → proj₁ SNAP-total ≡ ss'
  SNAP-complete ss' SNAP = refl

  SNAP-deterministic : ∀ {ss' ss''}
                     → lstate ⊢ ss ⇀⦇ tt ,SNAP⦈ ss'
                     → lstate ⊢ ss ⇀⦇ tt ,SNAP⦈ ss'' → ss' ≡ ss''
  SNAP-deterministic SNAP SNAP = refl

module _ {e : Epoch} (prs : PoolReapState) where
  POOLREAP-total : ∃[ prs' ] _ ⊢ prs ⇀⦇ e ,POOLREAP⦈ prs'
  POOLREAP-total = -, POOLREAP

  POOLREAP-complete
    : ∀ prs' → _ ⊢ prs ⇀⦇ e ,POOLREAP⦈ prs' → proj₁ POOLREAP-total ≡ prs'
  POOLREAP-complete prs' POOLREAP = refl

  POOLREAP-deterministic
    : ∀ {prs' prs''}
    → _ ⊢ prs ⇀⦇ e ,POOLREAP⦈ prs'
    → _ ⊢ prs ⇀⦇ e ,POOLREAP⦈ prs''
    → prs' ≡ prs''
  POOLREAP-deterministic POOLREAP POOLREAP = refl

module _ {eps : EpochState} {e : Epoch} where

  open EpochState eps hiding (es)

  prs = $\begin{pmatrix} \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#2153}{\htmlId{1999}{\htmlClass{Function}{\text{u0}}}}\, \,\htmlId{2002}{\htmlClass{Symbol}{\text{.}}}\,\,\href{Ledger.Conway.Specification.Epoch.html#19130}{\htmlId{2003}{\htmlClass{Field}{\text{utxoSt'}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.html#2170}{\htmlId{2013}{\htmlClass{Function}{\text{acnt}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#2128}{\htmlId{2020}{\htmlClass{Function}{\text{cs}}}}\, \,\htmlId{2023}{\htmlClass{Symbol}{\text{.}}}\,\,\href{Ledger.Conway.Specification.Certs.html#4861}{\htmlId{2024}{\htmlClass{Field}{\text{dState}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#2153}{\htmlId{2033}{\htmlClass{Function}{\text{u0}}}}\, \,\htmlId{2036}{\htmlClass{Symbol}{\text{.}}}\,\,\href{Ledger.Conway.Specification.Epoch.html#19074}{\htmlId{2037}{\htmlClass{Field}{\text{pState'}}}}\, \end{pmatrix}$
    where
      open LState
      open CertState
      open EPOCH-Updates0
      cs = ls .certState
      u0 = EPOCH-updates0 fut ls

  EPOCH-total : ∃[ eps' ] _ ⊢ eps ⇀⦇ e ,EPOCH⦈ eps'
  EPOCH-total =
    -, EPOCH
         ( SNAP-total .proj₂
         , RATIFIES-total' .proj₂
         , POOLREAP-total prs .proj₂)

  private
    EPOCH-state : Snapshots → RatifyState → PoolReapState → EpochState
    EPOCH-state ss fut' ($\begin{pmatrix} \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#2472}{\htmlId{2472}{\htmlClass{Bound}{\text{utxoSt''}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#2483}{\htmlId{2483}{\htmlClass{Bound}{\text{acnt'}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#2491}{\htmlId{2491}{\htmlClass{Bound}{\text{dState'}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#2501}{\htmlId{2501}{\htmlClass{Bound}{\text{pState''}}}}\, \end{pmatrix}$) =
      let
        EPOCHUpdates es govSt' dState'' _ gState' _ acnt'' =
          EPOCH-updates fut ls dState' acnt'
        certState' = $\begin{pmatrix} \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#2557}{\htmlId{2655}{\htmlClass{Bound}{\text{dState''}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#2501}{\htmlId{2666}{\htmlClass{Bound}{\text{pState''}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#2568}{\htmlId{2677}{\htmlClass{Bound}{\text{gState'}}}}\, \end{pmatrix}$
       in
          record
            { acnt = acnt''
            ; ss = ss
            ; ls = $\begin{pmatrix} \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#2472}{\htmlId{2787}{\htmlClass{Bound}{\text{utxoSt''}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#2550}{\htmlId{2798}{\htmlClass{Bound}{\text{govSt'}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#2640}{\htmlId{2807}{\htmlClass{Bound}{\text{certState'}}}}\, \end{pmatrix}$
            ; es = es
            ; fut = fut'
            }

  EPOCH-deterministic : ∀ eps' eps''
                      → _ ⊢ eps ⇀⦇ e ,EPOCH⦈ eps'
                      → _ ⊢ eps ⇀⦇ e ,EPOCH⦈ eps''
                      → eps' ≡ eps''
  EPOCH-deterministic
      eps'
      eps''
      (EPOCH
        {utxoSt'' = utxoSt''₁}
        {acnt' = acnt'₁}
        {dState' = dState'₁}
        {pState'' = pState'₁}
        (p₁ , p₂ , p₃)
      )
      (EPOCH
        {utxoSt'' = utxoSt''₂}
        {acnt' = acnt'₂}
        {dState' = dState'₂}
        {pState'' = pState'₂}
        (p₁' , p₂' , p₃')
      ) =
    cong₂ _$_ (cong₂ EPOCH-state ss'≡ss'' fut'≡fut'') prs'≡prs''
    where
      ss'≡ss'' : EpochState.ss eps' ≡ EpochState.ss eps''
      ss'≡ss'' = SNAP-deterministic p₁ p₁'

      fut'≡fut'' : EpochState.fut eps' ≡ EpochState.fut eps''
      fut'≡fut'' = RATIFIES-deterministic-≡
                    (cong (λ x → record
                                   { stakeDistrs = mkStakeDistrs (Snapshots.mark x) _ _ _ _ _
                                   ; currentEpoch = _
                                   ; dreps = _
                                   ; ccHotKeys = _
                                   ; treasury = _
                                   }) ss'≡ss'')
                                   refl refl p₂ p₂'

      prs'≡prs'' : $\begin{pmatrix} \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#3136}{\htmlId{4167}{\htmlClass{Bound}{\text{utxoSt''₁}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#3164}{\htmlId{4179}{\htmlClass{Bound}{\text{acnt'₁}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#3191}{\htmlId{4188}{\htmlClass{Bound}{\text{dState'₁}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#3221}{\htmlId{4199}{\htmlClass{Bound}{\text{pState'₁}}}}\, \end{pmatrix}$ ≡
                   $\begin{pmatrix} \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#3295}{\htmlId{4234}{\htmlClass{Bound}{\text{utxoSt''₂}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#3323}{\htmlId{4246}{\htmlClass{Bound}{\text{acnt'₂}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#3350}{\htmlId{4255}{\htmlClass{Bound}{\text{dState'₂}}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#3380}{\htmlId{4266}{\htmlClass{Bound}{\text{pState'₂}}}}\, \end{pmatrix}$
      prs'≡prs'' = POOLREAP-deterministic prs p₃ p₃'

  EPOCH-complete : ∀ eps' → _ ⊢ eps ⇀⦇ e ,EPOCH⦈ eps' → proj₁ EPOCH-total ≡ eps'
  EPOCH-complete eps' p = EPOCH-deterministic (proj₁ EPOCH-total) eps' (proj₂ EPOCH-total) p

  abstract
    EPOCH-total' : ∃[ eps' ] _ ⊢ eps ⇀⦇ e ,EPOCH⦈ eps'
    EPOCH-total' = EPOCH-total

    EPOCH-complete' : ∀ eps' → _ ⊢ eps ⇀⦇ e ,EPOCH⦈ eps' → proj₁ EPOCH-total' ≡ eps'
    EPOCH-complete' = EPOCH-complete

instance
  Computational-EPOCH : Computational _⊢_⇀⦇_,EPOCH⦈_ ⊥
  Computational-EPOCH .computeProof Γ s sig = success EPOCH-total'
  Computational-EPOCH .completeness Γ s sig s' h = cong success (EPOCH-complete' s' h)

module _ {e : Epoch} where

  NEWEPOCH-total : ∀ nes'' → ∃[ nes' ] _ ⊢ nes'' ⇀⦇ e ,NEWEPOCH⦈ nes'
  NEWEPOCH-total nes with e ≟ NewEpochState.lastEpoch nes + 1 | NewEpochState.ru nes | inspect NewEpochState.ru nes
  ... | yes p | just ru | PE.[ refl ] =  $\begin{pmatrix} \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#4956}{\htmlId{5203}{\htmlClass{Bound}{\text{e}}}}\, \\ \,\htmlId{5207}{\htmlClass{Symbol}{\text{\_}}}\, \\ \,\htmlId{5211}{\htmlClass{Symbol}{\text{\_}}}\, \\ \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#4521}{\htmlId{5215}{\htmlClass{Function}{\text{EPOCH-total'}}}}\, \,\htmlId{5228}{\htmlClass{Symbol}{\text{.}}}\,\,\href{Data.Product.Base.html#636}{\htmlId{5229}{\htmlClass{Field}{\text{proj₁}}}}\, \\ \,\href{Agda.Builtin.Maybe.html#194}{\htmlId{5237}{\htmlClass{InductiveConstructor}{\text{nothing}}}}\, \\ \,\htmlId{5247}{\htmlClass{Symbol}{\text{\_}}}\, \end{pmatrix}$
                                      , NEWEPOCH-New (p , EPOCH-total' .proj₂)
  ... | yes p | nothing | PE.[ refl ] = $\begin{pmatrix} \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#4956}{\htmlId{5372}{\htmlClass{Bound}{\text{e}}}}\, \\ \,\htmlId{5376}{\htmlClass{Symbol}{\text{\_}}}\, \\ \,\htmlId{5380}{\htmlClass{Symbol}{\text{\_}}}\, \\ \,\href{Data.Product.Base.html#636}{\htmlId{5384}{\htmlClass{Field}{\text{proj₁}}}}\, \,\href{Ledger.Conway.Specification.Epoch.Properties.Computational.html#4521}{\htmlId{5390}{\htmlClass{Function}{\text{EPOCH-total'}}}}\, \\ \,\href{Agda.Builtin.Maybe.html#194}{\htmlId{5405}{\htmlClass{InductiveConstructor}{\text{nothing}}}}\, \\ \,\htmlId{5415}{\htmlClass{Symbol}{\text{\_}}}\, \end{pmatrix}$
                                      , NEWEPOCH-No-Reward-Update (p , EPOCH-total' .proj₂)
  ... | no ¬p | _ | _ = -, NEWEPOCH-Not-New ¬p

  NEWEPOCH-complete : ∀ nes nes' → _ ⊢ nes ⇀⦇ e ,NEWEPOCH⦈ nes' → proj₁ (NEWEPOCH-total nes) ≡ nes'
  -- NEWEPOCH-complete nes nes' h with e ≟ NewEpochState.lastEpoch nes + 1 | NewEpochState.ru nes | h
  NEWEPOCH-complete nes nes' h with e ≟ NewEpochState.lastEpoch nes + 1 | NewEpochState.ru nes | inspect NewEpochState.ru nes | h
  ... | yes p | just ru | PE.[ refl ] | NEWEPOCH-New (x , x₁) rewrite EPOCH-complete' _ x₁ = refl
  ... | yes p | ru | PE.[ refl ] | NEWEPOCH-Not-New x = ⊥-elim $ x p
  ... | yes p | nothing | PE.[ refl ] | NEWEPOCH-No-Reward-Update (x , x₁) rewrite EPOCH-complete' _ x₁ = refl
  ... | no ¬p | ru | PE.[ refl ] | NEWEPOCH-New (x , x₁)  = ⊥-elim $ ¬p x
  ... | no ¬p | ru | PE.[ refl ] | NEWEPOCH-Not-New x = refl
  ... | no ¬p | nothing | PE.[ refl ] | NEWEPOCH-No-Reward-Update (x , x₁) = ⊥-elim $ ¬p x

instance
