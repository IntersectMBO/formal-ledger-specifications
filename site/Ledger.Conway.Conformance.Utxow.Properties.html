open import stdlib-meta.Tactic.GenError using (genErrors)

instance
  Computational-UTXOW : Computational ___,UTXOW_ String
  Computational-UTXOW = record {Go}
    where module Go  s tx (let H ,  H? = UTXOW-inductive-premises {tx}{s}) where

    open Computational Computational-UTXO
      renaming (computeProof to computeProof'; completeness to completeness')

    computeProof : ComputationResult String ( (  s  tx ,UTXOW_))
    computeProof =
      case H? of  where
        (yes (p , p , p , p , p , p , p , p)) 
          map (map (UTXOW-inductive p p p p p p p p)) (computeProof'  s tx)
        (no p)  failure $ genErrors p

    completeness :  s'    s  tx ,UTXOW s'
                         map proj computeProof  success s'
    completeness s' (UTXOW-inductive p p p p p p p p h) with H?
    ... | no p = -elim $ p ((p , p , p , p , p , p , p , p))
    ... | yes _ with computeProof'  s tx | completeness' _ _ _ _ h
    ... | success _ | refl = refl
