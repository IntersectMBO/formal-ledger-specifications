SVCrypto : CryptoStructure
SVCrypto = record
  { Implementation
  ; pkk = SVPKKScheme
  }
  where
  -- Dummy private key crypto scheme
  SVPKKScheme : PKKScheme
  SVPKKScheme = record
    { Implementation
    ; isSigned         =  a b m  a + b  m
    ; sign             = _+_
    ; isSigned-correct =  where (sk , sk , refl) _ _ h  h
    }

    instance _ = SVCrypto

    SVScriptStructure : ScriptStructure
SVScriptStructure = record
  { p1s = P1ScriptStructure-HTL
  ; hashRespectsUnion = hashRespectsUnion
  ; ps = SVP2ScriptStructure
  }
  where

    instance Hashable-PlutusScript : Hashable Implementation.PlutusScript 
             Hashable-PlutusScript = record { hash =  x  proj x }

    hashRespectsUnion :  {A B }
       Hashable A   Hashable B 
       Hashable (A  B) 
    hashRespectsUnion ha hb = record { hash =  { (inj x)  Hashable.hash ha x ; (inj y)  Hashable.hash hb y }}

    SVP2ScriptStructure : PlutusStructure
    SVP2ScriptStructure =  record
      { Implementation
      ; validPlutusScript =  _ _ _ _  
      }

    instance _ = SVScriptStructure
