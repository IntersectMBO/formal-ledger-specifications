instance
  Computational-UTXOS : Computational ___,UTXOS_ String
  Computational-UTXOS = record {go} where
    module go ( : UTxOEnv) (s : UTxOState) (tx : Tx)
      (let open UTxOState s)
      (let H-Yes ,  H-Yes? = Scripts-Yes-premises {} {tx} {utxo} {deposits})
      (let H-No  ,  H-No?  = Scripts-No-premises  {} {tx} {utxo}) where
      open Tx tx renaming (body to txb); open TxBody txb
      open UTxOEnv  renaming (pparams to pp)
      sLst = collectP2ScriptsWithContext pp tx utxo

      computeProof =
        case H-Yes? , H-No? of  where
          (yes p , no _ )  success (_ , (Scripts-Yes p))
          (no _  , yes p)  success (_ , (Scripts-No p))
          (_     , _    )  failure "isValid check failed"

      completeness :  s'    s  tx ,UTXOS s'  map proj computeProof  success s'
      completeness _ (Scripts-Yes p) with H-No? | H-Yes?
      ... | yes (_ , refl) | _     = case proj p of  ()
      ... | no _           | yes _ = refl
      ... | no _           | no p = case p p of  ()
      completeness _ (Scripts-No p) with H-Yes? | H-No?
      ... | yes (_ , _ , refl) | _     = case proj p of  ()
      ... | no _               | yes _ = refl
      ... | no _               | no p = case p p of  ()

instance
  Computational-UTXO' : Computational ___,UTXO_ String
  Computational-UTXO' = record {Go}
    where
      module Go  s tx (let H ,  H? = UTXO-premises { = } {s = s} {tx = tx}) where

        open Computational Computational-UTXOS
          renaming (computeProof to computeProof'; completeness to completeness')

        genErr :  H  String
        genErr  p = case dec-de-morgan p of  where
          (inj a)  " TxBody.txIns (Tx.body tx)  "
          (inj b)  case dec-de-morgan b of  where
            (inj a)  " TxBody.txIns (Tx.body tx)  dom (UTxOOf s)"
            (inj b)  case dec-de-morgan b of  where
                (inj a')  " refInputs  dom utxo "
                (inj b')  case dec-de-morgan b' of  where
                  (inj a)  " inInterval (UTxOEnv.slot ) (txvldt (Tx.body tx))"
                  (inj b)  case dec-de-morgan b of  where
                    (inj a)  " feesOK pp tx utxo"
                    (inj b)  case dec-de-morgan b of  where
                        (inj a) 
                          let
                            pp = PParamsOf 
                            txb = TxBodyOf tx
                            con = consumed pp s txb
                            prod = produced pp s txb
                            showValue = show  coin
                          in
                            ( "consumed (UTxOEnv.pparams ) s (Tx.body tx)  produced (PParamsOf ) s (Tx.body tx)"
                            + "\n  consumed =\t\t" + showValue con
                            + "\n    ins  =\t\t" + showValue (balance (UTxOOf s  txb .TxBody.txIns))
                            + "\n    mint =\t\t" + showValue (TxBody.mint txb)
                            + "\n    depositRefunds =\t" + showValue (inject (depositRefunds pp s txb))
                            + "\n  produced =\t\t" + showValue prod
                            + "\n    outs =\t\t" + showValue (balance $ outs txb)
                            + "\n    fee  =\t\t" + show (FeesOf tx)
                            + "\n    newDeposits  =\t" + show (newDeposits pp s txb)
                            + "\n    donation  =\t\t" + show (DonationsOf txb)
                            )
                        (inj b)  case dec-de-morgan b of  where
                          (inj a)  " coin (TxBody.mint (Tx.body tx))  0"
                          (inj b)  case dec-de-morgan b of  where
                              (inj a)  "((Tx.txsize tx) Data.Nat.Base. maxTxSize (UTxOEnv.pparams ))"
                              (inj b)  case dec-de-morgan b of  where
                                (inj a)  "[ (_ , txout)  txOuts .proj ] inject (utxoEntrySize txout * coinsPerUTxOByte pp)  getValue txout"
                                (inj b)  case dec-de-morgan b of  where
                                    (inj a)  "[ (_ , txout)  txOuts .proj ] serSize (getValue txout)  maxValSize pp"
                                    (inj b)  case dec-de-morgan b of  where
                                      (inj a)  "[ (a , _)  range txOuts ] Sum.All (const ) ( a  a .BootstrapAddr.attrsSize  64) a"
                                      (inj _)  "something else broke"

        computeProofH : Dec H  ComputationResult String ([ s' ]   s  tx ,UTXO s')
        computeProofH (yes (x , y , z , e , k , l , m , c , v , j , n , o , p , q , r , t , u)) =
            map (UTXO-inductive _ _ _ x y z e k l m c v j n o p q r t u) <$> computeProof'  s tx
        computeProofH (no p) = failure $ genErr p

        computeProof : ComputationResult String ([ s' ]   s  tx ,UTXO s')
        computeProof = computeProofH H?

        completeness :  s'    s  tx ,UTXO s'  map proj computeProof  success s'
        completeness s' (UTXO-inductive _ _ _ x y z w k l m c v j n o p q r t u h) with H?
        ... | no p = -elim $ p (x , y , z , w , k , l , m , c , v , j , n , o , p , q , r , t , u)
        ... | yes _ with computeProof'  s tx | completeness' _ _ _ _ h
        ... | success _ | refl = refl

  Computational-UTXO : Computational ___,UTXO_ String
  Computational-UTXO = Computational-UTXO'

  utxoState utxoState'  : UTxOState

UTXO-step = compute  Computational-UTXO

                             utxoState  tx ,UTXO utxoState'
