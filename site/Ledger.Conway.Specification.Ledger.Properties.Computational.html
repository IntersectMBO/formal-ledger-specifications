    Ledger: Computational¶

    This module proves that the LEDGER transition system is computational.

    {-# OPTIONS --safe #-}

open import Ledger.Core.Specification.Abstract
open import Ledger.Core.Specification.Transaction

module Ledger.Conway.Specification.Ledger.Properties.Computational
  (txs : TransactionStructure) (open TransactionStructure txs)
  (abs : AbstractFunctions txs)
  where

open import Ledger.Prelude

open import Ledger.Conway.Specification.Certs govStructure
open import Ledger.Conway.Specification.Certs.Properties.Computational govStructure
open import Ledger.Conway.Specification.Gov txs
open import Ledger.Conway.Specification.Gov.Properties.Computational txs
open import Ledger.Conway.Specification.Ledger txs abs
open import Ledger.Conway.Specification.Transaction txs abs
open import Ledger.Conway.Specification.Utxo txs abs
open import Ledger.Conway.Specification.Utxow txs abs
open import Ledger.Conway.Specification.Utxow.Properties.Computational txs abs

open import Data.Bool.Properties using (-not)

    instance
  _ = Monad-ComputationResult

  Computational-LEDGER : Computational ___,LEDGER_ String
  Computational-LEDGER = record {go}
    where
    open Computational ... renaming (computeProof to comp; completeness to complete)
    computeUtxow = comp {STS = ___,UTXOW_}
    computeCerts = comp {STS = ___,CERTS_}
    computeGov   = comp {STS = ___,GOVS_}

    module go
      ( : LEnv)   (let open LEnv )
      (s : LState) (let open LState s)
      (tx : Tx)    (let open Tx tx renaming (body to txb); open TxBody txb)
      where
      utxo = UTxOEnv  record { LEnv  }
      cert = CertEnv  $\begin{pmatrix} \,\href{Ledger.Core.Specification.Epoch.html#954}{\htmlId{1848}{\htmlClass{Function}{\text{epoch}}}}\, \,\href{Ledger.Conway.Specification.Ledger.html#1073}{\htmlId{1854}{\htmlClass{Field}{\text{slot}}}}\, \\ \,\href{Ledger.Conway.Specification.Ledger.html#1131}{\htmlId{1861}{\htmlClass{Field}{\text{pparams}}}}\, \\ \,\href{Ledger.Conway.Specification.Transaction.html#2394}{\htmlId{1871}{\htmlClass{Function}{\text{txGovVotes}}}}\, \\ \,\href{Ledger.Conway.Specification.Transaction.html#2227}{\htmlId{1884}{\htmlClass{Function}{\text{txWithdrawals}}}}\, \\ \,\htmlId{1900}{\htmlClass{Symbol}{\text{\_}}}\, \end{pmatrix}$
      gov : CertState  GovEnv
      gov =  certState  $\begin{pmatrix} \,\href{Ledger.Conway.Specification.Transaction.html#2125}{\htmlId{1965}{\htmlClass{Function}{\text{txId}}}}\, \\ \,\href{Ledger.Core.Specification.Epoch.html#954}{\htmlId{1972}{\htmlClass{Function}{\text{epoch}}}}\, \,\href{Ledger.Conway.Specification.Ledger.html#1073}{\htmlId{1978}{\htmlClass{Field}{\text{slot}}}}\, \\ \,\href{Ledger.Conway.Specification.Ledger.html#1131}{\htmlId{1985}{\htmlClass{Field}{\text{pparams}}}}\, \\ \,\href{Ledger.Conway.Specification.Ledger.html#1096}{\htmlId{1995}{\htmlClass{Field}{\text{ppolicy}}}}\, \\ \,\href{Ledger.Conway.Specification.Ledger.html#1157}{\htmlId{2005}{\htmlClass{Field}{\text{enactState}}}}\, \\ \,\href{Ledger.Conway.Specification.Ledger.Properties.Computational.html#1951}{\htmlId{2018}{\htmlClass{Bound}{\text{certState}}}}\, \\ \,\htmlId{2030}{\htmlClass{Symbol}{\text{\_}}}\, \end{pmatrix}$

      computeProof : ComputationResult String ([ s' ]   s  tx ,LEDGER s')
      computeProof = case isValid  true of  where
        (yes p)  do
          (utxoSt' , utxoStep)  computeUtxow utxo utxoSt tx
          (certSt' , certStep)  computeCerts cert certState txCerts
          (govSt'  , govStep)   computeGov (gov certSt') (rmOrphanDRepVotes certSt' govSt) (txgov txb)
          success (_ , LEDGER-V p utxoStep certStep govStep)
        (no p)  do
          (utxoSt' , utxoStep)  computeUtxow utxo utxoSt tx
          success (_ , LEDGER-I (-not p) utxoStep)

      completeness :  s'    s  tx ,LEDGER s'  (proj <$> computeProof)  success s'
      completeness ledgerSt (LEDGER-V v utxoStep certStep govStep)
        with isValid  true
      ... | no v = contradiction v v
      ... | yes refl
        with computeUtxow utxo utxoSt tx | complete _ _ _ _ utxoStep
      ... | success (utxoSt' , _) | refl
        with computeCerts cert certState txCerts | complete _ _ _ _ certStep
      ... | success (certSt' , _) | refl
        with computeGov (gov certSt') (rmOrphanDRepVotes certSt' govSt ) (txgov txb)
             | complete {STS = ___,GOVS_} (gov certSt') _ _ _ govStep
      ... | success (govSt' , _) | refl = refl
      completeness ledgerSt (LEDGER-I i utxoStep)
        with isValid  true
      ... | yes refl = case i of  ()
      ... | no v
        with computeUtxow utxo utxoSt tx | complete _ _ _ _ utxoStep
      ... | success (utxoSt' , _) | refl = refl
