  Computational-DELEG .computeProof de st =
    let open DelegEnv de; open DState st in
     where
    (delegate c mv mc d)  case  (c  dom rewards  d  pparams .PParams.keyDeposit)
                                 (c  dom rewards  d  0)
                                 mv  map (just  credVoter DRep) delegatees 
                                    fromList ( nothing  just abstainRep  just noConfidenceRep  [] )
                                 mc  map just (dom pools)   nothing 
                                 of  where
      (yes p)  success (-, DELEG-delegate p)
      (no p)  failure (genErrors p)
    (dereg c d)  case  (c , 0)  rewards  of  where
      (yes p)  success (-, DELEG-dereg p)
      (no p)  failure (genErrors p)
    (reg c d)  case  c  dom rewards
                      (d  pparams .PParams.keyDeposit  d  0)
                      of  where
      (yes p)  success (-, DELEG-reg p)
      (no p)  failure (genErrors p)

  Computational-DELEG .completeness de st (delegate c mv mc d)
    s' (DELEG-delegate p) rewrite dec-yes ( (c  dom (DState.rewards st)  d  DelegEnv.pparams de .PParams.keyDeposit)

  Computational-DELEG .completeness de st (dereg c d) _ (DELEG-dereg p)

  Computational-CERTBASE .computeProof ce cs _ =
    let open CertEnv ce; open PParams pp

        refresh = mapPartial getDRepVote (fromList votes)

      let
        open DState (dState cs )
        open DState (dState cs') renaming (rewards to rewards')
        module        = IsEquivalence (-isEquivalence {Credential  Coin})
        wdrlsCC         = map (map RwdAddr.stake) (wdrls )
        zeroMap         = constMap (map RwdAddr.stake (dom wdrls)) 0
        rwds--decomp  = (rewards  dom wdrlsCC )  (rewards  dom wdrlsCC)
      in
        begin
          getCoin rewards
             -getCoin rwds--decomp rewards
                ( .trans (disjoint-- (disjoint-sym res-ex-disjoint))
                           (.trans -sym (res-ex- (_? dom wdrlsCC))) ) 
          getCoin rwds--decomp
             indexedSum'- (rewards  dom wdrlsCC ) (rewards  dom wdrlsCC)
                              (disjoint-sym res-ex-disjoint) 
          getCoin (rewards  dom wdrlsCC ) + getCoin (rewards  dom wdrlsCC )
             cong (getCoin (rewards  dom wdrlsCC ) +_)
               ( getCoin-cong (rewards  dom wdrlsCC) wdrlsCC (res-subset{m = rewards} wdrlsCCrwds) ) 
          getCoin (rewards  dom wdrlsCC ) + getCoin wdrlsCC
             cong (getCoin (rewards  dom wdrlsCC ) +_) (-getCoin (wdrls ) (injOn wdrls constNetworkId))

             cong (_+ getCoin wdrls)
                ( begin
                  getCoin (zeroMap  rewards)
                     -getCoin (zeroMap  rewards) (zeroMap  (rewards  dom zeroMap ))

                  getCoin (zeroMap  (rewards  dom zeroMap ))
                     indexedSum'- zeroMap (rewards  dom zeroMap )
                                      (disjoint-sym res-comp-dom) 
                  getCoin zeroMap + getCoin (rewards  dom zeroMap )
                     cong ( u  u + getCoin (rewards  dom zeroMap )) sumConstZero

    sts-pov  : {s s : CertState}  ReflexiveTransitiveClosure {sts = ___,CERT_}  s l s
              getCoin s  getCoin s
    sts-pov (BS-base Id-nop) = refl
    sts-pov (BS-ind x xs) = trans (CERT-pov x) (sts-pov xs)

    CERTS-pov : {s s : CertState}    s  l ,CERTS s  getCoin s  getCoin s + getCoin (CertEnv.wdrls )
    CERTS-pov (RTC {s' = s'} {s'' = s} (bsts , BS-base Id-nop)) = CERTBASE-pov bsts
    CERTS-pov (RTC (bsts , BS-ind x sts)) = trans  (CERTBASE-pov bsts)
                                                   (cong  (_+ getCoin (CertEnv.wdrls ))
                                                          (trans (CERT-pov x) (sts-pov sts)))
