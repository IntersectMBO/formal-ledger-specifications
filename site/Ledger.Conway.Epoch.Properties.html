module _ (lstate : LState) (ss : Snapshots) where
  SNAP-total : ∃[ ss' ] lstate ⊢ ss ⇀⦇ tt ,SNAP⦈ ss'
  SNAP-total = -, SNAP

  SNAP-complete : ∀ ss' → lstate ⊢ ss ⇀⦇ tt ,SNAP⦈ ss' → proj₁ SNAP-total ≡ ss'
  SNAP-complete ss' SNAP = refl

module _ {eps : EpochState} {e : Epoch} where

  open EpochState eps hiding (es)
  open LState ls; open CertState certState
  open RatifyState fut using (removed)

  -- repated definitions from EPOCH to satisfy the type checker
  open UTxOState
  open PState; open DState; open GState
  open Acnt; open EnactState; open GovActionState

  esW      = RatifyState.es fut
  es       = record esW { withdrawals = ∅ }
  tmpGovSt = filter (λ x → ¿ ¬ proj₁ x ∈ mapˢ proj₁ removed ¿) govSt
  orphans  = fromList (getOrphans es tmpGovSt)
  removed' = removed ∪ orphans
  removedGovActions = flip concatMapˢ removed' λ (gaid , gaSt) →
    mapˢ (returnAddr gaSt ,_) ((utxoSt .deposits ∣ ❴ GovActionDeposit gaid ❵) ˢ)
  govActionReturns = aggregate₊ (mapˢ (λ (a , _ , d) → a , d) removedGovActions ᶠˢ)

  trWithdrawals  = esW .withdrawals
  totWithdrawals = ∑[ x ← trWithdrawals ] x

  retired    = (pState .retiring) ⁻¹ e
  payout     = govActionReturns ∪⁺ trWithdrawals
  refunds    = pullbackMap payout toRwdAddr (dom (dState .rewards))
  unclaimed  = getCoin payout ∸ getCoin refunds

  govSt' = filter (λ x → ¿ ¬ proj₁ x ∈ mapˢ proj₁ removed' ¿) govSt

  dState' = $\begin{pmatrix} \,\href{Ledger.Conway.Certs.html#5368}{\htmlId{2172}{\htmlClass{Function}{\text{dState}}}}\, \,\htmlId{2179}{\htmlClass{Symbol}{\text{.}}}\,\,\href{Ledger.Conway.Certs.html#3868}{\htmlId{2180}{\htmlClass{Field}{\text{voteDelegs}}}}\, \\ \,\href{Ledger.Conway.Certs.html#5368}{\htmlId{2193}{\htmlClass{Function}{\text{dState}}}}\, \,\htmlId{2200}{\htmlClass{Symbol}{\text{.}}}\,\,\href{Ledger.Conway.Certs.html#3907}{\htmlId{2201}{\htmlClass{Field}{\text{stakeDelegs}}}}\, \\  \,\href{Ledger.Conway.Certs.html#5368}{\htmlId{2216}{\htmlClass{Function}{\text{dState}}}}\, \,\htmlId{2223}{\htmlClass{Symbol}{\text{.}}}\,\,\href{Ledger.Conway.Certs.html#3947}{\htmlId{2224}{\htmlClass{Field}{\text{rewards}}}}\, \,\href{Axiom.Set.Map.Dec.html#1854}{\htmlId{2232}{\htmlClass{Function Operator}{\text{∪⁺}}}}\, \,\href{Ledger.Conway.Epoch.Properties.html#1974}{\htmlId{2235}{\htmlClass{Function}{\text{refunds}}}}\, \end{pmatrix}$

  pState' = $\begin{pmatrix} \,\href{Ledger.Conway.Certs.html#5388}{\htmlId{2260}{\htmlClass{Function}{\text{pState}}}}\, \,\htmlId{2267}{\htmlClass{Symbol}{\text{.}}}\,\,\href{Ledger.Conway.Certs.html#4519}{\htmlId{2268}{\htmlClass{Field}{\text{pools}}}}\, \,\href{Axiom.Set.Map.html#10748}{\htmlId{2274}{\htmlClass{Function Operator}{\text{∣}}}}\, \,\href{Ledger.Conway.Epoch.Properties.html#1886}{\htmlId{2276}{\htmlClass{Function}{\text{retired}}}}\, \,\href{Axiom.Set.Map.html#10748}{\htmlId{2284}{\htmlClass{Function Operator}{\text{ᶜ}}}}\, \\ \,\href{Ledger.Conway.Certs.html#5388}{\htmlId{2288}{\htmlClass{Function}{\text{pState}}}}\, \,\htmlId{2295}{\htmlClass{Symbol}{\text{.}}}\,\,\href{Ledger.Conway.Certs.html#4556}{\htmlId{2296}{\htmlClass{Field}{\text{retiring}}}}\, \,\href{Axiom.Set.Map.html#10748}{\htmlId{2305}{\htmlClass{Function Operator}{\text{∣}}}}\, \,\href{Ledger.Conway.Epoch.Properties.html#1886}{\htmlId{2307}{\htmlClass{Function}{\text{retired}}}}\, \,\href{Axiom.Set.Map.html#10748}{\htmlId{2315}{\htmlClass{Function Operator}{\text{ᶜ}}}}\, \end{pmatrix}$

  gState' = $\begin{pmatrix} (\,\href{Class.ToBool.html#389}{\htmlId{2335}{\htmlClass{Function Operator}{\text{if}}}}\, \,\href{Data.List.Base.html#4703}{\htmlId{2338}{\htmlClass{Function}{\text{null}}}}\, \,\href{Ledger.Conway.Epoch.Properties.html#2091}{\htmlId{2343}{\htmlClass{Function}{\text{govSt'}}}}\, \,\href{Class.ToBool.html#389}{\htmlId{2350}{\htmlClass{Function Operator}{\text{then}}}}\, \,\href{Axiom.Set.Map.html#6068}{\htmlId{2355}{\htmlClass{Function}{\text{mapValues}}}}\, (\,\htmlId{2366}{\htmlClass{Number}{\text{1}}}\, \,\href{Class.HasAdd.Core.html#162}{\htmlId{2368}{\htmlClass{Field Operator}{\text{+\_}}}}\,) (\,\href{Ledger.Conway.Certs.html#5408}{\htmlId{2373}{\htmlClass{Function}{\text{gState}}}}\, \,\htmlId{2380}{\htmlClass{Symbol}{\text{.}}}\,\,\href{Ledger.Conway.Certs.html#4896}{\htmlId{2381}{\htmlClass{Field}{\text{dreps}}}}\,) \,\href{Class.ToBool.html#389}{\htmlId{2388}{\htmlClass{Function Operator}{\text{else}}}}\, (\,\href{Ledger.Conway.Certs.html#5408}{\htmlId{2394}{\htmlClass{Function}{\text{gState}}}}\, \,\htmlId{2401}{\htmlClass{Symbol}{\text{.}}}\,\,\href{Ledger.Conway.Certs.html#4896}{\htmlId{2402}{\htmlClass{Field}{\text{dreps}}}}\,)
            , \,\href{Ledger.Conway.Certs.html#5408}{\htmlId{2424}{\htmlClass{Function}{\text{gState}}}}\, \,\htmlId{2431}{\htmlClass{Symbol}{\text{.}}}\,\,\href{Ledger.Conway.Certs.html#4919}{\htmlId{2432}{\htmlClass{Field}{\text{ccHotKeys}}}}\, \,\href{Axiom.Set.Map.html#10678}{\htmlId{2442}{\htmlClass{Function Operator}{\text{∣}}}}\, \,\href{Ledger.Conway.Enact.html#2512}{\htmlId{2444}{\htmlClass{Function}{\text{ccCreds}}}}\, (\,\href{Ledger.Conway.Epoch.Properties.html#1383}{\htmlId{2453}{\htmlClass{Function}{\text{es}}}}\, \,\htmlId{2456}{\htmlClass{Symbol}{\text{.}}}\,\,\href{Ledger.Conway.Enact.html#1683}{\htmlId{2457}{\htmlClass{Field}{\text{cc}}}}\,) \end{pmatrix}$

  certState' : CertState
  certState' = $\begin{pmatrix} \,\href{Ledger.Conway.Epoch.Properties.html#2160}{\htmlId{2506}{\htmlClass{Function}{\text{dState'}}}}\, \\ \,\href{Ledger.Conway.Epoch.Properties.html#2248}{\htmlId{2516}{\htmlClass{Function}{\text{pState'}}}}\, \\ \,\href{Ledger.Conway.Epoch.Properties.html#2322}{\htmlId{2526}{\htmlClass{Function}{\text{gState'}}}}\, \end{pmatrix}$

  utxoSt' = $\begin{pmatrix} \,\href{Ledger.Conway.Ledger.html#1548}{\htmlId{2551}{\htmlClass{Function}{\text{utxoSt}}}}\, \,\htmlId{2558}{\htmlClass{Symbol}{\text{.}}}\,\,\href{Ledger.Conway.Utxo.html#6104}{\htmlId{2559}{\htmlClass{Field}{\text{utxo}}}}\, \\ \,\href{Ledger.Conway.Ledger.html#1548}{\htmlId{2566}{\htmlClass{Function}{\text{utxoSt}}}}\, \,\htmlId{2573}{\htmlClass{Symbol}{\text{.}}}\,\,\href{Ledger.Conway.Utxo.html#6126}{\htmlId{2574}{\htmlClass{Field}{\text{fees}}}}\, \\ \,\href{Ledger.Conway.Ledger.html#1548}{\htmlId{2581}{\htmlClass{Function}{\text{utxoSt}}}}\, \,\htmlId{2588}{\htmlClass{Symbol}{\text{.}}}\,\,\href{Ledger.Conway.Utxo.html#6148}{\htmlId{2589}{\htmlClass{Field}{\text{deposits}}}}\, \,\href{Axiom.Set.Map.html#10748}{\htmlId{2598}{\htmlClass{Function Operator}{\text{∣}}}}\, \,\href{abstract-set-theory.FiniteSetTheory.html#519}{\htmlId{2600}{\htmlClass{Function}{\text{mapˢ}}}}\, (\,\href{Data.Product.Base.html#636}{\htmlId{2606}{\htmlClass{Field}{\text{proj₁}}}}\, \,\href{Function.Base.html#1115}{\htmlId{2612}{\htmlClass{Function Operator}{\text{∘}}}}\, \,\href{Data.Product.Base.html#650}{\htmlId{2614}{\htmlClass{Field}{\text{proj₂}}}}\,) \,\href{Ledger.Conway.Epoch.Properties.html#1574}{\htmlId{2621}{\htmlClass{Function}{\text{removedGovActions}}}}\, \,\href{Axiom.Set.Map.html#10748}{\htmlId{2639}{\htmlClass{Function Operator}{\text{ᶜ}}}}\, \\ \,\htmlId{2643}{\htmlClass{Number}{\text{0}}}\, \end{pmatrix}$

  acnt' = record acnt
    { treasury  = acnt .treasury ∸ totWithdrawals + utxoSt .donations + unclaimed }

  ls' = $\begin{pmatrix} \,\href{Ledger.Conway.Epoch.Properties.html#2539}{\htmlId{2765}{\htmlClass{Function}{\text{utxoSt'}}}}\, \\ \,\href{Ledger.Conway.Epoch.Properties.html#2091}{\htmlId{2775}{\htmlClass{Function}{\text{govSt'}}}}\, \\ \,\href{Ledger.Conway.Epoch.Properties.html#2466}{\htmlId{2784}{\htmlClass{Function}{\text{certState'}}}}\, \end{pmatrix}$

  EPOCH-total : ∃[ eps' ] _ ⊢ eps ⇀⦇ e ,EPOCH⦈ eps'
  EPOCH-total = -, EPOCH (RATIFIES-total' .proj₂) (SNAP-total ls ss .proj₂)

  EPOCH-complete : ∀ eps' → _ ⊢ eps ⇀⦇ e ,EPOCH⦈ eps' → proj₁ EPOCH-total ≡ eps'
  EPOCH-complete eps' (EPOCH p₁ p₂) = cong₂ (λ ss fut → record { acnt = acnt' ; ss = ss ; ls = ls' ; es = _ ; fut = fut }) (SNAP-complete _ _ _ p₂)
    (RATIFIES-complete' (subst ty (cong Snapshots.mark (sym (SNAP-complete _ _ _ p₂))) p₁))
    where
      ty : Snapshot → Set
      ty x = record
        { stakeDistrs = mkStakeDistrs x govSt' _ _
        ; currentEpoch = _
        ; dreps = _
        ; ccHotKeys = _
        ; treasury = _
        } ⊢ _ ⇀⦇ govSt' ,RATIFIES⦈ _

  abstract
    EPOCH-total' : ∃[ eps' ] _ ⊢ eps ⇀⦇ e ,EPOCH⦈ eps'
    EPOCH-total' = EPOCH-total

    EPOCH-complete' : ∀ eps' → _ ⊢ eps ⇀⦇ e ,EPOCH⦈ eps' → proj₁ EPOCH-total' ≡ eps'
    EPOCH-complete' = EPOCH-complete

instance
  Computational-EPOCH : Computational _⊢_⇀⦇_,EPOCH⦈_ ⊥
  Computational-EPOCH .computeProof Γ s sig = success EPOCH-total'
  Computational-EPOCH .completeness Γ s sig s' h = cong success (EPOCH-complete' s' h)

module _ {e : Epoch} where

  NEWEPOCH-total : ∀ nes'' → ∃[ nes' ] _ ⊢ nes'' ⇀⦇ e ,NEWEPOCH⦈ nes'
  NEWEPOCH-total nes with e ≟ NewEpochState.lastEpoch nes + 1 | NewEpochState.ru nes | inspect NewEpochState.ru nes
  ... | yes p | just ru | PE.[ refl ] =  $\begin{pmatrix} \,\href{Ledger.Conway.Epoch.Properties.html#3937}{\htmlId{4184}{\htmlClass{Bound}{\text{e}}}}\, \\ \,\href{Ledger.Conway.Epoch.Properties.html#3502}{\htmlId{4188}{\htmlClass{Function}{\text{EPOCH-total'}}}}\, \,\htmlId{4201}{\htmlClass{Symbol}{\text{.}}}\,\,\href{Data.Product.Base.html#636}{\htmlId{4202}{\htmlClass{Field}{\text{proj₁}}}}\, \\ \,\href{Agda.Builtin.Maybe.html#194}{\htmlId{4210}{\htmlClass{InductiveConstructor}{\text{nothing}}}}\, \end{pmatrix}$
                                      , NEWEPOCH-New (p , EPOCH-total' .proj₂)
  ... | yes p | nothing | PE.[ refl ] = $\begin{pmatrix} \,\href{Ledger.Conway.Epoch.Properties.html#3937}{\htmlId{4341}{\htmlClass{Bound}{\text{e}}}}\, \\ \,\href{Data.Product.Base.html#636}{\htmlId{4345}{\htmlClass{Field}{\text{proj₁}}}}\, \,\href{Ledger.Conway.Epoch.Properties.html#3502}{\htmlId{4351}{\htmlClass{Function}{\text{EPOCH-total'}}}}\, \\ \,\href{Agda.Builtin.Maybe.html#194}{\htmlId{4366}{\htmlClass{InductiveConstructor}{\text{nothing}}}}\, \end{pmatrix}$
                                      , NEWEPOCH-No-Reward-Update (p , EPOCH-total' .proj₂)
  ... | no ¬p | _ | _ = -, NEWEPOCH-Not-New ¬p

  NEWEPOCH-complete : ∀ nes nes' → _ ⊢ nes ⇀⦇ e ,NEWEPOCH⦈ nes' → proj₁ (NEWEPOCH-total nes) ≡ nes'
  -- NEWEPOCH-complete nes nes' h with e ≟ NewEpochState.lastEpoch nes + 1 | NewEpochState.ru nes | h
  NEWEPOCH-complete nes nes' h with e ≟ NewEpochState.lastEpoch nes + 1 | NewEpochState.ru nes | inspect NewEpochState.ru nes | h 
  ... | yes p | just ru | PE.[ refl ] | NEWEPOCH-New (x , x₁) rewrite EPOCH-complete' _ x₁ = refl
  ... | yes p | ru | PE.[ refl ] | NEWEPOCH-Not-New x = ⊥-elim $ x p
  ... | yes p | nothing | PE.[ refl ] | NEWEPOCH-No-Reward-Update (x , x₁) rewrite EPOCH-complete' _ x₁ = refl
  ... | no ¬p | ru | PE.[ refl ] | NEWEPOCH-New (x , x₁)  = ⊥-elim $ ¬p x
  ... | no ¬p | ru | PE.[ refl ] | NEWEPOCH-Not-New x = refl
  ... | no ¬p | nothing | PE.[ refl ] | NEWEPOCH-No-Reward-Update (x , x₁) = ⊥-elim $ ¬p x

instance
  Computational-NEWEPOCH : Computational _⊢_⇀⦇_,NEWEPOCH⦈_ ⊥
  Computational-NEWEPOCH .computeProof Γ s sig = success (NEWEPOCH-total _)
  Computational-NEWEPOCH .completeness Γ s sig s' h = cong success (NEWEPOCH-complete _ s' h)
