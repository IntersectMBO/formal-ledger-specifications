open import Ledger.Core.Specification.Gov.Base

module Ledger.Conway.Specification.Enact.Properties.Computational (gs : _) (open GovStructure gs) where

open import Ledger.Prelude

open import Ledger.Core.Specification.Gov.Actions gs hiding (yes; no)
open import Ledger.Conway.Specification.Enact gs

open EnactState

open Computational ...

module _ { : EnactEnv} {eSt : EnactState} {ga : GovAction} where

  -- TODO: Make this follow from computationalness or viceversa
  ENACT-deterministic :  {eSt' eSt''}
                         eSt  ga ,ENACT eSt'
                         eSt  ga ,ENACT eSt''
                       eSt'  eSt''
  ENACT-deterministic Enact-NoConf Enact-NoConf = refl
  ENACT-deterministic (Enact-UpdComm _) (Enact-UpdComm _) = refl
  ENACT-deterministic Enact-NewConst Enact-NewConst = refl
  ENACT-deterministic Enact-HF Enact-HF = refl
  ENACT-deterministic Enact-PParams Enact-PParams = refl
  ENACT-deterministic (Enact-Wdrl _) (Enact-Wdrl _) = refl
  ENACT-deterministic Enact-Info Enact-Info = refl

instance
  Computational-ENACT : Computational ___,ENACT_ String
  Computational-ENACT .computeProof  s =
    let open EnactEnv  renaming (treasury to t; epoch to e) in
     where
    $\begin{pmatrix} \,\href{Ledger.Core.Specification.Gov.Actions.html#2440}{\htmlId{1392}{\htmlClass{InductiveConstructor}{\text{NoConfidence}}}}\,        \\ \,\htmlId{1414}{\htmlClass{Symbol}{\text{\_}}}\,               \end{pmatrix}$  success (_ , Enact-NoConf)
    $\begin{pmatrix} \,\href{Ledger.Core.Specification.Gov.Actions.html#2478}{\htmlId{1469}{\htmlClass{InductiveConstructor}{\text{UpdateCommittee}}}}\,     \\ \,\htmlId{1491}{\htmlClass{Symbol}{\text{(}}}\,\,\href{Ledger.Conway.Specification.Enact.Properties.Computational.html#1492}{\htmlId{1492}{\htmlClass{Bound}{\text{new}}}}\, , \,\href{Ledger.Conway.Specification.Enact.Properties.Computational.html#1498}{\htmlId{1498}{\htmlClass{Bound}{\text{rem}}}}\, , \,\href{Ledger.Conway.Specification.Enact.Properties.Computational.html#1504}{\htmlId{1504}{\htmlClass{Bound}{\text{q}}}}\,\,\htmlId{1505}{\htmlClass{Symbol}{\text{)}}}\, \end{pmatrix}$ 
      case  [ term  range new ]
               term  CCMaxTermLengthOf s +' e  of  where
      (yes p)  success (-, Enact-UpdComm
        (subst ( x  [ term  range new ] term  x) (sym ++') p))
      (no p)  failure "ENACT failed at [ term  range new ] term  (CCMaxTermLengthOf s + e)"
    $\begin{pmatrix} \,\href{Ledger.Core.Specification.Gov.Actions.html#2516}{\htmlId{1826}{\htmlClass{InductiveConstructor}{\text{NewConstitution}}}}\,     \\ \,\htmlId{1848}{\htmlClass{Symbol}{\text{\_}}}\,               \end{pmatrix}$  success (-, Enact-NewConst)
    $\begin{pmatrix} \,\href{Ledger.Core.Specification.Gov.Actions.html#2554}{\htmlId{1904}{\htmlClass{InductiveConstructor}{\text{TriggerHardFork}}}}\,     \\ \,\htmlId{1926}{\htmlClass{Symbol}{\text{\_}}}\,               \end{pmatrix}$  success (-, Enact-HF)
    $\begin{pmatrix} \,\href{Ledger.Core.Specification.Gov.Actions.html#2592}{\htmlId{1976}{\htmlClass{InductiveConstructor}{\text{ChangePParams}}}}\,       \\ \,\htmlId{1998}{\htmlClass{Symbol}{\text{\_}}}\,               \end{pmatrix}$  success (-, Enact-PParams)
    $\begin{pmatrix} \,\href{Ledger.Core.Specification.Gov.Actions.html#2668}{\htmlId{2053}{\htmlClass{InductiveConstructor}{\text{Info}}}}\,                \\ \,\htmlId{2075}{\htmlClass{Symbol}{\text{\_}}}\,               \end{pmatrix}$  success (-, Enact-Info)
    $\begin{pmatrix} \,\href{Ledger.Core.Specification.Gov.Actions.html#2630}{\htmlId{2127}{\htmlClass{InductiveConstructor}{\text{TreasuryWithdrawal}}}}\,  \\ \,\href{Ledger.Conway.Specification.Enact.Properties.Computational.html#2149}{\htmlId{2149}{\htmlClass{Bound}{\text{wdrl}}}}\,            \end{pmatrix}$ 
      case  [ x  s .withdrawals  wdrl ] x  t  of  where
        (yes p)   success (-, Enact-Wdrl p)
        (no _)    failure "ENACT failed at [ x  (s .withdrawals  wdrl)  ] x  t"
  Computational-ENACT .completeness  s action _ p
    with action | p
  ... | $\begin{pmatrix} \,\htmlId{2450}{\htmlClass{DottedPattern Symbol}{\text{.}}}\,\,\href{Ledger.Core.Specification.Gov.Actions.html#2440}{\htmlId{2451}{\htmlClass{DottedPattern InductiveConstructor}{\text{NoConfidence}}}}\,        \\ \,\htmlId{2473}{\htmlClass{Symbol}{\text{\_}}}\,               \end{pmatrix}$ | Enact-NoConf   = refl
  ... | $\begin{pmatrix} \,\htmlId{2527}{\htmlClass{DottedPattern Symbol}{\text{.}}}\,\,\href{Ledger.Core.Specification.Gov.Actions.html#2478}{\htmlId{2528}{\htmlClass{DottedPattern InductiveConstructor}{\text{UpdateCommittee}}}}\,     \\ \,\htmlId{2550}{\htmlClass{Symbol}{\text{(}}}\,\,\href{Ledger.Conway.Specification.Enact.Properties.Computational.html#2551}{\htmlId{2551}{\htmlClass{Bound}{\text{new}}}}\, , \,\href{Ledger.Conway.Specification.Enact.Properties.Computational.html#2557}{\htmlId{2557}{\htmlClass{Bound}{\text{rem}}}}\, , \,\href{Ledger.Conway.Specification.Enact.Properties.Computational.html#2563}{\htmlId{2563}{\htmlClass{Bound}{\text{q}}}}\,\,\htmlId{2564}{\htmlClass{Symbol}{\text{)}}}\, \end{pmatrix}$ | Enact-UpdComm p
    rewrite dec-yes
      ( [ term  range new ] term
            CCMaxTermLengthOf s +' EnactEnv.epoch  )
      (subst ( x  [ term  range new ] term  x) ++' p) .proj
      = refl
  ... | $\begin{pmatrix} \,\htmlId{2794}{\htmlClass{DottedPattern Symbol}{\text{.}}}\,\,\href{Ledger.Core.Specification.Gov.Actions.html#2516}{\htmlId{2795}{\htmlClass{DottedPattern InductiveConstructor}{\text{NewConstitution}}}}\,     \\ \,\htmlId{2817}{\htmlClass{Symbol}{\text{\_}}}\,               \end{pmatrix}$ | Enact-NewConst = refl
  ... | $\begin{pmatrix} \,\htmlId{2871}{\htmlClass{DottedPattern Symbol}{\text{.}}}\,\,\href{Ledger.Core.Specification.Gov.Actions.html#2554}{\htmlId{2872}{\htmlClass{DottedPattern InductiveConstructor}{\text{TriggerHardFork}}}}\,     \\ \,\htmlId{2894}{\htmlClass{Symbol}{\text{\_}}}\,               \end{pmatrix}$ | Enact-HF       = refl
  ... | $\begin{pmatrix} \,\htmlId{2948}{\htmlClass{DottedPattern Symbol}{\text{.}}}\,\,\href{Ledger.Core.Specification.Gov.Actions.html#2592}{\htmlId{2949}{\htmlClass{DottedPattern InductiveConstructor}{\text{ChangePParams}}}}\,       \\ \,\htmlId{2971}{\htmlClass{Symbol}{\text{\_}}}\,               \end{pmatrix}$ | Enact-PParams  = refl
  ... | $\begin{pmatrix} \,\htmlId{3025}{\htmlClass{DottedPattern Symbol}{\text{.}}}\,\,\href{Ledger.Core.Specification.Gov.Actions.html#2668}{\htmlId{3026}{\htmlClass{DottedPattern InductiveConstructor}{\text{Info}}}}\,                \\ \,\htmlId{3048}{\htmlClass{Symbol}{\text{\_}}}\,               \end{pmatrix}$ | Enact-Info     = refl
  ... | $\begin{pmatrix} \,\htmlId{3102}{\htmlClass{DottedPattern Symbol}{\text{.}}}\,\,\href{Ledger.Core.Specification.Gov.Actions.html#2630}{\htmlId{3103}{\htmlClass{DottedPattern InductiveConstructor}{\text{TreasuryWithdrawal}}}}\,  \\ \,\href{Ledger.Conway.Specification.Enact.Properties.Computational.html#3125}{\htmlId{3125}{\htmlClass{Bound}{\text{wdrl}}}}\,            \end{pmatrix}$ | Enact-Wdrl p
    rewrite dec-yes ( [ x  s .withdrawals  wdrl ] x  EnactEnv.treasury  ) p .proj
    = refl
