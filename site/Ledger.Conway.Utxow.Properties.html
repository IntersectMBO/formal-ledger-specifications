  (txs : _) (open TransactionStructure txs)

open import Ledger.Conway.Utxow txs abs

instance
  Computational-UTXOW : Computational ___,UTXOW_ String
  Computational-UTXOW = record {Go}
    where module Go  s tx (let H ,  H? = UTXOW-inductive-premises {s = s} {tx = tx}) where

    open Computational Computational-UTXO
      renaming (computeProof to computeProof'; completeness to completeness')

    genErr :  H  String
    genErr  p = case dec-de-morgan p of  where
      (inj a)  " [ (vk , )  vkSigs ] isSigned vk (txidBytes txid) "
      (inj b)  case dec-de-morgan b of  where
        (inj a)  "[ s  scriptsP1 ] validP1Script witsKeyHashes txvldt s"
        (inj b)  case dec-de-morgan b of  where
          (inj a)  "witsVKeyNeeded utxo txb  witsKeyHashes"
          (inj b)  case dec-de-morgan b of  where
            (inj a)  "(neededHashes  refScriptHashes)  witsScriptHashes"
            (inj b)  case dec-de-morgan b of  where
              (inj a)  "inputHashes  txdatsHashes"
              (inj b)  case dec-de-morgan b of  where
                (inj a)  "txdatsHashes  (inputHashes  allOutHashes  getDataHashes (range (utxo  refInputs)))"
                (inj b)  case dec-de-morgan b of  where
                  (inj a)  "languages  allowedLanguages"
                  (inj b)  "txADhash  map hash txAD"

    computeProof : ComputationResult String ( (  s  tx ,UTXOW_))
    computeProof =
      case H? of  where
        (yes (p , p , p , p , p , p , p , p)) 
          map (map (UTXOW-inductive p p p p p p p p)) (computeProof'  s tx)
        (no p)  failure $ genErr p

    completeness :  s'    s  tx ,UTXOW s'
                         map proj computeProof  success s'
    completeness s' (UTXOW-inductive p p p p p p p p h) with H?
    ... | no p = -elim $ p ((p , p , p , p , p , p , p , p))
    ... | yes _ with computeProof'  s tx | completeness' _ _ _ _ h
    ... | success _ | refl = refl
