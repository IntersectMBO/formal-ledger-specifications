instance
  Computational-DELEG : Computational ___,DELEG_ String
  Computational-DELEG .computeProof de st =
    let open DelegEnv de; open DState st in
     where
    (delegate c mv mc d)  case  (c  dom rewards  d  pparams .PParams.keyDeposit)
                                 (c  dom rewards  d  0)
                                 mv  map (just  vDelegCredential) delegatees 
                                    fromList ( nothing  just vDelegAbstain  just vDelegNoConfidence  [] )
                                 mc  map just (dom pools)   nothing 
                                 of  where
      (yes p)  success (-, DELEG-delegate p)
      (no p)  failure (genErrors p)
    (dereg c d)  case  (c , 0)  rewards  of  where
      (yes p)  success (-, DELEG-dereg p)
      (no p)  failure (genErrors p)
    (reg c d)  case  c  dom rewards
                      (d  pparams .PParams.keyDeposit  d  0)
                      of  where
      (yes p)  success (-, DELEG-reg p)
      (no p)  failure (genErrors p)
    _  failure "Unexpected certificate in DELEG"
  Computational-DELEG .completeness de st (delegate c mv mc d)
    s' (DELEG-delegate p) rewrite dec-yes ( (c  dom (DState.rewards st)  d  DelegEnv.pparams de .PParams.keyDeposit)
                                 (c  dom (DState.rewards st)  d  0)
                                 mv  map (just  vDelegCredential) (DelegEnv.delegatees de)  fromList ( nothing  just vDelegAbstain  just vDelegNoConfidence  [] )
                                 mc  map just (dom (DelegEnv.pools de))   nothing 
                                           ) p .proj = refl
  Computational-DELEG .completeness de st (dereg c d) _ (DELEG-dereg p)
    rewrite dec-yes ( (c , 0)  (DState.rewards st) ) p .proj = refl
  Computational-DELEG .completeness de st (reg c d) _ (DELEG-reg p)
    rewrite dec-yes ( c  dom (DState.rewards st)  (d  DelegEnv.pparams de .PParams.keyDeposit  d  0) ) p .proj = refl

  Computational-POOL : Computational ___,POOL_ String
  Computational-POOL .computeProof _ st (regpool c _) = success (-, POOL-regpool)
  Computational-POOL .computeProof _ _ (retirepool c e) = success (-, POOL-retirepool)
  Computational-POOL .computeProof _ _ _ = failure "Unexpected certificate in POOL"
  Computational-POOL .completeness _ st (regpool c _) _ POOL-regpool = refl
  Computational-POOL .completeness _ _ (retirepool _ _) _ POOL-retirepool = refl

  Computational-GOVCERT : Computational ___,GOVCERT_ String
  Computational-GOVCERT .computeProof ce st (regdrep c d _) =
    let open CertEnv ce; open PParams pp in
    case  (d  drepDeposit  c  dom (GState.dreps st))
          (d  0  c  dom (GState.dreps st))  of  where
      (yes p)  success (-, GOVCERT-regdrep p)
      (no p)  failure (genErrors p)
  Computational-GOVCERT .computeProof _ st (deregdrep c _) =
    let open GState st in
    case c ? dom dreps of  where
      (yes p)  success (-, GOVCERT-deregdrep p)
      (no p)   failure (genErrors p)
  Computational-GOVCERT .computeProof ce st (ccreghot c _) =
    let open CertEnv ce; open GState st in
    case  ((c , nothing)  ccHotKeys )  c  coldCreds  of  where
      (yes p)  success (-, GOVCERT-ccreghot p)
      (no p)  failure (genErrors p)
  Computational-GOVCERT .computeProof _ _ _ = failure "Unexpected certificate in GOVCERT"
  Computational-GOVCERT .completeness ce st
    (regdrep c d _) _ (GOVCERT-regdrep p)
    rewrite dec-yes
       (let open CertEnv ce; open PParams pp; open GState st in
        (d  drepDeposit  c  dom dreps)  (d  0  c  dom dreps))
       p .proj = refl
  Computational-GOVCERT .completeness _ st
    (deregdrep c _) _ (GOVCERT-deregdrep p)
    rewrite dec-yes (c ? dom (GState.dreps st)) p .proj = refl
  Computational-GOVCERT .completeness ce st
    (ccreghot c _) _ (GOVCERT-ccreghot p)
    rewrite dec-yes ( (((c , nothing)  (GState.ccHotKeys st) )  c  CertEnv.coldCreds ce) ) p .proj = refl

  ... | success _ | refl = refl
  Computational-CERT .completeness ce cs
    dCert@(reg c d) cs' (CERT-deleg h)
    with computeProof $\begin{pmatrix} \,\href{Ledger.Conway.Specification.Certs.html#4210}{\htmlId{6070}{\htmlClass{Field}{\text{CertEnv.pp}}}}\, \,\href{Ledger.Conway.Specification.Certs.Properties.Computational.html#6001}{\htmlId{6081}{\htmlClass{Bound}{\text{ce}}}}\, \\ \,\href{Ledger.Conway.Specification.Certs.html#4538}{\htmlId{6086}{\htmlClass{Field}{\text{PState.pools}}}}\, \,\htmlId{6099}{\htmlClass{Symbol}{\text{(}}}\,\,\href{Ledger.Conway.Specification.Certs.html#4894}{\htmlId{6100}{\htmlClass{Field}{\text{pState}}}}\, \,\href{Ledger.Conway.Specification.Certs.Properties.Computational.html#6004}{\htmlId{6107}{\htmlClass{Bound}{\text{cs}}}}\,\,\htmlId{6109}{\htmlClass{Symbol}{\text{)}}}\, \\ \,\href{Class.IsSet.html#916}{\htmlId{6113}{\htmlClass{Function}{\text{dom}}}}\, \,\htmlId{6117}{\htmlClass{Symbol}{\text{(}}}\,\,\href{Ledger.Conway.Specification.Certs.html#4706}{\htmlId{6118}{\htmlClass{Field}{\text{GState.dreps}}}}\, \,\htmlId{6131}{\htmlClass{Symbol}{\text{(}}}\,\,\href{Ledger.Conway.Specification.Certs.html#4914}{\htmlId{6132}{\htmlClass{Field}{\text{gState}}}}\, \,\href{Ledger.Conway.Specification.Certs.Properties.Computational.html#6004}{\htmlId{6139}{\htmlClass{Bound}{\text{cs}}}}\,\,\htmlId{6141}{\htmlClass{Symbol}{\text{))}}}\, \end{pmatrix}$ (dState cs) dCert | completeness _ _ _ _ h
  ... | success _ | refl = refl
  Computational-CERT .completeness ce cs
    dCert@(dereg c _) cs' (CERT-deleg h)
    with computeProof $\begin{pmatrix} \,\href{Ledger.Conway.Specification.Certs.html#4210}{\htmlId{6327}{\htmlClass{Field}{\text{CertEnv.pp}}}}\, \,\href{Ledger.Conway.Specification.Certs.Properties.Computational.html#6256}{\htmlId{6338}{\htmlClass{Bound}{\text{ce}}}}\, \\ \,\href{Ledger.Conway.Specification.Certs.html#4538}{\htmlId{6343}{\htmlClass{Field}{\text{PState.pools}}}}\, \,\htmlId{6356}{\htmlClass{Symbol}{\text{(}}}\,\,\href{Ledger.Conway.Specification.Certs.html#4894}{\htmlId{6357}{\htmlClass{Field}{\text{pState}}}}\, \,\href{Ledger.Conway.Specification.Certs.Properties.Computational.html#6259}{\htmlId{6364}{\htmlClass{Bound}{\text{cs}}}}\,\,\htmlId{6366}{\htmlClass{Symbol}{\text{)}}}\, \\ \,\href{Class.IsSet.html#916}{\htmlId{6370}{\htmlClass{Function}{\text{dom}}}}\, \,\htmlId{6374}{\htmlClass{Symbol}{\text{(}}}\,\,\href{Ledger.Conway.Specification.Certs.html#4706}{\htmlId{6375}{\htmlClass{Field}{\text{GState.dreps}}}}\, \,\htmlId{6388}{\htmlClass{Symbol}{\text{(}}}\,\,\href{Ledger.Conway.Specification.Certs.html#4914}{\htmlId{6389}{\htmlClass{Field}{\text{gState}}}}\, \,\href{Ledger.Conway.Specification.Certs.Properties.Computational.html#6259}{\htmlId{6396}{\htmlClass{Bound}{\text{cs}}}}\,\,\htmlId{6398}{\htmlClass{Symbol}{\text{))}}}\, \end{pmatrix}$ (dState cs) dCert | completeness _ _ _ _ h

  Computational-PRE-CERT : Computational ___,PRE-CERT_ String
  Computational-PRE-CERT .computeProof ce cs _ =
    let open CertEnv ce; open PParams pp
        open GState (gState cs); open DState (dState cs)
        refresh = mapPartial (isGovVoterDRep  voter) (fromList votes)
        refreshedDReps  = mapValueRestricted (const (CertEnv.epoch ce + drepActivity)) dreps refresh
    in case  filter isKeyHash (map RwdAddr.stake (dom wdrls))  dom voteDelegs
               map (map RwdAddr.stake) (wdrls )  rewards   of  where
      (yes p)  success (-, CERT-pre p)
      (no p)  failure (genErrors p)
  Computational-PRE-CERT .completeness ce st _ st' (CERT-pre p)
    rewrite let dState = CertState.dState st; open DState dState in
      dec-yes  filter isKeyHash (map RwdAddr.stake (dom (CertEnv.wdrls ce)))  dom voteDelegs
                 map (map RwdAddr.stake) (CertEnv.wdrls ce )  rewards  
        p .proj = refl

  -- POST-CERT has no premises, so computing always succeeds
  -- with the unique post-state and proof CERT-post.
  Computational-POST-CERT : Computational ___,POST-CERT_ String
  Computational-POST-CERT .computeProof ce cs tt = success ( cs' , CERT-post)
    where
      dreps : DReps
      dreps = GState.dreps (gState cs)
      validVoteDelegs : VoteDelegs
      validVoteDelegs = (VoteDelegsOf cs) ^ ( map vDelegCredential (dom dreps)  fromList (vDelegNoConfidence  vDelegAbstain  []) )
      cs' : CertState
      cs' = $\begin{pmatrix} \begin{pmatrix} \,\href{Ledger.Conway.Specification.Certs.Properties.Computational.html#8613}{\htmlId{8816}{\htmlClass{Function}{\text{validVoteDelegs}}}}\, \\ \,\href{Ledger.Conway.Specification.Certs.html#2307}{\htmlId{8834}{\htmlClass{Field}{\text{StakeDelegsOf}}}}\, \,\href{Ledger.Conway.Specification.Certs.Properties.Computational.html#8503}{\htmlId{8848}{\htmlClass{Bound}{\text{cs}}}}\, \\ \,\href{Ledger.Conway.Specification.Certs.html#2142}{\htmlId{8853}{\htmlClass{Field}{\text{RewardsOf}}}}\, \,\href{Ledger.Conway.Specification.Certs.Properties.Computational.html#8503}{\htmlId{8863}{\htmlClass{Bound}{\text{cs}}}}\, \end{pmatrix} \\ \,\href{Ledger.Conway.Specification.Certs.html#5238}{\htmlId{8870}{\htmlClass{Field}{\text{PStateOf}}}}\, \,\href{Ledger.Conway.Specification.Certs.Properties.Computational.html#8503}{\htmlId{8879}{\htmlClass{Bound}{\text{cs}}}}\, \\ \,\href{Ledger.Conway.Specification.Certs.html#5346}{\htmlId{8884}{\htmlClass{Field}{\text{GStateOf}}}}\, \,\href{Ledger.Conway.Specification.Certs.Properties.Computational.html#8503}{\htmlId{8893}{\htmlClass{Bound}{\text{cs}}}}\, \end{pmatrix}$

  -- Completeness: the relational proof pins s' to exactly `post`,
  -- and computeProof returns success at that same state; so refl.
  Computational-POST-CERT .completeness ce cs _ cs' CERT-post = refl
