  -- Properties of domains of maps of type m  m ---------------------

  -- 1. If `k  dom m  dom m` (for m, m maps), then `(k , v)  m  m` for some `v`.
  dom- : {m m : A  B}  k  dom m  dom m   B (   (k , )  m  m)
  dom- k = from dom (domdom k)

  -- 2. If `(k , v)  m  m`, then `k  dom m  dom m`.
  -dom : {m m : A  B}{k : A} {v : B}  (k , v)  m  m  k  dom m  dom m
  -dom {v = v} kv = domdom (to dom (v , kv))

  -- 3. The image of a key `k  dom m  dom m` under the map `m  m` is
  --    `fold id id __ (unionThese m m k p)`.
  __ : (m m : A  B)  k  dom m  dom m  B
  __ {k} m m p = fold id id __ (unionThese m m k p)

  -- 4. F[ m , m ] takes a key `k` and a proof of `k  dom m  dom m` and returns
  --    the pair `(k , v)` where `v` is the unique image of `k` under `m  m`.
  --    i.e., `(k , v)  m  m`.
  F[_,_] : (m m : A  B)   A (_ dom m  dom m)  A  B
  F[ m , m ] (x , x) = x ,  m  m  x

  -- 5. A simpler version of `lookup`; it doesn't require tactics.
  lookup : (m : A  B)  k  dom m  B
  lookup _ = proj  (from dom)

  -- 6. Proof that the value you get from `lookup` is in the image of the map.
  -lookup : (m : A  B)(k : k  dom m)  (k , lookup m k)  m
  -lookup m k = proj (from dom k)

  -- 7. Irrelevance of the proof of `k  dom m` used in `lookup`.
  lookup-irrelevance : (m : A  B) {k k : k  dom m}
                        lookup m k  lookup m k
  lookup-irrelevance m {k} {k} = m .proj (-lookup m k) (-lookup m k)

  -- 8. If `v` is the image of `k` under `m`, then it must be `lookup m km`!
  -lookup : (m : A  B) {km : k  dom m}  (k , v)  m  v  lookup m km
  -lookup m {km} kv = m .proj kv (-lookup m km)

  lookup : (m : A  B) {k : k  dom m}  lookup m k  lookup m k
  lookup {k = k} _ {k} = refl

    -- 0. The `-incl-set` lemma is useful for proving some properties of `__`.
    -incl-set : {X :  A} {a : A} (aX : a  X)   (a  X)    (a , )  incl-set X
    -incl-set {X} {a} aX =
      Data.Product.map ( {aX} eq  -mapPartial {f = incl-set' X} .to (a , aX , eq))
                        lem
      where
        lem :  (a  X)  aX  incl-set' X a  just (a , aX)
        lem with a ? X
        ... | yes aX = aX , refl
        ... | no  aX  = -elim (aX aX)

    -- Properties of values of  --------------------------------------------------

    -- 1. If `k  dom m  dom m` holds, then there is a particular proof `k`
    --    of that fact such that `(k ,  m  m  k)  m  m`.
    -- k  : {m m : A  B}  k  dom m  dom m
    --               (k  dom m  dom m)  k  (k ,  m  m  k)  m  m
    -- k {k = k} k with -incl-set k
    -- ... | k , kk = k , to -map ((k , k) , refl , kk)

    -- Actually, we won't use the general statement above; we only need the following
    -- version which picks a particular proof of the fact that `k  dom m  dom m`.
    -- In fact, for computing the value, the proof is irrelevant (see property 3 below).

    -- 2. We can obtain the particular proof mentioned in 3. using `-incl-set`.
    k'  : {m m : A  B} (k : k  dom m  dom m)
                  (k ,  m  m  (-incl-set k .proj))  m  m
    k' {k = k} {m} {m} k = goal
      where
      k : k  dom m  dom m
      k = -incl-set k .proj  --  <= the particular proof mentioned in 1 above.

      kk : (k , -incl-set k .proj)  incl-set (dom m  dom m)
      kk = -incl-set k .proj

      goal : F[ m , m ] (k , -incl-set k .proj)  map F[ m , m ] (incl-set (dom m  dom m))
                                                -- this ^^^^^^^^^^^^^^^^^^^^^^^^^^^^^ is `(m  m)`
      goal = to -map ((k , k) , refl , kk)

    -- 3. The value associated with a key doesn't depend on the proof of key membership.
    fold-irrelevance : {m m : A  B} {k k : k  (dom m  dom m)}
                       m  m  k   m  m  k
    fold-irrelevance {k = k} {m = m} {m} {k} with k ? dom m | k ? dom m
    ... | yes km | yes km = refl
    ... | yes km | no km  = refl
    ... | no km  | yes km = refl
    ... | no km  | no km with from - k
    ... | inj km = -elim (km km)
    ... | inj km = -elim (km km)

    -- 4. If `(k , v)  m  m`, then there is a particular proof `k` of
    --    `k  dom m  dom m` such that `v   m  m  k`.
    -unique-val  : {m m : A  B} (k : k  dom m  dom m)  (k , v)  m  m
                    v   m  m  (-incl-set k .proj)
    -unique-val {k = k} {m = m} {m} k kv =
      (m  m) .proj kv (to -map ((k , -incl-set k .proj) , refl , -incl-set k .proj))

    -- 5. If `k  dom m  dom m`, and `k  dom m` and `k  dom m`, then there's a
    --    proof `k` of `k  dom m  dom m` such that ` m  m  k`.
    lulu  :  {m m : A  B} (k : k  dom m  dom m)
                    {km : k  dom m} {km : k  dom m}
                    (k  dom m  dom m)  k   m  m  k  lookup m km  lookup m km
    lulu {k} {m} {m} k {km} {km} with k ? dom m | k ? dom m
    ... | no km   | _       = -elim (km km)
    ... | _         | no km = -elim (km km)
    ... | yes km | yes km = k , goal
      where
      open -Reasoning
      goal : lookup m km  lookup m km  lookup m km  lookup m km
      goal = begin
        lookup m km  lookup m km   cong (_ lookup m km) (lookup-irrelevance m) 
        lookup m km  lookup m km    cong (lookup m km _ ) (lookup-irrelevance m) 
        lookup m km  lookup m km    

    -- 5.' Again, we won't need the general statement (`lulu`), but instead the version below
    --     which picks a particular proof of `k  dom m  dom m`.
    lulu'  :  {m m : A  B} (kv : (k , v)  m  m)
                     {km : k  dom m} {km : k  dom m}
                     m  m  (-incl-set (-dom kv) .proj)  lookup m km  lookup m km
    lulu' kv with lulu (-dom kv)
    ... | k , v = trans fold-irrelevance v

    res-dom :  m {ks}{a : A}{b : B}  (a , b)  (m  ks )  (a , b)  m  a  ks
    res-dom m x = ex- x , (-res-dom $ -dom x) .proj

    deconstruct-  :  {m m m : A  B} {a : A}
                       {a : a  dom m  dom m}
                       {a : a  dom m  dom m}
                      m  m   m  m  a   m  m  a

    deconstruct- {m} {m} {m} {a} {a} mm
      with a ? dom m | a ? dom m | a ? dom m
    ... | yes am | yes am | yes am =
      cong ( (b : B)  lookup m a  b)
           (proj m
             (proj mm (proj $ from dom am))  -- : (a , lookup m a)  (m )
             (proj (from dom am))                -- : (a , lookup m a)  (m )
           )
    ... | no  am | yes am | yes am =
      proj m (proj mm (proj (from dom am))) (proj (from dom am))
    ... | yes am | no  am | no am = refl
    ... | _ | yes am | no am = -elim (am (dom (proj mm) am))
    ... | _ | no  am | yes am = -elim (am (dom (proj mm) am))
    ... | no  am | no  am | no am with from - a
    ... | inj am = -elim (am am)
    ... | inj am = -elim (am am)


    fold--union-comm  :  {m m : A  B} {a : A}
                          {a : a  dom m  dom m}
                          {a : a  dom m  dom m}
                          m  m  (-incl-set a .proj)
                          m  m  (-incl-set a .proj)

    fold--union-comm {m} {m} {a} {a} with a ? dom m | a ? dom m
    ... | yes am | yes am = comm (lookup m a) (lookup m a)
    ... | no  am | yes am = refl
    ... | yes am | no  am = refl
    ... | no  am | no  am with from - a
    ... | inj am = -elim (am am)
    ... | inj am = -elim (am am)


    -comm- : {m m : A  B}  m  m  m  m
    -comm- {m} {m} {a} {b} ab with a ? dom m | a ? dom m
    ... | yes am | _ = to -map $ (a , -incl-set a .proj)
                                  , -, (refl , b) , -incl-set a .proj
      where
      a : a  unions (fromList (dom m  dom m  [])) .proj
      a = to -unions (dom m , to -fromList (there (here refl)) , am)

      a : a  dom m  dom m
      a = to - (inj am)

      b : b  fold id id __ (unionThese m m a $ -incl-set a .proj)
      b = trans (-unique-val (to - $ inj am) ab) fold--union-comm

    ... | no am | yes am = to -map $ (a , -incl-set a .proj)
                                        , -, (refl , b) , -incl-set a .proj
      where
      a : a  unions (fromList (dom m  dom m  [])) .proj
      a = to -unions (dom m , to -fromList (here refl) , am)

      a : a  dom m  dom m
      a = to - $ inj am

      b : b  fold id id __ (unionThese m m a (-incl-set a .proj))
      b = trans (-unique-val (to - (inj am)) ab) fold--union-comm

    ... | no  am | no am with from - (-dom ab)
    ... | inj am = -elim (am am)
    ... | inj am = -elim (am am)


    -comm : {m m : A  B}  m  m  m  m
    -comm = -comm- , -comm-

    -comm-val  :  {m m : A  B}
                    {km : k  dom m  dom m}
                    {km : k  dom m  dom m}
                    m  m  km   m  m  km
    -comm-val {k = k} {m = m}{m}{km}{km} = (m  m) .proj kv (-comm- kv)
      where
      kv : (k ,  m  m  km)  m  m
      kv = subst (   (k , )  m  m) fold-irrelevance (k' km)

      kv : (k ,  m  m  km)  m  m
      kv = subst ( x  (k , x)  m  m) fold-irrelevance (k' km)


    -cong- : {m m m : A  B}  m  m  m  m  m  m
    -cong- {m}{m}{m} mm@(mm , mm) {k} {v} kv with from -map kv
    ... | (.k , k) , refl , s =
      let k'' , inclset = -incl-set k'
          F : (k , v)  F[ m , m ] (k , k'')
          F = -, (refl , deconstruct- {a = k} mm)
      in  to (-map {f = F[ m , m ]}) ((k , k'') , F , inclset)
      where
      a-dom : k  dom m  dom m
      a-dom = domdom (to dom (v , kv))

      domdom : dom m  dom m
      domdom = dom mm

      k' : k  dom m  dom m
      k' = -cong- id domdom a-dom

    -cong-l : {m : A  B}  (m _ ) Preserves __  __
    -cong-l mm@(mm , mm) = (-cong- mm) , -cong- (mm , mm)

    -cong-r : {m : A  B}  ( _ m) Preserves __  __
    -cong-r mm .proj kvmm = proj -comm (-cong- mm (proj -comm kvmm))
    -cong-r mm@(mm , mm) .proj kvmm =
      proj -comm (-cong- (mm , mm) (proj -comm kvmm))

    -dom-id : (m : A  B)  dom m  dom m  dom ({A  B})
    -dom-id m = begin
      dom m  -identity (dom m) 
      dom m    -cong .refl dom 
      dom m  dom ({A  B})
      
      where
      open SetoidReasoning (-Setoid{A})
      module  = IsEquivalence (-isEquivalence {A})

    -id-dom :  (m : A  B)  k  dom m    k  dom m  dom ({A  B})
    -id-dom m = mk (-dom-id m .proj) (-dom-id m .proj)

    -id-lemma  :  (m : A  B)
                    (km : k  dom m)
                    (k : k  dom m  dom ({A  B}))
                   lookup m km   m  {A  B}  k

    -id-lemma {k} m kdomm kdomm with k ? dom m | k ? dom ({A  B})
    ... | _ | yes  k = -elim (-elim (-dom k))
    ... | no  km | no  k = case from - kdomm of  where
      (inj km)  -elim (km km)
      (inj k)  -elim (k k)
    ... | yes km | no  k with from -map kdomm
    ... | (.k , v) , refl , kvm = m .proj kvm (-lookup m km)
                                  -- goal : v  lookup m km --

    -id-r : (m : A  B)  m  {A  B}  m
    -id-r m .proj {(k , v)} kvm with from -map kvm
    ... | (.k , k) , refl , snd = subst  (   (k , )  m)
                                          (-id-lemma m (from (-id-dom m) k) k)
                                          (-lookup m $ from (-id-dom m) k)

    -id-r m .proj {(k , v)} kvm with to dom (v , kvm)
    ... | km =
      subst (   (k , )  m  {A  B}) (trans lu v) (k' k)
      where
      k : k  dom m  dom ({A  B})
      k = to (-id-dom m) km

      lu :  m  ({A  B})  (-incl-set k .proj)  lookup m (-incl-set km .proj)
      lu = sym $ -id-lemma m (-incl-set km .proj) (-incl-set k .proj)

      v : lookup m (-incl-set km .proj)  v
      v = sym $ m .proj kvm (-lookup m (-incl-set km .proj))

    restrict-cong : (m m : A  B) {ks :  A}  m  m  (m  ks )  (m  ks )

    ... | ab , a = res-dom m (mm ab , a)
    restrict-cong m m (_ , mm) .proj ab with res-dom m ab
    ... | ab , a = res-dom m (mm ab , a)

    ... | b , kbfilter = to dom (b , proj ((from -filter) kbfilter))

    cong-filter m m eq .proj Pm = filter- m (-dom-filter m (-dom Pm) .proj) (eq .proj (-filter .from Pm .proj))

    -filter-P : (m m : A  B)  (k , v)  filter P m  filter P m  P (k , v)
    -filter-P {k = k}{v} m m kv with (from - (-dom kv))
    ... | inj k = -dom-filter m k .proj
    ... | inj k = -dom-filter m k .proj

                           lookup m k  lookup (filter P m) k
    lookuplookup-filter m k k =
      (m .proj) (-lookup m k) (proj (from -filter (-lookup (filter P m) k)))

    -filter  : (m m : A  B) {a : A} {b : B}
                (a , b)  filter P m  filter P m
                a  dom m  a  dom m  (a , b)  filter P m
    -filter m m {a} {b} ab' a a =
      subst (   (a , )  filter P m) (sym $ --l af ab' af)
                                            (from dom af .proj)
      where
      af : a  dom (filter P m)
      af = -dom-filter m (-filter-P m m ab' , a)

      af : a  dom (filter P m)
      af = a  (-dom-filter-dom m)

    -filter-lookup  :  (m m : A  B) {a : A} {b : B}
                        (a , b)  filter P m  filter P m
                        (a : a  dom m)  a  dom m

      -- filter--distr  -----------------------------------------------------------------------------
      -- Note: this property only holds because P is not looking at the value.
      -- Counter-example if it does look at the value:
      -- Suppose `m = m = {(0, 1)}`, `P (0, 1)`, and ` P (0, 2)`.
      -- Then `m  m  {(0, 2)}` so (lhs) `filter P (m  m)` is empty,
      -- but `(filter P m)  (filter P m) = {(0, 1)}` so (rhs) `filter P m  filter P m` contains {(0, 2)}.

      filter--distr- : (m m : A  B)  filter P m  filter P m  filter P (m  m)
      filter--distr- m m {k} {v} kvPmPm =
        case  P k  of  where
          (yes pk)  yes-case pk
          (no pk)  -elim (pk ([ -dom-filter-P m , -dom-filter-P m ] kPmkPm))
        where
          open -Reasoning
          kPmkPm : k  dom (filter P m)  k  dom (filter P m)
          kPmkPm = - .from (domdom (-map kvPmPm))

          kPmkPm : These (k  dom (filter P m)) (k  dom (filter P m))
          kPmkPm with k ? dom (filter P m) | k ? dom (filter P m) | kPmkPm
          ... | yes  | yes  | _       = these  
          ... | yes  | no  _  | _       = this 
          ... | no  _  | yes  | _       = that 
          ... | no   | no  _  | inj  = -elim ( )
          ... | no  _  | no   | inj  = -elim ( )

          yes-case : P k  (k , v)  filter P (m  m)
          yes-case pk = -filter .to (pk , kvmm)
            where
              kmkm : k  dom m  k  dom m
              kmkm = map- (dom-filter- m) (dom-filter- m) kPmkPm

              kmm : k  dom m  dom m
              kmm = - .to kmkm

              kmm : k  dom (m  m)
              kmm = domdom kmm

              [kvmm] :  (k  dom m  dom m) ( kmm  (k ,  m  m  kmm)  m  m)
              [kvmm] = _ , -map (-incl-set kmm .proj)

              kmm : k  dom m  dom m
              kmm  = [kvmm] .proj

              kmm : k  dom (m  m)
              kmm  = domdom kmm

              v : B
              v =  m  m  kmm

              kvmm : (k , v)  m  m
              kvmm = [kvmm] .proj

              v=v : v  v
              v=v = --filter kvPmPm kmm kmm kPmkPm

              kvmm : (k , v)  m  m
              kvmm = subst (   (k , )  m  m) (sym v=v) kvmm

      filter--distr- : (m m : A  B)  filter P (m  m)  filter P m  filter P m
      filter--distr- m m {k} {v} kvPmm with from -filter kvPmm
      ... | Pkv , kv with k ? dom m | k ? dom m
      ... | yes k | yes k =
        subst (   (k , )  m  m) ((m  m) .proj kb kv) (b .proj)
          where
          open -Reasoning
          m m : A  B; m = filter P m; m = filter P m

          kdom : k  dom m  dom m
          kdom = to - $ inj (-dom-filter m (Pkv , k))

          b :  B    (k , )  m  m
          b =  m  m  (-incl-set kdom .proj) , k' kdom

          b : B
          b = b .proj

          kb : (k , b)  filter P (m  m)
          kb = filter--distr- m m (b .proj)

          kb : (k , b)  m  m
          kb = (from -filter kb) .proj

      ... | no k | yes k =
        subst (   (k , )  m  m) ((m  m) .proj kb kv) (b .proj)
          where
          open -Reasoning
          m m : A  B; m = filter P m; m = filter P m

          kdom : k  dom m  dom m
          kdom = to - $ inj (-dom-filter m (Pkv , k))

          b :  B    (k , )  m  m
          b =  m  m  (-incl-set kdom .proj) , k' kdom

          b : B
          b = b .proj

          kb : (k , b)  filter P (m  m)
          kb = filter--distr- m m (b .proj)

          kb : (k , b)  m  m
          kb = (from -filter kb) .proj


      ... | yes k | no k =
        subst (   (k , )  m  m) ((m  m) .proj kb kv) (b .proj)
          where
          open -Reasoning
          m m : A  B; m = filter P m; m = filter P m

          kdom : k  dom m  dom m
          kdom = to - $ inj (-dom-filter m (Pkv , k))

          b :  B    (k , )  m  m
          b =  m  m  (-incl-set kdom .proj) , k' kdom

          b : B
          b = b .proj

          kb : (k , b)  filter P (m  m)
          kb = filter--distr- m m (b .proj)

          kb : (k , b)  m  m
          kb = (from -filter kb) .proj

      ... | no k | no k with from - (-dom kv)
      ... | inj k = -elim (k k)
      ... | inj k = -elim (k k)

      filter--distr : (m m : A  B)  filter P (m  m)  filter P m  filter P m
      filter--distr m m = filter--distr- m m , filter--distr- m m


      filter-singleton-true : P k  filter P  k , v    k , v 
      filter-singleton-true p .proj = proj  (from -filter)
      filter-singleton-true {k}{v} p .proj {a} x = to -filter (subst P (sym (from -singleton x)) p , x)

      filter-restrict :  m {ks}  filter P (m  ks )  filter P m  ks 
      filter-restrict m {ks} .proj {a , b} h with from -filter h
      ... | Pa , abmks with res-dom m abmks
      ... | abm , aks = res-dom (filter P m) (to -filter (Pa , abm) , aks)
      filter-restrict m {ks} .proj {a , b} h with res-dom (filter P m) h
      ... | abm , aks = to -filter (from -filter abm .proj
                                       , res-dom m (from -filter abm .proj , aks))

      -filter-res- : {x : A  B} (m : A  B)  x  filter P m   k   P x  [ b ] x  (k , b)
      -filter-res- m x = proj (from -filter $ res- x) , res-singleton''{m = filter P m} x
