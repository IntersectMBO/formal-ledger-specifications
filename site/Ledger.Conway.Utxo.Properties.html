instance
  Computational-UTXOS : Computational ___,UTXOS_ String
  Computational-UTXOS = record {go} where
    module go ( : UTxOEnv) (s : UTxOState) (tx : Tx)
      (let open UTxOState s)
      (let H-Yes ,  H-Yes? = Scripts-Yes-premises {} {tx} {utxo} {deposits})
      (let H-No  ,  H-No?  = Scripts-No-premises  {} {tx} {utxo}) where
      open Tx tx renaming (body to txb); open TxBody txb
      open UTxOEnv  renaming (pparams to pp)
      sLst = collectPhaseTwoScriptInputs pp tx utxo

      computeProof =
        case H-Yes? , H-No? of  where
          (yes p , no _ )  success (_ , (Scripts-Yes p))
          (no _  , yes p)  success (_ , (Scripts-No p))
          (_     , _    )  failure "isValid check failed"

      completeness :  s'    s  tx ,UTXOS s'  map proj computeProof  success s'
      completeness _ (Scripts-Yes p) with H-No? | H-Yes?
      ... | yes (_ , refl) | _     = case proj p of  ()
      ... | no _           | yes _ = refl
      ... | no _           | no p = case p p of  ()
      completeness _ (Scripts-No p) with H-Yes? | H-No?
      ... | yes (_ , _ , refl) | _     = case proj p of  ()
      ... | no _               | yes _ = refl
      ... | no _               | no p = case p p of  ()

      begin

          .+ newDeposits pp utxoState txb

    .+_ dep .+ .0           .+-identity _ 
    .+_ dep 

  ref-tot-0 : ref  0  tot  0
  ref-tot-0 ref0 with dep
  ... | .+_ n     = -elim (ref0 refl)
  ... | .negsuc n = refl

  refdep : ref  dep
  refdep with ref  0
  ... | no p =  $ _ ,_ $ begin
    ref + uDep  +-comm ref uDep 
    uDep + ref  dep-ref $ ref-tot-0 p 
    dep        
  ... | yes p rewrite p = zn

  deposits-change : uDep  dep + tot - ref
  deposits-change = .+-injective $ begin
    .+_ uDep                                  .+-identity _ 
    .+_ uDep .+ .0                         cong! (.+-inverse (.+_ dep)) 
    .+_ uDep .+ (.-_ (.+_ dep) .+ (.+_ dep))
       .+-assoc (.+_ uDep) (.-_ (.+_ dep)) (.+_ dep) 
    (.+_ uDep .- (.+_ dep)) .+ (.+_ dep)  cong! (.m-nmn uDep dep) 
    dep .+ (.+_ dep)                        .+-comm dep (.+_ dep) 
    (.+_ dep) .+ dep                        cong! deposits-change' 
    (.+_ dep) .+ (tot - ref)                 .distrib--+-pos dep tot ref 
    (dep + tot) - ref                          .- (mnmn+o tot refdep) 
    .+_ (dep + tot - ref) 

  split-balance :  keys  cbalance utxo  cbalance (utxo  keys ) + cbalance (utxo  keys)

                      cbalance utxo
                         balance-cong-coin {utxo = (utxo  keys )  (utxo  keys)}{utxo}
                          $ disjoint-- (disjoint-sym res-ex-disjoint)

                          - res-ex- (_? keys)

                         balance- {utxo  keys } {utxo  keys} $ flip res-ex-disjoint 
                      cbalance (utxo  keys ) + cbalance (utxo  keys)
                        
    where open IsEquivalence -isEquivalence renaming (trans to infixl 4 _-_)
