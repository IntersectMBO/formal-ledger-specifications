data ___,UTXO_ : UTxOEnv  UTxOState  Tx  UTxOState  Type where

  UTXO-inductive :
    let open Tx tx renaming (body to txb); open TxBody txb; open TxWitnesses wits
        open UTxOEnv  renaming (pparams to pp)
        open UTxOState s
        txouts = (mapValues txOutHash txouts)
        overhead = 160
    in
     txins                                 txins  refInputs  dom utxo
     txins  refInputs                     L.inInterval slot txvldt
     L.minfee pp utxo tx  txfee             (txrdmrs     L.collateralCheck pp tx utxo)
     consumed pp s txb  produced pp s txb   coin mint  0
     txsize  maxTxSize pp
     L.refScriptsSize utxo tx  pp .maxRefScriptSizePerTx

     [ (_ , txout)   txouts  ]
        inject ((overhead + L.utxoEntrySize txout) * coinsPerUTxOByte pp)  getValue txout
     [ (_ , txout)   txouts  ]
        serSize (getValue txout)  maxValSize pp
     [ (a , _)  range txouts ]
        Sum.All (const ) ( a  a .BootstrapAddr.attrsSize  64) a
     [ (a , _)  range txouts ]  netId a          NetworkId
     [ a  dom txwdrls ]          NetworkIdOf a    NetworkId
     txNetworkId ~ just NetworkId
     curTreasury ~ just treasury
       s  tx ,UTXOS s'
      
        s  tx ,UTXO s'
