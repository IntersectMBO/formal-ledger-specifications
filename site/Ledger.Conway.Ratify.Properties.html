module Ledger.Conway.Ratify.Properties (txs : _) (open TransactionStructure txs) where

open import Ledger.Conway.Gov txs
open import Ledger.Conway.GovernanceActions.Properties govStructure
open import Ledger.Conway.Enact govStructure
open import Ledger.Conway.Ratify txs

open Computational ... hiding (computeProof; completeness)

private
  module Implementation
     (s : RatifyState) (sig : GovActionID  _)
    (let gid , st = sig)
    where
    open RatifyState s
    open RatifyEnv ; open GovActionState st
    es'  = compute $\begin{pmatrix} \,\href{Ledger.Conway.Ratify.Properties.html#523}{\htmlId{638}{\htmlClass{Bound}{\text{gid}}}}\, \\ \,\href{Ledger.Conway.Ratify.html#7748}{\htmlId{644}{\htmlClass{Function}{\text{treasury}}}}\, \\ \,\href{Ledger.Conway.Ratify.html#7633}{\htmlId{655}{\htmlClass{Function}{\text{currentEpoch}}}}\, \end{pmatrix}$ es action
    acc? = accepted?  es st
    exp? = expired? currentEpoch st
    del? = delayed? (action .gaType) prevAction es delay

    opaque
      acceptConds? :  a  Dec (acceptConds  s a)
      acceptConds? _ = Dec-   accepted? _ _ _ 
         Dec-  Dec-   delayed? _ _ _ _     ComputationalDec'   .dec

    RATIFY-total : [ s' ]   s  sig ,RATIFY s'
    RATIFY-total
      with acceptConds? sig | exp?
    ... | yes p@(_ , _ , (_ , q)) | _ = -, RATIFY-Accept (p , q)
    ... | no p | no a = -, RATIFY-Continue (p , a)
    ... | no p | yes a = -, RATIFY-Reject (p , a)

    computeProof = success {Err = } RATIFY-total

    RATIFY-completeness :  s'    s  sig ,RATIFY s'  RATIFY-total .proj  s'
    RATIFY-completeness st (RATIFY-Accept (p , a)) with acceptConds? sig
    ... | no h = -elim (h p)
    ... | yes (_ , _ , _ , h) = cong ( st  $\begin{pmatrix} \,\href{Ledger.Conway.Ratify.Properties.html#1557}{\htmlId{1565}{\htmlClass{Bound}{\text{st}}}}\, \\ \,\htmlId{1571}{\htmlClass{Symbol}{\text{\_}}}\, \\ \,\htmlId{1575}{\htmlClass{Symbol}{\text{\_}}}\, \end{pmatrix}$) $ computationalrightUnique Computational-ENACT h a
    RATIFY-completeness s' (RATIFY-Reject (p , a))
      rewrite dec-no (acceptConds? _) p | dec-yes exp? a .proj = refl
    RATIFY-completeness s' (RATIFY-Continue (p , a))
      rewrite dec-no (acceptConds? _) p | dec-no exp? a = refl

    completeness = cong (success {Err = })  RATIFY-completeness

instance
  Computational-RATIFY : Computational ___,RATIFY_ 
  Computational-RATIFY = record {Implementation}

Computational-RATIFIES : Computational ___,RATIFIES_ 
Computational-RATIFIES = it

RATIFIES-total :  { s sig}  [ s' ]   s  sig ,RATIFIES s'
RATIFIES-total = ReflexiveTransitiveClosure-total (Implementation.RATIFY-total _ _ _)

RATIFIES-complete :  { s sig s'} 
    s  sig ,RATIFIES s'  RATIFIES-total {} {s} {sig} .proj  s'
RATIFIES-complete = computationalrightUnique Computational-RATIFIES (RATIFIES-total .proj)

opaque
  RATIFIES-total' :  { s sig}  [ s' ]   s  sig ,RATIFIES s'
  RATIFIES-total' = RATIFIES-total

  RATIFIES-complete' :  { s sig s'} 
      s  sig ,RATIFIES s'  RATIFIES-total' {} {s} {sig} .proj  s'
  RATIFIES-complete' = RATIFIES-complete
