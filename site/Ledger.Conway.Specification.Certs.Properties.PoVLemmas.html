getCoin-singleton :  _ : DecEq A  {(a , c) : A  Coin}  indexedSum' id  (a , c)   c

singletondom :   _ : DecEq A  (m : A  Coin) {(a , c) : A  Coin}
                 a  dom m  getCoin (m   (a , c) )  getCoin m
singletondom m {(a , c)} adom = -getCoin (m   (a , c) ) m (singleton-- {m = m} adom)

module _  ( indexedSum'- :  {A : Type}  _ : DecEq A  (m m' : A  Coin)
                               disjoint (dom m) (dom m')
                               getCoin (m  m')  getCoin m + getCoin m' )

  singletondom :   _ : DecEq A  (m : A  Coin) {(a , c) : A  Coin}
                    a  dom m  getCoin (m   (a , c) )  getCoin m + c
  singletondom m {(a , c)} adom = begin
    getCoin (m   a , c )
       indexedSum'- m  a , c 
         (  x y  adom (subst (_ dom m) (from -dom-singleton-pair y) x) ) 
    getCoin m + getCoin  a , c 
       cong (getCoin m +_) getCoin-singleton 
    getCoin m + c
      

  singleton0 :  _ : DecEq A   (m : A  Coin) {a : A}  getCoin (m   (a , 0) )  getCoin m
  singleton0 m {a} with a ? dom m
  ... | yes adom = singletondom m adom
  ... | no adom = trans (singletondom m adom) (+-identity (getCoin m))

    Lemma (CERTBASE rule preserves value).

    Informally.

    Let : CertEnv be a certificate environment, and let s, s' : CertState be certificate states such that s _ ,CERTBASE s'. Then, the value of s is equal to the value of s' plus the value of the withdrawals in . In other terms,

    getCoin s \(\) getCoin s' + getCoin ( .wdrls ).

    Formally.

        CERTBASE-pov : { : CertEnv} {s s' : CertState}
       [ a  dom (CertEnv.wdrls ) ] NetworkIdOf a  NetworkId
         s  _ ,CERTBASE s'
       getCoin s  getCoin s' + getCoin (CertEnv.wdrls )

    Proof.

        CERTBASE-pov  {   = }
                  {s   = cs}
                  {s'  = cs'}
                  validNetId
                  (CERT-base {pp}{vs}{e}{dreps}{wdrls} (_ , wdrlsCCrwds)) =
      let
        open DState (dState cs )
        open DState (dState cs') renaming (rewards to rewards')
        module        = IsEquivalence (-isEquivalence {Credential  Coin})
        wdrlsCC         = map (map RwdAddr.stake) (wdrls )
        zeroMap         = constMap (map RwdAddr.stake (dom wdrls)) 0
        rwds--decomp  = (rewards  dom wdrlsCC )  (rewards  dom wdrlsCC)
      in
        begin
          getCoin rewards
             -getCoin rwds--decomp rewards
                ( .trans (disjoint-- (disjoint-sym res-ex-disjoint))
                           (.trans -sym (res-ex- (_? dom wdrlsCC))) ) 
          getCoin rwds--decomp
             indexedSum'- (rewards  dom wdrlsCC ) (rewards  dom wdrlsCC)
                              (disjoint-sym res-ex-disjoint) 
          getCoin (rewards  dom wdrlsCC ) + getCoin (rewards  dom wdrlsCC )
             cong (getCoin (rewards  dom wdrlsCC ) +_)
               ( getCoin-cong (rewards  dom wdrlsCC) wdrlsCC (res-subset{m = rewards} wdrlsCCrwds) ) 
          getCoin (rewards  dom wdrlsCC ) + getCoin wdrlsCC
             cong (getCoin (rewards  dom wdrlsCC ) +_) (-getCoin (wdrls ) (injOn wdrls validNetId)) 
          getCoin (rewards  dom wdrlsCC ) + getCoin wdrls
             cong (_+ getCoin wdrls)
                ( begin
                  getCoin (zeroMap  rewards)
                     -getCoin (zeroMap  rewards) (zeroMap  (rewards  dom zeroMap ))
                                  (res-decomp zeroMap rewards) 
                  getCoin (zeroMap  (rewards  dom zeroMap ))
                     indexedSum'- zeroMap (rewards  dom zeroMap )
                                      (disjoint-sym res-comp-dom) 
                  getCoin zeroMap + getCoin (rewards  dom zeroMap )
                     cong ( u  u + getCoin (rewards  dom zeroMap )) sumConstZero 
                  0 + getCoin (rewards  (dom zeroMap) )
                     +-identity (getCoin (rewards  dom zeroMap )) 
                  getCoin (rewards  dom zeroMap )
                     -getCoin (rewards  dom zeroMap ) (rewards  dom wdrlsCC )
                       ( res-comp-cong
                         ( -Transitive (proj constMap-dom) (proj dom-mapmap-dom)
                         , -Transitive (proj dom-mapmap-dom) (proj constMap-dom) ) ) 
                  getCoin (rewards  dom wdrlsCC )
                     ) 
          getCoin (zeroMap  rewards) + getCoin wdrls

    Lemma (iteration of CERT rule preserves value).

    Informally. Let l be a list of DCerts, and let s, s be CertStates such that, starting with s and successively applying the CERT rule to with DCerts from the list l, we obtain s. Then, the value of s is equal to the value of s.

    Formally.

        sts-pov : { : CertEnv} {s s : CertState}
       ReflexiveTransitiveClosure {sts = ___,CERT_}  s l s
       getCoin s  getCoin s

    Proof.

        sts-pov (BS-base Id-nop) = refl
    sts-pov (BS-ind x xs) = trans (CERT-pov x) (sts-pov xs)
