rdptr : Tx ℓ → ScriptPurpose → Maybe RedeemerPtr
rdptr tx = λ where
  $\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#1232}{\htmlId{686}{\htmlClass{InductiveConstructor}{\text{Cert}}}}\,          \\ \,\href{Ledger.Dijkstra.Specification.Script.Validation.html#702}{\htmlId{702}{\htmlClass{Bound}{\text{h}}}}\, \end{pmatrix}$ → map (Cert           ,_) $ indexOfDCert          h (DCertsOf tx)
  $\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#1237}{\htmlId{778}{\htmlClass{InductiveConstructor}{\text{Reward}}}}\,        \\ \,\href{Ledger.Dijkstra.Specification.Script.Validation.html#794}{\htmlId{794}{\htmlClass{Bound}{\text{h}}}}\, \end{pmatrix}$ → map (Reward         ,_) $ indexOfRewardAddress  h (WithdrawalsOf tx)
  $\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#1227}{\htmlId{875}{\htmlClass{InductiveConstructor}{\text{Mint}}}}\,          \\ \,\href{Ledger.Dijkstra.Specification.Script.Validation.html#891}{\htmlId{891}{\htmlClass{Bound}{\text{h}}}}\, \end{pmatrix}$ → map (Mint           ,_) $ indexOfPolicyId       h (policies (MintedValueOf tx))
  $\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#1221}{\htmlId{983}{\htmlClass{InductiveConstructor}{\text{Spend}}}}\,         \\ \,\href{Ledger.Dijkstra.Specification.Script.Validation.html#999}{\htmlId{999}{\htmlClass{Bound}{\text{h}}}}\, \end{pmatrix}$ → map (Spend          ,_) $ indexOfTxIn           h (SpendInputsOf tx)
  $\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#1244}{\htmlId{1080}{\htmlClass{InductiveConstructor}{\text{Vote}}}}\,          \\ \,\href{Ledger.Dijkstra.Specification.Script.Validation.html#1096}{\htmlId{1096}{\htmlClass{Bound}{\text{h}}}}\, \end{pmatrix}$ → map (Vote           ,_) $ indexOfVote           h (map GovVote.voter (ListOfGovVotesOf tx))
  $\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#1249}{\htmlId{1200}{\htmlClass{InductiveConstructor}{\text{Propose}}}}\,       \\ \,\href{Ledger.Dijkstra.Specification.Script.Validation.html#1216}{\htmlId{1216}{\htmlClass{Bound}{\text{h}}}}\, \end{pmatrix}$ → map (Propose        ,_) $ indexOfProposal       h (ListOfGovProposalsOf tx)
  $\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#1257}{\htmlId{1304}{\htmlClass{InductiveConstructor}{\text{Guard}}}}\,         \\ \,\href{Ledger.Dijkstra.Specification.Script.Validation.html#1320}{\htmlId{1320}{\htmlClass{Bound}{\text{h}}}}\, \end{pmatrix}$ → map (Guard          ,_) $ indexOfGuard          h (setToList (GuardsOf tx))

indexedRdmrs : Tx ℓ → ScriptPurpose → Maybe (Redeemer × ExUnits)
indexedRdmrs tx sp = maybe (λ x → lookupᵐ? (RedeemersOf tx) x) nothing (rdptr tx sp)


getDatumSpend
  : Tx ℓ
  → UTxO
  → TxIn
  → Maybe Datum
getDatumSpend tx utxo₀ txin
  with lookupᵐ? utxo₀ txin
... | just (_ , _ , just d , _) =
  case d of λ where
    (inj₁ d) → just d
    (inj₂ h) → lookupHash h (DataOf tx)
... | _ = nothing


txInfo : (ℓ : TxLevel) → UTxO → Tx ℓ → TxInfo

txInfo TxLevelTop utxo tx =
  record  { realizedInputs      = utxo ∣ (SpendInputsOf tx)
          ; txOuts              = TxOutsOf tx
          ; txFee               = FeesOf? tx
          ; mint                = MintedValueOf tx
          ; txCerts             = DCertsOf tx
          ; txWithdrawals       = WithdrawalsOf tx
          ; txVldt              = ValidIntervalOf tx
          ; vkKey               = RequiredSignerHashesOf tx
          ; txGuards            = GuardsOf tx
          ; txData              = DataOf tx
          ; txId                = TxIdOf tx
          ; txInfoSubTxs        = nothing
          ; txDirectDeposits    = DirectDepositsOf tx
          ; txBalanceIntervals  = BalanceIntervalsOf tx
          }

txInfo TxLevelSub utxo tx =
  record  { realizedInputs      = utxo ∣ (TxBody.txIns txBody)
          ; txOuts              = TxBody.txOuts txBody
          ; txFee               = nothing
          ; mint                = TxBody.mint txBody
          ; txCerts             = TxBody.txCerts txBody
          ; txWithdrawals       = TxBody.txWithdrawals txBody
          ; txVldt              = TxBody.txVldt txBody
          ; vkKey               = TxBody.requiredSignerHashes txBody
          ; txGuards            = TxBody.txGuards txBody
          ; txData              = DataOf tx
          ; txId                = TxBody.txId txBody
          ; txInfoSubTxs        = nothing
          ; txDirectDeposits    = DirectDepositsOf tx
          ; txBalanceIntervals  = BalanceIntervalsOf tx
          } where open Tx tx

txInfoForPurpose : {ℓ : TxLevel} → UTxO → Tx ℓ → ScriptPurpose → TxInfo

-- subtransactions: never get subTx infos (even if the ScriptPurpose is Guard).
txInfoForPurpose {TxLevelSub} utxo tx _ = txInfo TxLevelSub utxo tx

-- top-level transactions:
txInfoForPurpose {TxLevelTop} utxo tx sp with sp
   -- · guard scripts see subTx infos
... | $\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#1257}{\htmlId{3754}{\htmlClass{InductiveConstructor}{\text{Guard}}}}\, \\ \,\htmlId{3762}{\htmlClass{Symbol}{\text{\_}}}\, \end{pmatrix}$ =  record base { txInfoSubTxs = just subInfos }
                 where
                 base : TxInfo
                 base = txInfo TxLevelTop utxo tx
                 subInfos : List SubTxInfo
                 subInfos = map (txInfo TxLevelSub utxo) (SubTransactionsOf tx)
   -- · other top-level scripts see no subTx infos
... | _ = txInfo TxLevelTop utxo tx

txOutToDataHash : TxOut → Maybe DataHash
txOutToDataHash (_ , _ , d , _) = d >>= isInj₂

credentialToP2Script
  : Credential → ℙ Script → Maybe P2Script
credentialToP2Script c scripts =
  do sh ← isScriptObj c
     s  ← lookupHash sh scripts
     toP2Script s

credentialToP1Script
  : Credential → ℙ Script → Maybe P1Script
credentialToP1Script c scripts =
  do sh ← isScriptObj c
     s  ← lookupHash sh scripts
     toP1Script s


credsNeeded : UTxO → Tx ℓ → ℙ (ScriptPurpose × Credential)
credsNeeded utxo tx =
    mapˢ        (λ (i , o) → ($\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#1221}{\htmlId{4678}{\htmlClass{InductiveConstructor}{\text{Spend}}}}\,  \\ \,\href{Ledger.Dijkstra.Specification.Script.Validation.html#4666}{\htmlId{4687}{\htmlClass{Bound}{\text{i}}}}\, \end{pmatrix}$ , payCred (proj₁ o)))     ((utxo ∣ (SpendInputsOf tx ∪ collateralInputs tx)) ˢ)
  ∪ mapˢ        (λ a       → ($\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#1237}{\htmlId{4805}{\htmlClass{InductiveConstructor}{\text{Reward}}}}\, \\ \,\href{Ledger.Dijkstra.Specification.Script.Validation.html#4792}{\htmlId{4814}{\htmlClass{Bound}{\text{a}}}}\, \end{pmatrix}$ , CredentialOf a))        (dom ∣ WithdrawalsOf tx ∣)
  ∪ mapPartial  (λ c       → (($\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#1232}{\htmlId{4906}{\htmlClass{InductiveConstructor}{\text{Cert}}}}\,  \\ \,\href{Ledger.Dijkstra.Specification.Script.Validation.html#4892}{\htmlId{4914}{\htmlClass{Bound}{\text{c}}}}\, \end{pmatrix}$ ,_) <$> cwitness c))      (fromList (DCertsOf tx))
  ∪ mapˢ        (λ x       → ($\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#1227}{\htmlId{5003}{\htmlClass{InductiveConstructor}{\text{Mint}}}}\,   \\ \,\href{Ledger.Dijkstra.Specification.Script.Validation.html#4990}{\htmlId{5012}{\htmlClass{Bound}{\text{x}}}}\, \end{pmatrix}$ , ScriptObj x))           (policies (MintedValueOf tx))
  ∪ mapˢ        (λ v       → ($\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#1244}{\htmlId{5106}{\htmlClass{InductiveConstructor}{\text{Vote}}}}\,   \\ \,\href{Ledger.Dijkstra.Specification.Script.Validation.html#5093}{\htmlId{5115}{\htmlClass{Bound}{\text{v}}}}\, \end{pmatrix}$ , govVoterCredential v))  (fromList (map GovVoterOf (ListOfGovVotesOf tx)))
  ∪ mapPartial  (λ p       →  if PolicyOf p
                                 then (λ {sh} → just ($\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#1249}{\htmlId{5297}{\htmlClass{InductiveConstructor}{\text{Propose}}}}\, \\ \,\href{Ledger.Dijkstra.Specification.Script.Validation.html#5216}{\htmlId{5307}{\htmlClass{Bound}{\text{p}}}}\, \end{pmatrix}$ , ScriptObj sh))
                                 else nothing)                           (fromList (ListOfGovProposalsOf tx))
  ∪ mapˢ        (λ c       → ($\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#1257}{\htmlId{5472}{\htmlClass{InductiveConstructor}{\text{Guard}}}}\, \\ \,\href{Ledger.Dijkstra.Specification.Script.Validation.html#5459}{\htmlId{5480}{\htmlClass{Bound}{\text{c}}}}\, \end{pmatrix}$ , c))                      (GuardsOf tx)

  where
    collateralInputs : Tx ℓ → ℙ TxIn
    collateralInputs {TxLevelTop} tx = CollateralInputsOf tx
    collateralInputs {TxLevelSub} tx = ∅


opaque


  collectP2ScriptsWithContext
    : PParams
    → Tx ℓ
    → UTxO
    → ℙ Script
    → List (P2Script × List Data × ExUnits × CostModel)
  collectP2ScriptsWithContext pp tx utxo allScripts
    = concat (setToList (mapˢ toScript (credsNeeded utxo tx)))
    where
      context : ScriptPurpose → Data
      context sp = valContext (txInfoForPurpose utxo tx sp) sp

      getScript : Credential → ScriptPurpose → Maybe (P2Script × Redeemer × ExUnits × CostModel)
      getScript c sp = do
             script               ← isScriptObj c >>= λ sh → lookupHash sh allScripts >>= toP2Script
             (reedemer , exUnits) ← indexedRdmrs tx sp
             costModel            ← lookupᵐ? (PParams.costmdls pp) (language script)
             return (script , reedemer , exUnits , costModel)

      assembleData : Redeemer → ScriptPurpose → List (List Data)
      assembleData redeemer sp@($\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Transaction.html#1221}{\htmlId{6575}{\htmlClass{InductiveConstructor}{\text{Spend}}}}\, \\ \,\href{Ledger.Dijkstra.Specification.Script.Validation.html#6583}{\htmlId{6583}{\htmlClass{Bound}{\text{txin}}}}\, \end{pmatrix}$)
        = [ fromMaybe (getDatumSpend tx utxo txin) ++ (redeemer ∷ [ context sp ]) ]
      assembleData redeemer sp           = [ redeemer ∷ [ context sp ] ]

      toScript
        : ScriptPurpose × Credential
        → List (P2Script × List Data × ExUnits × CostModel)
      toScript (sp , c) =
        do (script , reedemer , exUnits , costModel) ← fromMaybe (getScript c sp)
           data′ ← assembleData reedemer sp
           [ (script , data′ , exUnits , costModel) ]

evalP2Scripts : List (P2Script × List Data × ExUnits × CostModel) → Bool
evalP2Scripts = all (λ (s , d , eu , cm) → ¿ validP2Script cm d eu s ¿ᵇ)
