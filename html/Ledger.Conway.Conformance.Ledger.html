record LState : Type where
  constructor ⟦_,_,_⟧ˡ
  field
    utxoSt     : UTxOState
    govSt      : GovState
    certState  : CertState

instance
  unquoteDecl HasCast-LState = derive-HasCast
    [ (quote LState , HasCast-LState) ]

private variable
  Γ : LEnv
  s s' s'' : LState
  utxoSt' : UTxOState
  govSt' : GovState
  certState' : CertState
  tx : Tx

open UTxOState

data

  _⊢_⇀⦇_,LEDGER⦈_ : LEnv → LState → Tx → LState → Type

  where

  LEDGER-V :
    let open LState s; txb = tx .body; open TxBody txb; open LEnv Γ
        open CertState certState; open DState dState
        utxoSt'' = record utxoSt' { deposits = updateDeposits pparams txb (deposits utxoSt') }
     in
    ∙  isValid tx ≡ true
    ∙  record { LEnv Γ } ⊢ utxoSt ⇀⦇ tx ,UTXOW⦈ utxoSt'
    ∙  $\begin{pmatrix} \,\href{Ledger.Core.Specification.Epoch.html#858}{\htmlId{1506}{\htmlClass{Function}{\text{epoch}}}}\, \,\href{Ledger.Conway.Specification.Ledger.html#761}{\htmlId{1512}{\htmlClass{Function}{\text{slot}}}}\, \\ \,\href{Ledger.Conway.Specification.Ledger.html#819}{\htmlId{1519}{\htmlClass{Function}{\text{pparams}}}}\, \\ \,\href{Ledger.Conway.Specification.Transaction.html#791}{\htmlId{1529}{\htmlClass{Function}{\text{txGovVotes}}}}\, \\ \,\href{Ledger.Conway.Specification.Transaction.html#624}{\htmlId{1542}{\htmlClass{Function}{\text{txWithdrawals}}}}\, \\ \,\href{Ledger.Conway.Specification.Ledger.html#3268}{\htmlId{1558}{\htmlClass{Function}{\text{allColdCreds}}}}\, \,\href{Ledger.Conway.Conformance.Ledger.html#820}{\htmlId{1571}{\htmlClass{Function}{\text{govSt}}}}\, \,\href{Ledger.Conway.Specification.Ledger.html#845}{\htmlId{1577}{\htmlClass{Function}{\text{enactState}}}}\, \end{pmatrix}$ ⊢ certState ⇀⦇ txCerts ,CERTS⦈ certState'
    ∙  $\begin{pmatrix} \,\href{Ledger.Conway.Specification.Transaction.html#522}{\htmlId{1641}{\htmlClass{Function}{\text{txId}}}}\, \\ \,\href{Ledger.Core.Specification.Epoch.html#858}{\htmlId{1648}{\htmlClass{Function}{\text{epoch}}}}\, \,\href{Ledger.Conway.Specification.Ledger.html#761}{\htmlId{1654}{\htmlClass{Function}{\text{slot}}}}\, \\ \,\href{Ledger.Conway.Specification.Ledger.html#819}{\htmlId{1661}{\htmlClass{Function}{\text{pparams}}}}\, \\ \,\href{Ledger.Conway.Specification.Ledger.html#784}{\htmlId{1671}{\htmlClass{Function}{\text{ppolicy}}}}\, \\ \,\href{Ledger.Conway.Specification.Ledger.html#845}{\htmlId{1681}{\htmlClass{Function}{\text{enactState}}}}\, \\  \,\href{Ledger.Conway.Conformance.Ledger.html#1058}{\htmlId{1695}{\htmlClass{Generalizable}{\text{certState'}}}}\, \\ \,\href{Class.IsSet.html#916}{\htmlId{1708}{\htmlClass{Function}{\text{dom}}}}\,
    \,\href{Ledger.Conway.Conformance.Certs.html#731}{\htmlId{1716}{\htmlClass{Function}{\text{rewards}}}}\, \end{pmatrix}$ ⊢ govSt ⇀⦇ txgov txb ,GOVS⦈ govSt'
       ────────────────────────────────
       Γ ⊢ s ⇀⦇ tx ,LEDGER⦈ $\begin{pmatrix} \,\href{Ledger.Conway.Conformance.Ledger.html#1321}{\htmlId{1831}{\htmlClass{Bound}{\text{utxoSt''}}}}\, \\ \,\href{Ledger.Conway.Conformance.Ledger.html#1038}{\htmlId{1842}{\htmlClass{Generalizable}{\text{govSt'}}}}\, \\ \,\href{Ledger.Conway.Conformance.Ledger.html#1058}{\htmlId{1851}{\htmlClass{Generalizable}{\text{certState'}}}}\, \end{pmatrix}$


  LEDGER-I : let open LState s; txb = tx .body; open TxBody txb; open LEnv Γ in
    ∙  isValid tx ≡ false
    ∙  record { LEnv Γ } ⊢ utxoSt ⇀⦇ tx ,UTXOW⦈ utxoSt'
       ────────────────────────────────
       Γ ⊢ s ⇀⦇ tx ,LEDGER⦈ $\begin{pmatrix} \,\href{Ledger.Conway.Conformance.Ledger.html#1016}{\htmlId{2098}{\htmlClass{Generalizable}{\text{utxoSt'}}}}\, \\ \,\href{Ledger.Conway.Conformance.Ledger.html#820}{\htmlId{2108}{\htmlClass{Function}{\text{govSt}}}}\, \\ \,\href{Ledger.Conway.Conformance.Ledger.html#846}{\htmlId{2116}{\htmlClass{Function}{\text{certState}}}}\, \end{pmatrix}$

pattern LEDGER-V⋯ w x y z = LEDGER-V (w , x , y , z)
pattern LEDGER-I⋯ y z     = LEDGER-I (y , z)

_⊢_⇀⦇_,LEDGERS⦈_ : LEnv → LState → List Tx → LState → Type
_⊢_⇀⦇_,LEDGERS⦈_ = ReflexiveTransitiveClosure {sts = _⊢_⇀⦇_,LEDGER⦈_}
