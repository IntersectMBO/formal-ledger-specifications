GovVoter' : Type
GovVoter' = GovRole × Credential

instance
  mkGovVoter' : Convertible GovVoter GovVoter'
  mkGovVoter' .to $\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Gov.Actions.html#427}{\htmlId{955}{\htmlClass{InductiveConstructor}{\text{CC}}}}\,   \\ \,\href{Ledger.Dijkstra.Foreign.Gov.Core.html#962}{\htmlId{962}{\htmlClass{Bound}{\text{c}}}}\, \end{pmatrix}$ = CC   , c
  mkGovVoter' .to $\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Gov.Actions.html#430}{\htmlId{999}{\htmlClass{InductiveConstructor}{\text{DRep}}}}\, \\ \,\href{Ledger.Dijkstra.Foreign.Gov.Core.html#1006}{\htmlId{1006}{\htmlClass{Bound}{\text{c}}}}\, \end{pmatrix}$ = DRep , c
  mkGovVoter' .to $\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Gov.Actions.html#435}{\htmlId{1043}{\htmlClass{InductiveConstructor}{\text{SPO}}}}\,  \\ \,\href{Ledger.Dijkstra.Foreign.Gov.Core.html#1050}{\htmlId{1050}{\htmlClass{Bound}{\text{c}}}}\, \end{pmatrix}$ = SPO  , KeyHashObj c
  mkGovVoter' .from (CC   , c)   = $\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Gov.Actions.html#427}{\htmlId{1115}{\htmlClass{InductiveConstructor}{\text{CC}}}}\,   \\ \,\href{Ledger.Dijkstra.Foreign.Gov.Core.html#1106}{\htmlId{1122}{\htmlClass{Bound}{\text{c}}}}\, \end{pmatrix}$
  mkGovVoter' .from (DRep , c)   = $\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Gov.Actions.html#430}{\htmlId{1165}{\htmlClass{InductiveConstructor}{\text{DRep}}}}\, \\ \,\href{Ledger.Dijkstra.Foreign.Gov.Core.html#1156}{\htmlId{1172}{\htmlClass{Bound}{\text{c}}}}\, \end{pmatrix}$
  mkGovVoter' .from (SPO  , c)   =
    case c of λ where
      (KeyHashObj kh) → $\begin{pmatrix} \,\href{Ledger.Dijkstra.Specification.Gov.Actions.html#435}{\htmlId{1261}{\htmlClass{InductiveConstructor}{\text{SPO}}}}\, \\ \,\href{Ledger.Dijkstra.Foreign.Gov.Core.html#1253}{\htmlId{1267}{\htmlClass{Bound}{\text{kh}}}}\, \end{pmatrix}$
      (ScriptObj _)   → error "mkGovVoter: Converting from SPO with ScriptObj credential"

  HsTy-GovVoter = MkHsType GovVoter (HsType GovVoter')
  Conv-GovVoter : Convertible GovVoter (HsType GovVoter')
  Conv-GovVoter = mkGovVoter' ⨾ Convertible-Pair

unquoteDecl = do
  hsTypeAlias GovVoter

record GovVote' : Type where
  field
    gid         : GovActionID
    voter       : GovVoter'
    vote        : Vote
    anchor      : Maybe Anchor

instance
  mkGovVote' : Convertible GovVote GovVote'
  mkGovVote' = λ where
    .to v   → let module v = GovVote v in record { gid = v.gid ; voter = to v.voter  ; vote = v.vote ; anchor = v.anchor }
    .from v → let module v = GovVote' v in record { gid = v.gid ; voter = from v.voter ; vote = v.vote ; anchor = v.anchor }

  HsTy-GovVote' = autoHsType GovVote' ⊣ withConstructor "MkGovVote"
                                      • withName "GovVote"
  Conv-GovVote' = autoConvert GovVote'

  HsTy-GovVote = MkHsType GovVote (HsType GovVote')
  Conv-GovVote = mkGovVote' ⨾ Conv-GovVote'
