-- ** Smart constructors

-- arguments
pattern hArg? = hArg unknown
pattern vArg? = vArg unknown
pattern iArg? = iArg unknown

-- variables
pattern ♯ n = var n []
pattern ♯_⟦_⟧ n x = var n (vArg x ∷ [])
pattern ♯_⟦_∣_⟧ n x y = var n (vArg x ∷ vArg y ∷ [])

-- patterns
pattern `_ x = Pattern.var x
pattern `∅   = Pattern.absurd 0

-- clauses
pattern ⟦⦅_⦆∅⟧ tel = absurd-clause tel (vArg `∅ ∷ [])
pattern ⟦∅⟧ = absurd-clause [] (vArg `∅ ∷ [])

pattern ⟦⇒_⟧ k = clause [] [] k
pattern ⟦⦅_⦆⇒_⟧ tel k = clause tel [] k

pattern ⟦_⇒_⟧ x k = clause [] (vArg x ∷ []) k
pattern ⟦_⦅_⦆⇒_⟧ x tel k = clause tel (vArg x ∷ []) k

pattern ⟦_∣_⇒_⟧ x y k = clause [] (vArg x ∷ vArg y ∷ []) k
pattern ⟦_∣_⦅_⦆⇒_⟧ x y tel k = clause tel (vArg x ∷ vArg y ∷ []) k

pattern ⟦_∣_∣_⇒_⟧ x y z k = Clause.clause [] (vArg x ∷ vArg y ∷ vArg z ∷ []) k
pattern ⟦_∣_∣_⦅_⦆⇒_⟧ x y z tel k = Clause.clause tel (vArg x ∷ vArg y ∷ vArg z ∷ []) k

-- lambdas
pattern `λ_⇒_       x     k = lam visible (abs x k)
pattern `λ⟦_∣_⇒_⟧   x y   k = `λ x ⇒ `λ y ⇒ k
pattern `λ⟦_∣_∣_⇒_⟧ x y z k = `λ x ⇒ `λ y ⇒ `λ z ⇒ k

pattern `λ⟅_⟆⇒_     x k = lam hidden (abs x k)
pattern `λ⦃_⦄⇒_     x k = lam instance′ (abs x k)

pattern `λ⦅_⦆∅ tel = pat-lam (⟦⦅ tel ⦆∅⟧ ∷ []) []
pattern `λ∅ = pat-lam (⟦∅⟧ ∷ []) []

pattern `λ⟦_⇒_⟧ p k = pat-lam ($\begin{pmatrix} \,\href{Reflection.Syntax.html#2720}{\htmlId{2720}{\htmlClass{Bound}{\text{p}}}}\, \,\href{Reflection.Syntax.html#1952}{\htmlId{2722}{\htmlClass{InductiveConstructor Operator}{\text{⇒}}}}\, \,\href{Reflection.Syntax.html#2724}{\htmlId{2724}{\htmlClass{Bound}{\text{k}}}}\, \end{pmatrix}$ ∷ []) []
pattern `λ⟦_⦅_⦆⇒_⟧ p tel k = pat-lam ($\begin{pmatrix} \,\href{Reflection.Syntax.html#2777}{\htmlId{2777}{\htmlClass{Bound}{\text{p}}}}\, \,\href{Reflection.Syntax.html#1998}{\htmlId{2779}{\htmlClass{InductiveConstructor Operator}{\text{⦅}}}}\, \,\href{Reflection.Syntax.html#2781}{\htmlId{2781}{\htmlClass{Bound}{\text{tel}}}}\, \,\href{Reflection.Syntax.html#1998}{\htmlId{2785}{\htmlClass{InductiveConstructor Operator}{\text{⦆⇒}}}}\, \,\href{Reflection.Syntax.html#2788}{\htmlId{2788}{\htmlClass{Bound}{\text{k}}}}\, \end{pmatrix}$ ∷ []) []

pattern `λ⟦_⇒_∣_⇒_⟧ p₁ k₁ p₂ k₂ = pat-lam ($\begin{pmatrix} \,\href{Reflection.Syntax.html#2847}{\htmlId{2847}{\htmlClass{Bound}{\text{p₁}}}}\, \,\href{Reflection.Syntax.html#1952}{\htmlId{2850}{\htmlClass{InductiveConstructor Operator}{\text{⇒}}}}\, \,\href{Reflection.Syntax.html#2852}{\htmlId{2852}{\htmlClass{Bound}{\text{k₁}}}}\, \end{pmatrix}$ ∷ $\begin{pmatrix} \,\href{Reflection.Syntax.html#2861}{\htmlId{2861}{\htmlClass{Bound}{\text{p₂}}}}\, \,\href{Reflection.Syntax.html#1952}{\htmlId{2864}{\htmlClass{InductiveConstructor Operator}{\text{⇒}}}}\, \,\href{Reflection.Syntax.html#2866}{\htmlId{2866}{\htmlClass{Bound}{\text{k₂}}}}\, \end{pmatrix}$ ∷ []) []
pattern `λ⟦_⦅_⦆⇒_∣_⦅_⦆⇒_⟧ p₁ tel₁ k₁ p₂ tel₂ k₂ = pat-lam ($\begin{pmatrix} \,\href{Reflection.Syntax.html#2941}{\htmlId{2941}{\htmlClass{Bound}{\text{p₁}}}}\, \,\href{Reflection.Syntax.html#1998}{\htmlId{2944}{\htmlClass{InductiveConstructor Operator}{\text{⦅}}}}\, \,\href{Reflection.Syntax.html#2946}{\htmlId{2946}{\htmlClass{Bound}{\text{tel₁}}}}\, \,\href{Reflection.Syntax.html#1998}{\htmlId{2951}{\htmlClass{InductiveConstructor Operator}{\text{⦆⇒}}}}\, \,\href{Reflection.Syntax.html#2954}{\htmlId{2954}{\htmlClass{Bound}{\text{k₁}}}}\, \end{pmatrix}$ ∷ $\begin{pmatrix} \,\href{Reflection.Syntax.html#2963}{\htmlId{2963}{\htmlClass{Bound}{\text{p₂}}}}\, \,\href{Reflection.Syntax.html#1998}{\htmlId{2966}{\htmlClass{InductiveConstructor Operator}{\text{⦅}}}}\, \,\href{Reflection.Syntax.html#2968}{\htmlId{2968}{\htmlClass{Bound}{\text{tel₂}}}}\, \,\href{Reflection.Syntax.html#1998}{\htmlId{2973}{\htmlClass{InductiveConstructor Operator}{\text{⦆⇒}}}}\, \,\href{Reflection.Syntax.html#2976}{\htmlId{2976}{\htmlClass{Bound}{\text{k₂}}}}\, \end{pmatrix}$ ∷ []) []

-- function application
pattern _∙ n = def n []
pattern _∙⟦_⟧ n x = def n (vArg x ∷ [])
pattern _∙⟦_∣_⟧ n x y = def n (vArg x ∷ vArg y ∷ [])
pattern _∙⟦_∣_∣_⟧ n x y z = def n (vArg x ∷ vArg y ∷ vArg z ∷ [])
pattern _∙⟦_∣_∣_∣_⟧ n x y z w = def n (vArg x ∷ vArg y ∷ vArg z ∷ vArg w ∷ [])

pattern _◆ n = con n []
pattern _◆⟦_⟧ n x = con n (vArg x ∷ [])
pattern _◆⟦_∣_⟧ n x y = con n (vArg x ∷ vArg y ∷ [])

pattern _◇ n = Pattern.con n []
pattern _◇⟦_⟧ n x = Pattern.con n (vArg x ∷ [])
pattern _◇⟦_∣_⟧ n x y = Pattern.con n (vArg x ∷ vArg y ∷ [])

pattern `Set = sort (set (quote 0ℓ ∙))
