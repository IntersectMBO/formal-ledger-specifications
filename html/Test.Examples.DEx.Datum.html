data Label : Set where

  Always : ℚ -> ℕ -> Label

  unquoteDecl DecEq-Label = derive-DecEq
    ((quote Label , DecEq-Label) ∷ [])

data Input : Set where
  Update   : ℕ -> ℚ -> Input
  Exchange : ℕ -> ℕ -> Input
  Close    : Input
instance
  unquoteDecl DecEq-Input = derive-DecEq
    ((quote Input , DecEq-Input) ∷ [])
