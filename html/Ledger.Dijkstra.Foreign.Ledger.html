                                                • fieldPrefix "sle"

                                          • fieldPrefix "le"

                                              • fieldPrefix "ls"

ledger-step : HsType (LedgerEnv → LedgerState → Tx TxLevelTop → ComputationResult String LedgerState)
ledger-step = to (compute Computational-LEDGER)
