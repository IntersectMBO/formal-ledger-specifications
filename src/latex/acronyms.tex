%% This is the file acronyms.tex that we use to collect all acronyms and
%% descriptions of definitions of variable names.
%%
%% Example acronym definition:  \acrodef{iohk}[IOHK]Input-output Hong Kong}
%% Usage examples:
%%                \acs{iohk}  - [s]hort version of the acronym
%%                \acl{iohk}  - [l]ong version---i.e., the phrase for which the acronym stands
%%                \acp{iohk}  - [p]lural form of the acronym
%%                \acfp{iohk} -- [f]ull version in [p]lural form
%%                \acsp{iohk} -- [s]hort version in [p]lural form
%%                \aclp{iohk} -- [l]ong version in [p]lural form
%%                \acfi{iohk} -- [f]ull version [i]talicized
%%                \acsu{iohk} -- [s]hort version, and acronym is marked as having been [u]sed at least once
%%                \aclu{iohk} -- [l]ong version, and acronym is marked as having been [u]sed at least once
%%
%% We must use the lower case `iohk` in each of the above acronym command examples simply becuase that's
%% the name that we assigned to the acronym when we defined it (`\acrodef{iohk}[IOHK]{Input-output Hong Kong}`).
%% We could also have named it `foo` (`\acrodef{foo}{IOHK}{Input-output Hong Kong}`) and then, in
%% each of the commands above, we would replace `iohk` with `foo`; e.g., `\acs{iohk}` would become `\acs{foo}`
%% ...but the result would be the same.

%% IOHK acronyms
